## World State Date calls ##
macro = {
	description = "Print the WS Date"
	definition = "GetWorldstateDate(Worldstate)" #"$DAY$ $MONTH$, $YEAR$"
	replace_with = "Concatenate(Concatenate(Concatenate( GetWorldStateDayInt(WorldStateName), ' '), Concatenate( GetWorldStateMonth(WorldStateName), ', ')), GetWorldStateYearInt(WorldStateName))"
}
macro = {
	description = "Print the shorter WS Date"
	definition = "GetWorldstateDateShort(Worldstate)" #"$DAY$ $MONTH_SHORT$ $YEAR$"
	replace_with = "Concatenate(Concatenate(Concatenate( GetWorldStateDayInt(WorldStateName), ' '), Concatenate( GetWorldStateMonthShort(WorldStateName), ' ')), GetWorldStateYearInt(WorldStateName))"
}
macro = {
	description = "Print the shorter WS Date without the year"
	definition = "GetWorldstateDateShortNoYear(Worldstate)" #"$DAY$ $MONTH_SHORT$"
	replace_with = "Concatenate(Concatenate( GetWorldStateDayInt(WorldStateName), ' '), Concatenate( GetWorldStateMonthShort(WorldStateName), ' '))"
}
macro = {
	description = "Print the shorter WS Date without the year"
	definition = "GetWorldstateDateLong(Worldstate)" #"$DAY|O$ of $MONTH$, $YEAR$ IS"
	replace_with = "Concatenate(Concatenate(Concatenate( GetWorldStateDayInt(WorldStateName), ' of '), Concatenate( GetWorldStateMonth(WorldStateName), ', ')), GetWorldStateYear(WorldStateName))"
}

sjal_nicknames.1000 = { #Shown new nickname, chance to react in various ways
	type = character_event
	content_source = dlc_gh
	title = sjal_nicknames.1.t
	desc = sjal_nicknames.1.desc
	left_portrait = {
		character = root
		triggered_animation = {
			trigger = {
				has_bad_nickname = yes
			}
  			animation = shame
  		}
	}
	theme = dynasty
	trigger = {
		has_any_nickname = yes
	}

	option = { #Accept nickname
		name = sjal_nicknames.1.a #Memorable
		trigger = {
			has_bad_nickname = no
		}
		add_prestige = minor_prestige_gain
		ai_chance = {
			base = 1
		}
	}
	
	option = { #Accept bad nickname
		name = {
			trigger = {
				OR = {
					has_trait = humble
					has_trait = trusting
					has_trait = forgiving
					has_trait = content
					has_trait = calm
				}
				NOR = {
					has_trait = arrogant
				}
			}
			text = sjal_nicknames.1.a_bad_1 #I guess it fits me...
		}
		name = sjal_nicknames.1.a_bad_2 #What?! These absolute...
		trigger = {
			has_bad_nickname = yes
		}
		stress_impact = {
			base = medium_stress_impact_gain
			humble = minor_stress_impact_gain
			arrogant = major_stress_impact_gain
		}
		add_prestige = minor_prestige_loss
		ai_chance = {
			base = 1
		}
	}

	option = { #Reject nickname
		name = {
			trigger = {
				OR = {
					NOR = {
						has_trait = humble
						has_trait = trusting
						has_trait = forgiving
						has_trait = content
						has_trait = calm
					}
					has_trait = arrogant
				}
			}
			text = sjal_nicknames.1.b_1 #I will not be remembered this way!
		}
		name = sjal_nicknames.1.b_2 #They can do better than that
		trigger = {
			has_bad_nickname = no
		}
		random_list = {
			50 = {
				desc = sjal_nicknames.1.b.success
				remove_nickname = yes
			}
			25 = {
				desc = sjal_nicknames.1.b.failure
				add_stress = minor_stress_gain
			}
			25 = {
				desc = sjal_nicknames.1.b.new_nickname
				trigger_event = {
					on_action = sjal_general_nickname_selection
				}
			}
		}
		
		ai_chance = {
			base = 0
		}
	}

	option = { #Reject bad nickname
		name = sjal_nicknames.1.b_bad #I will not be remembered this way!
		trigger = {
			has_bad_nickname = yes
		}
		random_list = {
			50 = {
				desc = sjal_nicknames.1.b.success
				remove_nickname = yes
			}
			25 = {
				desc = sjal_nicknames.1.b.failure
				add_stress = medium_stress_impact_gain
				add_prestige = medium_prestige_loss
			}
			25 = {
				desc = sjal_nicknames.1.b.new_nickname
				trigger_event = {
					on_action = sjal_general_nickname_selection
				}
			}
		}
		ai_chance = {
			base = 2
		}
	}

	option = { #Take clan-name
		name = {
			trigger = {
				has_bad_nickname = yes
			}
			text = sjal_nicknames.1.c_1 #I will wear it with honor
		}
		name = sjal_nicknames.1.c_2 #My kin will remember me
		add_internal_flag = special
		trigger = {
			sjal_nickname_has_translation = yes
		}
		
		hidden_effect = {
			if = {
				limit = {
					exists = house
					exists = house.house_head
					is_house_head = no
				}
				found_cadet_house_decision_effect = {
					CHARACTER = root
					PRESTIGE = 0
				}
			}
		}
		translate_sjalvolki_nickname_effect = yes
		if = {
			limit = {
				has_bad_nickname = yes
			}
			add_stress = minor_stress_impact_gain
			add_prestige = medium_prestige_loss
		}
		else = {
			add_prestige = minor_prestige_gain
		}
		dynasty = { add_dynasty_prestige = minor_dynasty_prestige_gain }
		
		ai_chance = {
			base = 100
		}
	}
}

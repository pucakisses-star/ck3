	type = character_event
	content_source = dlc_gh
	theme = travel
	hidden = yes
	orphan = yes
	trigger = {
		is_available_travelling_adult = yes
		is_location_valid_for_travel_event_on_land = yes
		not = {
			has_character_flag = gh_travel_event_0001
		}
		# current_date < 1300.1.1
	}

	immediate = {
		add_character_flag = {
			flag = gh_travel_event_0001
			years = 5
		}
		if = {
			limit = {
				employs_court_position = travel_leader_court_position
			}
			random_court_position_holder = {
				limit = { has_court_position = travel_leader_court_position }
				save_scope_as = listener_1
			}
		}
		else_if = {
			limit = { exists = primary_spouse }
			primary_spouse = { save_scope_as = listener_1 }
		}
		else_if = {
			limit = { exists = cp:councillor_chancellor }
			cp:councillor_chancellor = { save_scope_as = listener_1 }
		}
		else = {
			random_courtier = {
				limit = { is_adult = yes }
				save_scope_as = listener_1
			}
		}
		#journey_event_setup_effect = yes

		# Create the character

		random_list = {
			1 = { # Elysian Pass
				random_list = {
					1 = { # Young
						random_list = {
							1 = {
								save_scope_value_as = {
									name = journey_character
									value = flag:elysia_aversarian_young_legionnaire
								}
								journey_event_character_effect = { TEMPLATE = journey_aversarian_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }
							}

								save_scope_value_as = {
									name = journey_character
									value = flag:elysia_aversarian_young_cavalry
								}

								save_scope_value_as = {
									name = journey_character
									value = flag:elysia_aversarian_young_deserter
								}
								journey_event_character_effect = { TEMPLATE = journey_aversarian_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }
							}
							1 = {
								save_scope_value_as = {
									name = journey_character
									value = flag:elysia_aversarian_young_slave
								}
								journey_event_character_effect = { TEMPLATE = journey_aversarian_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }
							}
							1 = {
								save_scope_value_as = {
									name = journey_character
									value = flag:elysia_aversarian_young_proud
								}
								journey_event_character_effect = { TEMPLATE = journey_aversarian_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }
							}
						}
					}
					1 = { # Old
						random_list = {
							1 = {
								save_scope_value_as = {
									name = journey_character
									value = flag:elysia_aversarian_old_lieutenant
								}
								journey_event_character_effect = { TEMPLATE = journey_aversarian_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }
							}
							1 = {
								save_scope_value_as = {
									name = journey_character
									value = flag:elysia_aversarian_old_proud
								}
								journey_event_character_effect = { TEMPLATE = journey_aversarian_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }

								save_scope_value_as = {
									name = journey_character

								}

									name = journey_character

								}
								journey_event_character_effect = { TEMPLATE = journey_aversarian_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }
							}
						}
					}
				}
			}
			1 = { # Kalathipsomi
				random_list = {
					1 = {
						save_scope_value_as = {
							name = journey_character
							value = flag:kalathipsomi_veteran_sjalvolki_young_proud
						}
						journey_event_character_effect = { TEMPLATE = journey_sjalvolki_veteran AGE_LOW = elysia_bottom_age_value AGE_HIGH = elysia_top_age_value }
					}
					1 = {
						save_scope_value_as = {
							name = journey_character

gh_journey_event.0002 = { # You decide whether or not to rob the hell out of the wanderer
	type = character_event
	content_source = dlc_gh
	title = gh_journey_event.0002.t
	desc = {
		desc = gh_journey_event.0002.desc
	}
	theme = travel
	left_portrait = {
		character = root
		animation = admiration
	}
	right_portrait = {
		character = scope:generated_journey_character
		animation = happiness
	}
	lower_center_portrait = {
		character = scope:listener_1
	}

	option = {
		name = gh_journey_event.0002.a
		#trigger_event = {
		#	id = gh_journey_event.0003
		#}
	}
	option = {
		name = gh_journey_event.0002.b
		mp_resume_travel_plan = yes
		if = {
			limit = { scope:generated_journey_character = { is_alive = yes } }
			scope:generated_journey_character = { death = { death_reason = death_vanished } }
		}
	}
	option = {
		name = gh_journey_event.0002.c
		add_gold = 10
	}
}

gh_travel_event.1001 = { # Cheer up your party with a song
	type = character_event
	content_source = dlc_gh
	title = gh_travel_event.1001.t
	desc = gh_travel_event.1001.desc
	theme = travel
	left_portrait = {
		character = root
		animation = admiration
	}
	right_portrait = {
		character = scope:listener_1
		animation = happiness
	}

	cooldown = { years = 10 }

	trigger = {
		is_available_travelling_adult = yes
		is_location_valid_for_travel_event_on_land = yes
	}



	immediate = {
		if = {
			limit = {
				employs_court_position = travel_leader_court_position
			}
			random_court_position_holder = {
				limit = { has_court_position = travel_leader_court_position }
				save_scope_as = listener_1
			}
		}
		else_if = {
			limit = { exists = primary_spouse }
			primary_spouse = { save_scope_as = listener_1 }
		}
		else_if = {
			limit = { exists = cp:councillor_chancellor }
			cp:councillor_chancellor = { save_scope_as = listener_1 }
		}
		else = {
			random_courtier = {
				limit = { is_adult = yes }
				save_scope_as = listener_1
			}
		}


	}



	option = {
		name = gh_travel_event.1001.a
		duel = {
			skill = diplomacy
			value = low_skill_rating
			20 = { #success! The men are in high spirits
				compare_modifier = {
					value = scope:duel_value
					multiplier = 1
				}
				desc = gh_travel_event.1001.a.exceptionalsuccess
				send_interface_toast = {
					title = gh_travel_event.1001.a.exceptionalsuccess
					add_prestige = minor_prestige_gain
				}
			}
			8 = { #Well they like the drinks at least
				compare_modifier = {
					value = scope:duel_value
				}
				desc = gh_travel_event.1001.a.success
				current_travel_plan = { delay_travel_plan = { days = 5 } }
			}
			5 = { #You ruin the evening
				compare_modifier = {
					value = scope:duel_value
				}
				desc = gh_travel_event.1001.a.failure
				current_travel_plan = { delay_travel_plan = { days = 25 } }
			}
		}
		stress_impact = {
			impatient = miniscule_stress_impact_gain
			callous = miniscule_stress_impact_gain
			gregarious = minor_stress_impact_loss
			shy = minor_stress_impact_gain
			deceitful = minor_stress_impact_loss
		}
	}
	option = {
		name = gh_travel_event.1001.b

		stress_impact = {
			shy = minor_stress_impact_loss
			gregarious = minor_stress_impact_gain
			patient = miniscule_stress_impact_loss
			callous = miniscule_stress_impact_loss
		}
	}
	after = {
		mp_resume_travel_plan = yes
	}
}

	content_source = dlc_gh
	title = gh_travel_event.3021.t
	desc = gh_travel_event.3021.desc
	theme = travel
	left_portrait = {
		character = root
		animation = worry
	}
	right_portrait = {
		character = scope:the_fool
		animation = sick
	}
	lower_right_portrait = {
		character = scope:physician
		trigger = {
			NAND = {
				exists = scope:travel_leader
				scope:physician ?= scope:travel_leader
			}
		}
	}
	cooldown = { years = 60 } # We shouldn't see this too often unfortunately
	trigger = {
		current_travel_plan.days_travelled >= 15 # Shouldnt see this if you start here
		is_available_travelling_adult = yes
		is_location_valid_for_travel_event_on_land = yes
		location = { geographical_region = GH_geographic_fools_coast }
	}
	immediate = {
		mp_delay_travel_plan = { DAYS = 90 }
		current_travel_plan = {
			random_entourage_character = {
				limit = { NOT = { has_trait = fools_curse } }
				save_scope_as = the_fool
			}
		}
		scope:the_fool = {
			contract_disease_effect = { DISEASE = fools_curse TREATMENT_EVENT = no } 
		}
		if = {
			limit = {
				any_court_position_holder = {
					type = executioner_court_position
				}
			}
			random_court_position_holder = {
				type = executioner_court_position
				save_scope_as = executioner
			}
		}
	}

	option = { # We must travel home immediately
		name = gh_travel_event.3021.a
		trigger = {
			NOT = { current_travel_plan.next_destination_province = root.capital_province }
		}
		current_travel_plan = { cancel_travel_plan = yes }
		ai_chance = {
			base = 25
			ai_value_modifier = {
				ai_boldness = 0.5
				ai_energy = -0.5
				ai_rationality = -0.5
			}
			modifier = {
				factor = 0
				is_landless_adventurer = yes
			}
		}
	}
	option = { # Physician, do what you can
		name = gh_travel_event.3021.b
		trigger = {
			custom_tooltip = {
				text = court_physician_not_employed_or_away_tt
				court_physician_available_trigger = yes
			}
		}
		show_as_unavailable = { always = yes }
		safe_disease_treatment_effect = { PATIENT = root TREATMENT_PICKER = root }
		ai_chance = {
			base = 500
		}
	}
	option = { # Im sorry
		name = gh_travel_event.3021.c
		trigger = {
			OR = {
				has_trait = callous
				has_trait = brave
				any_courtier = { has_court_position = executioner_court_position }
			}
		}
		show_as_unavailable = {
			any_is_pacifistic_trigger = yes
		}

		if = {
			limit = { exists = scope:executioner }
			scope:the_fool = { death = { death_reason = death_execution_killer killer = scope:executioner } }
		}
		else = {
			scope:the_fool = { death = { death_reason = death_execution_killer killer = root } }
		}

		stress_impact = {
			compassionate = major_stress_impact_gain
			sadistic = major_stress_impact_loss
			craven = medium_stress_impact_loss
			callous = medium_stress_impact_loss
		}
	}

	option = { # Leave them behind
		name = gh_travel_event.3021.d
		hidden_effect = { scope:the_fool = { death = { death_reason = death_vanished } } }

		stress_impact = {
			compassionate = minor_stress_impact_gain
			sadistic = minor_stress_impact_loss
		}
	}

	option = { # Well, dont mind if i do!
		name = gh_travel_event.3021.e
		add_internal_flag = fancy_coloring
		add_internal_flag = fancy_coloring_orange_stretch
		trigger = {
			has_trait = sadistic
		}
		show_as_unavailable = {
			NOT = { has_trait = sadistic }
		}
		scope:the_fool = { 
			add_trait = gh_dehanded
			death = { death_reason = death_execution_killer killer = root }
		}
		hidden_effect = {
			contract_disease_effect = { DISEASE = fools_curse TREATMENT_EVENT = no } # look it's your own fault at this point
		}
	}

	after = { 
		mp_resume_travel_plan = yes
	}
}

gh_colonization_events.0001 = { # Peace and Prosperity
	type = character_event
	content_source = dlc_gh
	title = gh_colonization_events.0001.t
	desc = gh_colonization_events.0001.desc
	theme = realm
	override_background = { reference = terrain_scope }
	left_portrait = {
		character = scope:acting_character
		triggered_animation = {
			trigger = {
				NOT = { has_trait = lazy }
				is_ai = no
			}
			animation = personality_rational
		}
		triggered_animation = {
			trigger = {
				has_trait = lazy
				is_ai = no
			}
			animation = boredom
		}
		triggered_animation = {
			trigger = { is_ai = yes }
			animation = chancellor
		}
	}

	trigger = {
		NOT = { has_character_flag = had_event_gh_colonization_events_0001 }
		scope:colony = { NOT = { exists = var:had_event_gh_colonization_events_0001 } }
	}

	weight_multiplier = {
		base = 1
	}
	
	immediate = {
		if = {
			limit = { any_court_position_holder = { type = colonial_administrator_court_position } }
			random_court_position_holder = {
				type = colonial_administrator_court_position
				save_scope_as = acting_character
			}
		}
		scope:colony = {
			title_province = { save_scope_as = background_terrain_scope }
			set_variable = {
				name = had_event_gh_colonization_events_0001
				years = 10
			}
		}
		add_character_flag = {
			flag = had_event_gh_colonization_events_0001
			years = 10
		}
	}

	option = { # Let the colony prosper
		name = gh_colonization_events.0001.a
		scope:colony = {
			add_county_modifier = {
				modifier = colonial_boom_modifier
				years = 5
			}
		}
		stress_impact = {
			generous = miniscule_stress_impact_loss
			greedy = medium_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_honor = 0.4
				ai_greed = -1
				ai_rationality = 0.2
				ai_energy = -0.2
				ai_compassion = 0.2
			}
		}
	}

	option = { # Tax the colony
		name = gh_colonization_events.0001.b
		scope:colony = {
			add_county_modifier = {
				modifier = colonial_prosperity_modifier
				years = 5
			}
		}
		add_gold = minor_gold_value
		stress_impact = {
			greedy = medium_stress_impact_loss
			generous = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_greed = 1
				ai_compassion = -0.5
			}
		}
	}

	option = { # Invest in infrastructure
		name = gh_colonization_events.0001.c
		scope:colony = {
			add_county_modifier = {
				modifier = infrastructure_expansion_modifier
				years = 15
			}
		}
		stress_impact = {
			ambitious = medium_stress_impact_loss
			diligent = medium_stress_impact_loss
			greedy = medium_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_greed = -0.5
				ai_rationality = 1
				ai_energy = 0.5
			}
		}
	}

	option = { # Implement draconic taxation
		name = gh_colonization_events.0001.d
		trigger = {
			OR = {
				has_trait = greedy
				has_trait = arbitrary
				has_trait = avaricious
			}
		}
		scope:story = {
			change_variable = {
				name = colonization_progress
				add = -20
			}
		}
		custom_tooltip = gh_colonization_events.0001.d.tt
		add_gold = major_gold_value
		stress_impact = {
			greedy = major_stress_impact_loss
			fickle = minor_stress_impact_loss
			arbitrary = minor_stress_impact_loss
			compassionate = medium_stress_impact_gain
			just = medium_stress_impact_gain
			generous = massive_stress_impact_gain
		}
		ai_chance = {
			base = 200
			ai_value_modifier = {
				ai_honor = -0.5
				ai_greed = 2
				ai_rationality = -0.5
				ai_compassion = -0.5
			}
		}
	}
}

gh_colonization_events.0002 = { # A Plague of Brigands
	type = character_event
	content_source = dlc_gh
	title = gh_colonization_events.0002.t
	desc = gh_colonization_events.0002.desc
	theme = realm
	override_background = { reference = terrain_scope }
	left_portrait = {

		triggered_animation = {
			trigger = {
				NOT = { has_trait = lazy }

			animation = chancellor
		}
	}
	right_portrait = {
		character = scope:court_adventurer
		animation = marshal
	}

	trigger = {
		NOT = { has_character_flag = had_event_gh_colonization_events_0002 }
		scope:colony = {
			NOT = { exists = var:had_event_gh_colonization_events_0002 }
			any_neighboring_county = {
				holder = root
				title_province = { NOT = { has_holding_type = colony_holding } }
			}
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = -0.5

			limit = { any_court_position_holder = { type = court_adventurer_court_position } }

		}
		add_character_flag = {
			flag = had_event_gh_colonization_events_0002
			years = 10
		}
	}

	option = { # Do nothing
		name = gh_colonization_events.0002.a
		scope:colony = {
			add_county_modifier = {
				modifier = bandit_stronghold_modifier
				years = 5
			}
		}
		custom_tooltip = {
			text = gh_colonization_events.0002.a.tt
			scope:colony = {
				every_neighboring_county = {
					limit = {
						holder = {
							OR = {
								this = root
								is_vassal_of = root
							}
						}
					}
					change_county_control = medium_county_control_loss
				}
			}
		}
		add_legitimacy = minor_legitimacy_loss
		stress_impact = {
			craven = medium_stress_impact_loss
			lazy = minor_stress_impact_loss
			just = medium_stress_impact_gain
			brave = medium_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_honor = -1
				ai_greed = 0.5
				ai_rationality = -0.5
				ai_energy = 1
				ai_boldness = -1
				ai_vengefulness = -0.5
				ai_compassion = -0.2
			}
		}
	}

	option = { # Increase patrols
		name = gh_colonization_events.0002.b
		scope:colony = {
			add_county_modifier = {
				modifier = frontier_banditry_modifier
				years = 5
			}
		}
		remove_short_term_gold = medium_gold_value
		stress_impact = {
			paranoid = minor_stress_impact_loss
			greedy = miniscule_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_greed = -0.5
				ai_rationality = 1
			}
		}
	}

	option = { # Hunt down the bandits

			}

		stress_impact = {
			brave = medium_stress_impact_loss
			just = medium_stress_impact_loss
			diligent = minor_stress_impact_loss
			greedy = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_honor = 1
				ai_greed = -1
				ai_rationality = 0.5
				ai_energy = 1
				ai_boldness = 1
			}
		}
	}

	option = { # Have the court adventurer deal with it
		name = gh_colonization_events.0002.d
		trigger = { exists = scope:court_adventurer }
		custom_tooltip = gh_colonization_events.0002.d.tt

gh_colonization_events.0004 = { # An Incompetent Administrator
	type = character_event
	content_source = dlc_gh
	title = gh_colonization_events.0004.t
	desc = gh_colonization_events.0004.desc
	theme = realm
	override_background = { reference = dungeon }
	right_portrait = {
		character = scope:idiot
		animation = prisondungeon
	}
	left_portrait = {
		character = scope:jailor
		animation = marshal
	}
	
	trigger = {
		NOT = { has_character_flag = had_event_gh_colonization_events_0004 }
		scope:colony = { NOT = { exists = var:had_event_gh_colonization_events_0004 } }
		any_knight = {
			exists = yes
			is_healthy = yes
			location = root.location
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = -0.5
			has_perk = surety_of_supply_perk
		}
		modifier = {
			add = 0.5
			any_court_position_holder = {
				type = colonial_administrator_court_position
				has_character_flag = colonial_administrator_attend_colonial_affairs_flag
			}
		}
	}
	
	immediate = {
		if = {
			limit = { any_court_position_holder = { type = colonial_administrator_court_position } }
			random_court_position_holder = {
				type = colonial_administrator_court_position
				save_scope_as = acting_character
			}
		}
		scope:colony = {
			title_province = { save_scope_as = background_terrain_scope }
			set_variable = {
				name = had_event_gh_colonization_events_0004
				years = 10
			}
			add_county_modifier = {
				modifier = colonial_mismanagement_modifier
				years = 5
			}
		}
		add_character_flag = {
			flag = had_event_gh_colonization_events_0004
			years = 10
		}
		create_character = {
			template = incompetent_administrator_template
			location = root.capital_province
			save_scope_as = idiot
		}
		random_knight = {
			limit = {
				is_healthy = yes
				location = root.location
			}
			save_scope_as = jailor
		}
	}

	option = { # Get them out of my sight
		name = gh_colonization_events.0004.a
		custom_tooltip = gh_colonization_events.0004.a.tt
		scope:idiot = { silent_disappearance_effect = yes }
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_energy = -0.3
				ai_vengefulness = -0.2
				ai_compassion = 0.5
			}
		}
	}

	option = { # They will rot in my dungeon
		name = gh_colonization_events.0004.b
		imprison = {
			target = scope:idiot
			type = dungeon
		}
		stress_impact = {
			callous = minor_stress_impact_loss
			forgiving = minor_stress_impact_gain
			compassionate = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_vengefulness = 0.4
				ai_compassion = -0.5
			}
		}
	}

	option = { # Kill this fool
		name = gh_colonization_events.0004.c
		execute_prisoner_effect = {
			VICTIM = scope:idiot
			EXECUTIONER = root
		}
		stress_impact = {
			sadistic = medium_stress_impact_loss
			callous = medium_stress_impact_loss
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_vengefulness = 1
				ai_compassion = -1
			}
		}
	}
}

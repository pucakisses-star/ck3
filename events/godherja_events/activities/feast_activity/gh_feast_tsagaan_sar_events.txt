# gh_feast_tsagaan_sar_events.0001 - Oltenis take off their masks on the Day of Olik

gh_feast_tsagaan_sar_events.0001 = {
	type = activity_event

	title = gh_feast_tsagaan_sar_events.0001.t
	desc = gh_feast_tsagaan_sar_events.0001.desc

	theme = feast_activity

	left_portrait = {
		character = root
		animation = ecstasy
	}

	option = { # Take off the mask, cynical character is glad
		flavor = gh_feast_tsagaan_sar_events.0001.a.flavor
		name = gh_feast_tsagaan_sar_events.0001.a
		trigger = { has_trait = cynical }
		add_character_flag = {
			flag = day_of_olik_discarded_mask
			weeks = 1
		}
		add_prestige = medium_prestige_gain
		custom_tooltip = gh_feast_tsagaan_sar_events.0001.maskless
		stress_impact = {
			cynical = medium_stress_impact_loss
			brave = minor_stress_impact_loss
			shy = minor_stress_impact_gain
			paranoid = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = 0.5
				ai_zeal = -0.5
				ai_sociability = 0.25
			}
		}
	}

	option = { # Take off the mask
		flavor = gh_feast_tsagaan_sar_events.0001.b.flavor
		name = gh_feast_tsagaan_sar_events.0001.b
		add_character_flag = {
			flag = day_of_olik_discarded_mask
			weeks = 1
		}
		add_piety = major_piety_value
		custom_tooltip = gh_feast_tsagaan_sar_events.0001.maskless
		stress_impact = {
			brave = minor_stress_impact_loss
			zealous = minor_stress_impact_loss
			craven = medium_stress_impact_gain
			shy = medium_stress_impact_gain
			paranoid = medium_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = 0.5
				ai_sociability = 0.25
			}
		}
	}

	option = { # Keep the mask
		flavor = gh_feast_tsagaan_sar_events.0001.c.flavor
		name = gh_feast_tsagaan_sar_events.0001.c
		add_piety = minor_piety_loss
		custom_tooltip = gh_feast_tsagaan_sar_events.0001.masked
		stress_impact = {
			paranoid = medium_stress_impact_loss
			craven = medium_stress_impact_loss
			shy = medium_stress_impact_loss
			brave = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = -0.25
				ai_sociability = -0.5
			}
		}
	}

	option = { # Keep the mask, too paranoid to do otherwise
		flavor = gh_feast_tsagaan_sar_events.0001.d.flavor
		name = gh_feast_tsagaan_sar_events.0001.d
		trigger = { has_trait = paranoid }
		add_prestige = minor_prestige_loss
		custom_tooltip = gh_feast_tsagaan_sar_events.0001.masked
		stress_impact = {
			paranoid = major_stress_impact_loss
			craven = medium_stress_impact_loss
			shy = medium_stress_impact_loss
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = -0.5
				ai_sociability = -0.25
			}
		}
	}

	after = {
		hidden_effect = {
			if = { # I LOVE RECURSION
				limit = { this = scope:host }
				involved_activity = {
					every_attending_character = { # That includes YOU!
						limit = { religion = religion:oltenic_religion } # Only relevant characters
						switch = {
							trigger = is_ai
							yes = { # AI characters decide based on MATHS
								random = {
									chance = {
										value = 80
										add = {
											value = ai_boldness
											multiply = 0.5
										}
										add = ai_sociability
										max = 100
										min = 20
									}
									add_character_flag = {
										flag = day_of_olik_discarded_mask
										weeks = 1
									}
								}
							}
							no = { # Non-AI characters do get to decide
								if = {
									limit = { NOT = { this = root } } # We have already decided above
									trigger_event = gh_feast_tsagaan_sar_events.0001
								}
								trigger_event = { # Trigger the resolution event
									id = gh_feast_tsagaan_sar_events.0002
									days = 1 # Delay to prevent event spam
								}
							}
						}
					}
				}
			}
		}
	}
}

gh_feast_tsagaan_sar_events.0002 = {
	type = activity_event

	title = gh_feast_tsagaan_sar_events.0002.t
	desc = gh_feast_tsagaan_sar_events.0002.desc

	theme = feast_activity

	left_portrait = {
		character = root
		animation = toast_goblet
	}

	right_portrait = {
		character = scope:featured_attendee
		animation = ecstasy
	}

	immediate = {
		involved_activity = {
			random_attending_character = {
				limit = { NOT = { this = root } }
				weight = {
					base = 1
					modifier = {
						add = 2
						has_relation_to = root
					}
					modifier = {
						add = 5
						is_landed = yes
					}
					modifier = {
						add = 8
						has_character_flag = day_of_olik_discarded_mask
					}
					modifier = {
						add = 3
						has_activity_intent = befriend_attendee_intent
						intent_target ?= root
					}
					modifier = {
						add = 3
						root = {
							has_activity_intent = befriend_attendee_intent
							intent_target ?= prev
						}
					}
				}
				save_scope_as = featured_attendee
			}
		}
		if = {
			limit = { has_character_flag = day_of_olik_discarded_mask }
			custom_tooltip = {
				text = gh_feast_tsagaan_sar_events.0002.maskless
				involved_activity = {
					every_attending_character = {
						limit = {
							NOT = { this = root }
							has_character_flag = day_of_olik_discarded_mask
						}
						add_opinion = {
							modifier = day_of_olik_discarded_mask_opinion
							target = root
						}
					}
				}
			}
		}
		else = {
			custom_tooltip = {
				text = gh_feast_tsagaan_sar_events.0002.masked
				involved_activity = {
					every_attending_character = {
						limit = {
							NOT = { this = root }
							NOT = { has_character_flag = day_of_olik_discarded_mask }
							faith = { has_doctrine_parameter = can_host_oltenic_feast_type } # Some might not be Oltenic, unlike above
						}
						add_opinion = {
							modifier = day_of_olik_kept_mask_opinion
							target = root
						}
					}
				}
			}
		}
	}

	option = {
		name = gh_feast_tsagaan_sar_events.0002.a
		stress_impact = {
			gregarious = medium_stress_impact_loss
			lifestyle_reveler = medium_stress_impact_loss
			gluttonous = minor_stress_impact_loss
			shy = minor_stress_impact_gain
		}
	}
}

namespace = gh_feast_default

# gh_feast_default.0001 - Guest gets angry over being served kumis

gh_feast_default.0001 = {
	type = activity_event
	title = gh_feast_default.0001.t
	desc = gh_feast_default.0001.desc
	theme = feast_activity

	left_portrait = {
		character = scope:fussy_guest
		animation = dismissal
	}
	right_portrait = {
		character = scope:offended_guest
		animation = anger
	}

	cooldown = { years = 1 }

	trigger = {
		NOT = { exists = local_var:had_gh_0001 }
		scope:activity = {
			activity_host = {
				this = root
				accustomed_to_drinking_kumis = yes
			}
			any_attending_character = { # Offended guest
				feast_default_adult_participant_trigger = yes
				culture = { has_same_culture_heritage = root.culture }
				NOT = { this = root }
			}
			any_attending_character = { # Fussy guest
				feast_default_adult_participant_trigger = yes
				accustomed_to_drinking_kumis = no
				NOT = { this = root }
			}
		}
		not_hosting_a_murder_feast = yes
	}

	immediate = {
		set_local_variable = {
			name = had_gh_0001
			value = yes
		}
		scope:activity = {
			random_attending_character = {
				limit = {
					feast_default_adult_participant_trigger = yes
					culture = { has_same_culture_heritage = root.culture }
				}
				save_scope_as = offended_guest
			}
			random_attending_character = {
				limit = {
					feast_default_adult_participant_trigger = yes
					accustomed_to_drinking_kumis = no
				}
				save_scope_as = fussy_guest
				add_opinion = {
					target = scope:offended_guest
					modifier = angry_opinion
					opinion = -20
				}
				reverse_add_opinion = {
					target = scope:offended_guest
					modifier = disgusted_opinion
					opinion = -20
				}
			}
		}
	}

	option = { # Ask the fussy guest to give another try
		name = gh_feast_default.0001.a
		duel = {
			skill = diplomacy
			value = medium_skill_rating
			50 = { # They develop a taste
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				desc = gh_feast_default.0001.a.success
				add_prestige = miniscule_prestige_gain
				reverse_add_opinion = {
					target = scope:fussy_guest
					modifier = grateful_opinion
					opinion = 20
				}
				culture = {
					change_cultural_acceptance = {
						target = scope:fussy_guest.culture
						value = minor_cultural_acceptance_gain
						desc = cultural_acceptance_embraced_festival
					}
				}
			}
			50 = { # They remain unconvinced
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				desc = gh_feast_default.0001.a.failure
				reverse_add_opinion = {
					target = scope:fussy_guest
					modifier = disgusted_opinion
					opinion = -10
				}
			}
		}
		stress_impact = {
			gregarious = minor_stress_impact_loss
			shy = minor_stress_impact_gain
		}
		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_boldness = 1
				ai_energy = 1
			}
			modifier = {
				has_trait = gregarious
				add = 25
			}
			modifier = {
				has_trait = shy
				add = -25
			}
		}
	}

	option = { # Make them drink with force
		name = gh_feast_default.0001.b
		trigger = { has_trait = sadistic }
		custom_tooltip = gh_feast_default.0001.b.tt
		add_dread = medium_dread_gain
		reverse_add_opinion = {
			target = scope:fussy_guest
			modifier = hate_opinion
			opinion = -60
		}
		progress_towards_rival_effect = {
			CHARACTER = scope:fussy_guest
			OPINION = 0
			REASON = rival_forced_to_drink_kumis
		}
		stress_impact = {
			sadistic = medium_stress_impact_loss
			arbitrary = minor_stress_impact_loss
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_energy = 0.25
				ai_compassion = -2
			}
		}
	}

	option = { # Charge them with breaching your hospitality
		name = gh_feast_default.0001.c
		trigger = {
			scope:fussy_guest = {
				OR = {
					is_landed = no
					is_vassal_of = root
				}
			}
		}
		add_dread = minor_dread_gain
		rightfully_imprison_character_effect = {
			TARGET = scope:fussy_guest
			IMPRISONER = root
		}
		reverse_add_opinion = {
			target = scope:fussy_guest
			modifier = angry_opinion
			opinion = -20
		}
		progress_towards_rival_effect = {
			CHARACTER = scope:fussy_guest
			OPINION = 0
			REASON = rival_unjustified_imprisonment_corresponding
		}
		stress_impact = {
			callous = minor_stress_impact_loss
			compassionate = medium_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = 0.25
				ai_compassion = -1
			}
		}
	}

	option = { # Tell offended guest to let go
		name = gh_feast_default.0001.d
		add_prestige = miniscule_prestige_loss
		reverse_add_opinion = {
			target = scope:offended_guest
			modifier = disappointed_opinion
			opinion = -10
		}
	}

	option = { # You see no reason to intervene
		name = gh_feast_default.0001.e
		trigger = {
			NOR = {
				has_trait = arbitrary
				has_trait = eccentric
				has_trait = wrathful
			}
		}
		scope:offended_guest = {
			duel = {
				skill = prowess
				target = scope:fussy_guest
				50 = { # Offended guest wins
					compare_modifier = {
						value = scope:duel_value
						multiplier = 3.5
						min = -49
					}
					scope:fussy_guest = {
						increase_wounds_no_death_effect = { REASON = punishment }
					}
				}
				50 = { # Fussy guest wins
					compare_modifier = {
						value = scope:duel_value
						multiplier = -3.5
						min = -49
					}
					scope:offended_guest = {
						increase_wounds_no_death_effect = { REASON = punishment }
					}
				}
			}
			progress_towards_rival_effect = {
				CHARACTER = scope:fussy_guest
				OPINION = 0
				REASON = rival_grave_insult
			}
		}
		stress_impact = {
			shy = minor_stress_impact_loss
			sadistic = miniscule_stress_impact_loss
			just = minor_stress_impact_gain
			compassionate = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_energy = -0.5
				ai_rationality = -0.5
				ai_compassion = -0.5
			}
		}
	}

	option = { # Finally, some entertainment
		name = gh_feast_default.0001.f
		trigger = {
			OR = {
				has_trait = arbitrary
				has_trait = eccentric
				has_trait = wrathful
			}
		}
		scope:offended_guest = {
			duel = {
				skill = prowess
				target = scope:fussy_guest
				50 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = 3.5
						min = -49
					}
					scope:fussy_guest = {
						increase_wounds_no_death_effect = { REASON = punishment }
					}
				}
				50 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = -3.5
						min = -49
					}
					scope:offended_guest = {
						increase_wounds_no_death_effect = { REASON = punishment }
					}
				}
			}
			progress_towards_rival_effect = {
				CHARACTER = scope:fussy_guest
				OPINION = 0
				REASON = rival_grave_insult
			}
		}
		stress_impact = {
			arbitrary = medium_stress_impact_loss
			sadistic = miniscule_stress_impact_loss
			just = minor_stress_impact_gain
			compassionate = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_energy = -0.5
				ai_rationality = -0.5
				ai_compassion = -0.5
			}
		}
	}
}

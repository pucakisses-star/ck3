scripted_trigger appreciates_marcher_literature = { # Character is an avid reader of Marcher literature
	OR = {
		faith_is_religion_or_syncretic_trigger = { RELIGION = lifepath_religion }
		knows_language = language_marcher
		knows_language = language_rivarisian
		knows_language = language_ojyczayn
		knows_language = language_konijkmener
		culture = {
			OR = {
				has_cultural_pillar = heritage_marcher
				has_cultural_pillar = heritage_konijkmener
				has_cultural_pillar = language_marcher
				has_cultural_pillar = language_rivarisian
				has_cultural_pillar = language_ojyczayn
				has_cultural_pillar = language_konijkmener
				has_cultural_tradition = tradition_chevalien_identity
			}
		}
	}
}

scripted_trigger appreciates_aversarian_literature = { # Character is an avid reader of Aversarian literature
	OR = {
		faith_is_religion_or_syncretic_trigger = { RELIGION = aversarinas_aagiokrata_religion }
		faith_is_religion_or_syncretic_trigger = { RELIGION = agionism_religion }
		knows_language = language_aversarian
		culture = {
			OR = {
				has_cultural_pillar = heritage_aversarian
				has_cultural_pillar = language_aversarian
				has_cultural_tradition = tradition_legacy_of_aversaria
				has_cultural_tradition = tradition_legacy_of_aversaria_lesser
			}
		}
	}
}

scripted_trigger appreciates_sarradonian_literature = { # Character is an avid reader of Sarradonian literature
	OR = {
		faith_is_religion_or_syncretic_trigger = { RELIGION = wardenism_religion }
		faith_is_religion_or_syncretic_trigger = { RELIGION = ritualist_religion }
		knows_language = language_thalite
		knows_language = language_koshar
		knows_language = language_dezhunic
		culture = {
			OR = {
				has_cultural_pillar = heritage_sarradonian
				has_cultural_pillar = heritage_kosharan
				has_cultural_pillar = heritage_kharadamatian
				has_cultural_pillar = heritage_dezhunic
				has_cultural_pillar = heritage_dawic
				has_cultural_pillar = language_thalite
				has_cultural_pillar = language_koshar
				has_cultural_pillar = language_dezhunic
			}
		}
	}
}

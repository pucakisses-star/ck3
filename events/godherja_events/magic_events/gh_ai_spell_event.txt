namespace = magic_ai

# on adding spells:
# the trigger should check the cost, and all other requirements. A scripted trigger is best, so changing is easier
# ai_value determines the likelihood the spell is cast.
# using a scripted effect is also preferable

scripted_effect set_ai_spell_overcharge_effect = {
	var:selected_spell = {
		random_in_list = {
			variable = spell_aspect_list
			save_scope_as = aspect_being_overcharged
		}
	}
	save_scope_value_as = {
		name = overcharge_increase_amount
		value = aspect_being_overcharged_max_value
	}
	magic_aspect_effect = { MODE = increase_overcharge }
	save_scope_value_as = {
		name = spell_success_chance_var
		value = var:selected_spell.spell_success_chance
	}
	if = {
		limit = {
			OR = {
				NOT = { exists = var:casting_through_court_magi_var }
				court_magi_spellcasting_acceptance_value > 0
			}
		}
		random = {
			chance = var:selected_spell.spell_success_chance
			save_scope_value_as = {
				name = spell_overcharge_selected
				value = yes
			}
		}
	}
	if = {
		limit = { NOT = { exists = scope:spell_overcharge_selected } }
		save_scope_value_as = {
			name = overcharge_decrease_amount
			value = {
				integer_range = {
					min = 0
					max = aspect_being_overcharged_max_value
				}
			}
		}
		magic_aspect_effect = { MODE = decrease_overcharge }
		if = {
			limit = {
				OR = {
					NOT = { exists = var:casting_through_court_magi_var }
					court_magi_spellcasting_acceptance_value > 0
				}
			}
			random = {
				chance = var:selected_spell.spell_success_chance
				save_scope_value_as = {
					name = spell_overcharge_selected
					value = yes
				}
			}
		}
		if = {
			limit = { NOT = { exists = scope:spell_overcharge_selected } }
			save_scope_value_as = {
				name = overcharge_decrease_amount
				value = 100
			}
			debug_log = "DECREASING OVERCHARGE"
			magic_aspect_effect = { MODE = decrease_overcharge }
		}
	}
}

magic_ai.0001 = {
	hidden = yes
	
	trigger = {
		OR = {
			can_access_magic = yes
			exists = cp:councillor_court_magi
		}
		OR = {
			has_stored_magic_trigger = { VALUE = spell_level_iteration_value } # Needs to have enough magic for at least a level 1 spell
			AND = {
				exists = cp:councillor_court_magi
				cp:councillor_court_magi = { has_stored_magic_trigger = { VALUE = spell_level_iteration_value } }
			}
		}
		trigger_if = {
			limit = { exists = var:spell_exposure }
			OR = {
				var:spell_exposure <= 0 # TODO: Make this more nuanced in the future, making it stay on the safe side for the moment
				exists = cp:councillor_court_magi
			}
		}
		is_pool_character = no
		is_physically_able_ai_adult = yes
	}
	
	immediate = {
		save_scope_as = spell_organizer
		remove_variable = casting_through_court_magi_var
		remove_variable = selected_spell
		clear_variable_list = spell_targets
		clear_variable_list = spell_target_titles
		random_list = {
			0 = {
				trigger = {
					can_access_magic = yes
					OR = {
						NOT = { exists = var:spell_exposure }
						var:spell_exposure <= 0
					}
				}
				modifier = {
					add = {
						value = total_spellcasting_skill_value
						multiply = 2
					}
				}
				save_scope_as = caster
			}
			0 = {
				trigger = { exists = cp:councillor_court_magi }
				modifier = {
					add = cp:councillor_court_magi.total_spellcasting_skill_value
				}
				cp:councillor_court_magi = { save_scope_as = caster }
				set_variable = casting_through_court_magi_var
			}
		}
		if = {
			limit = { NOT = { exists = scope:caster } }
			save_scope_as = caster
		}
		if = {
			limit = { highest_held_title_tier >= tier_county }
			random_held_title = {
				limit = { tier = tier_county }
				weight = {
					modifier = { factor = development_level }
					modifier = {
						factor = 15
						this = root.capital_county
					}
				}
				root = {
					add_to_variable_list = {
						name = spell_target_titles
						target = prev
					}
				}
				save_temporary_scope_as = spell_being_checked_ai_target_title
				save_temporary_scope_as = spell_being_checked_ai_target_title_friendly
			}
		}
		
		# First, check for a friendly spell to cast on oneself
		save_scope_as = spell_being_checked_ai_target
		scope:caster = {
			ordered_in_list = {
				variable = known_spell_list
				order_by = ai_spell_weight_value
				max = 3
				check_range_bounds = no
				limit = { 
					ai_spell_weight_value > 0
					scope:caster = { 
						has_stored_magic_trigger = { VALUE = prev.total_spell_cost }
					}
					scope:spell_organizer = {
						gold >= prev.total_spell_gold_cost
					}
					OR = {
						NOT = { exists = root.var:casting_through_court_magi_var }
						court_magi_spellcasting_acceptance_value > 0
					}
				}
				if = {
					limit = { NOT = { exists = scope:ai_friendly_selected_spell_scope } }
					random = {
						chance = spell_success_chance
						save_scope_as = ai_friendly_selected_spell_scope
						save_scope_value_as = {
							name = ai_friendly_selected_spell_scope_value
							value = scope:ai_friendly_selected_spell_scope.ai_spell_weight_value
						}
					}
				}
			}
		}
		
		#Vanilla murder logic copied and slightly modified from this part onward
		#The peculiar arrangement with an empty if statement & a separate random_in_list is intentional; they save on performance. Somehow.
		#JustCodeThings.
		if = {
			limit = { build_murder_targets_trigger = yes }
		
			random_in_list = {
				list = murder_targets
				limit = {
					NOR = {
						is_child_of = root
						is_grandchild_of = root
						is_great_grandchild_of = root
					}
					NOT = { this = root }
					# No targets you're Cowed by	
					NOT = {
						root = {
							has_dread_level_towards = {
								target = prev
								level = 2
							}
						}
					}
				}
				weight = {
					base = 0
					modifier = {
						add = -100
						has_relation_friend = root
					}
					modifier = {
						add = -25
						has_relation_lover = root
					}
					modifier = {
						add = -1000
						OR = {
							has_relation_best_friend = root
							has_relation_soulmate = root
						}
					}
					modifier = {
						add = root.ai_vengefulness
						root.ai_vengefulness > 0
						OR = {
							has_relation_rival = root
							has_relation_nemesis = root
							exposed_cheating_on_spouse_trigger = { SPOUSE = root }
						}
					}
					modifier = {
						add = root.ai_greed
						root.ai_greed > 0
						OR = {
							any_held_title = {
								tier >= root.highest_held_title_tier
								place_in_line_of_succession = {
									target = root
									value <= 3
								}
							}
							any_heir_title = {
								NOT = { holder = root }
								tier >= root.highest_held_title_tier
								place_in_line_of_succession = {
									target = root
									value <= 3
								}
							}
						}
					}
					modifier = { # If you're really good at murdering, killing claimants makes a lot of sense
						add = 50
						root = {
							OR = {
								has_perk = a_job_done_right_perk
								has_focus = intrigue_skulduggery_focus
							}
						}
						save_temporary_scope_as = the_murder_target
						root = {
							is_landed = yes
							primary_title.tier >= tier_duchy
							any_targeting_faction = {
								faction_is_type = claimant_faction
								special_character = {
									this = scope:the_murder_target
								}
							}
						}
					}
					opinion_modifier = { # Opinion Factor
						who = root
						opinion_target = this
						multiplier = -1.0
					}
					modifier = { # Avoid murdering sponsored characters, lest we draw the ire of their patron
						add = -50
						exists = inspiration
						inspiration = { exists = inspiration_sponsor }
					}
					# If involved in a struggle, you're markedly more likely to try and scheme against anyone inside that struggle.
					modifier = {
						add = 75
						save_temporary_scope_as = the_murder_target
						root = {
							any_character_struggle = {
								involvement = involved
								is_secondary_character_involvement_either_trigger = { CHAR = scope:the_murder_target }
							}
						}
					}
					modifier = { # Family Feud
						add = 50
						exists = scope:actor.house
						exists = scope:recipient.house
						scope:actor = {
							house.house_head = {
								any_owned_story = {
									story_type = story_cycle_house_feud
									has_variable = house_feud_house
									var:house_feud_house = scope:recipient.house
								}
							}
							has_opinion_modifier = {
								modifier = house_feud_opinion
								target = scope:recipient.house.house_head
							}
						}
					}
					modifier = { # More likely to cast against those that have cast against you
						add = 75
						save_temporary_scope_as = counterspell_target
						root = {
							has_opinion_modifier = {
								modifier = casted_unwanted_magic_me_crime
								target = scope:counterspell_target 
							}
						}
					}
				}
				random = {
					chance = 25
					modifier = {
						add = -100
						has_relation_friend = root
					}
					modifier = {
						add = -25
						has_relation_lover = root
					}
					modifier = {
						add = -1000
						OR = {
							has_relation_best_friend = root
							has_relation_soulmate = root
						}
					}
					modifier = {
						add = {
							value = 0
							subtract = root.ai_compassion
							divide = 3
						}
						root.ai_compassion > 0
					}
					modifier = {
						add = {
							value = 0
							subtract = root.ai_honor
							divide = 2
						}
						root.ai_honor > 0
					}
					modifier = {
						add = {
							value = root.ai_honor
							multiply = -1
							divide = 2
						}
						root.ai_honor < 0
					}
					modifier = {
						add = {
							value = root.ai_vengefulness
							multiply = 1.5
						}
						root.ai_vengefulness > 0
						OR = {
							has_relation_rival = root
							has_relation_nemesis = root
							exposed_cheating_on_spouse_trigger = { SPOUSE = root }
						}
					}
					modifier = { # Rational people understand that certain threats need to be taken care of
						add = {
							value = root.ai_rationality
							multiply = 0.5
						}
						root.ai_rationality > 0
						save_temporary_scope_as = the_murder_target
						root = {
							is_landed = yes
							primary_title.tier >= tier_duchy
							any_targeting_faction = {
								faction_is_type = claimant_faction
								special_character = {
									this = scope:the_murder_target
								}
							}
						}
					}
					modifier = { # If you're really good at murdering, killing claimants makes a lot of sense
						add = 50
						root = {
							OR = {
								has_perk = a_job_done_right_perk
								has_focus = intrigue_skulduggery_focus
							}
						}
						save_temporary_scope_as = the_murder_target
						root = {
							is_landed = yes
							primary_title.tier >= tier_duchy
							any_targeting_faction = {
								faction_is_type = claimant_faction
								special_character = {
									this = scope:the_murder_target
								}
							}
						}
					}
					modifier = {
						add = {
							value = root.ai_greed
							divide = 2
						}
						root.ai_greed > 0
						OR = {
							any_held_title = {
								tier >= root.highest_held_title_tier
								place_in_line_of_succession = {
									target = root
									value <=3
								}
							}
							any_heir_title = {
								NOT = { holder = root }
								tier >= root.highest_held_title_tier
								place_in_line_of_succession = {
									target = root
									value <=3
								}
							}
						}
					}
					# Halve the chance if they're Intimidating
					modifier = {
						factor = 0.5
						root = {
							has_dread_level_towards = {
								target = prev
								level >= 1
							}
						}
					}
					modifier = { # To fix a bug where lowborn characters want to murder random people...
						add = -1000
						NOT = { exists = dynasty }
						NOR = {
							has_relation_rival = root
							has_relation_nemesis = root
							exposed_cheating_on_spouse_trigger = { SPOUSE = root }
						}
					}
					# Agenda impact
					modifier = {
						root = {
							any_character_struggle = {
								involvement = involved
							}
						}
						is_important_or_vip_struggle_character = yes
						add = {
							value = 0
							if = {
								limit = {
									root = {
										any_character_struggle = {
											phase_has_catalyst = catalyst_gain_claim_on_title_fp2
										}
										has_character_flag = agenda_towards_escalation
									}
								}
								add = 100
							}
							else_if = {
								limit = {
									root = {
										any_character_struggle = {
											phase_has_catalyst = catalyst_gain_claim_on_title_fp2
										}
									}
								}
								add = -1000
							}
						}
					}
					modifier = {
						add = 50
						root = {
							has_character_flag = recently_affected_by_negative_magic
						}
					}
					save_scope_as = spell_being_checked_ai_target
					save_scope_as = spell_being_checked_ai_target_unfriendly
					root = {
						add_to_variable_list = {
							name = spell_targets
							target = scope:spell_being_checked_ai_target_unfriendly
						}
					}
					scope:spell_being_checked_ai_target_unfriendly = {
						if = {
							limit = { highest_held_title_tier >= tier_county }
							random_held_title = {
								limit = { tier = tier_county }
								weight = {
									modifier = { factor = development_level }
									modifier = {
										factor = 15
										this = root.capital_county
									}
								}
								root = {
									add_to_variable_list = {
										name = spell_target_titles
										target = prev
									}
									if = {
										limit = { exists = scope:spell_being_checked_ai_target_title_friendly }
										remove_list_variable = {
											name = spell_target_titles
											target = scope:spell_being_checked_ai_target_title_friendly
										}
									}
								}
								save_temporary_scope_as = spell_being_checked_ai_target_title_unfriendly
								save_temporary_scope_as = spell_being_checked_ai_target_title
							}
						}
					}
				}
			}
		}
		
		# Check for potential hostile spells
		if = {
			limit = { exists = scope:spell_being_checked_ai_target_unfriendly }
			scope:caster = {
				ordered_in_list = {
					variable = known_spell_list
					order_by = ai_spell_weight_inverse_value
					max = 3
					check_range_bounds = no
					limit = { 
						ai_spell_weight_inverse_value < 0
						scope:caster = { 
							has_stored_magic_trigger = { VALUE = prev.total_spell_cost }
							gold >= prev.total_spell_gold_cost
					}
						OR = {
							NOT = { exists = root.var:casting_through_court_magi_var }
							court_magi_spellcasting_acceptance_value > 0
						}
					}
					if = {
						limit = { NOT = { exists = scope:ai_unfriendly_selected_spell_scope } }
						random = {
							chance = {
								value = spell_success_chance
								max = 80
							}
							save_scope_as = ai_unfriendly_selected_spell_scope
							save_scope_value_as = {
								name = ai_unfriendly_selected_spell_scope_value
								value = scope:ai_unfriendly_selected_spell_scope.ai_spell_weight_inverse_value
							}
						}
					}
				}
			}
		}
		
		# Pick between the hostile and friendly spell
		random_list = {
			0 = {
				trigger = {
					exists = scope:ai_unfriendly_selected_spell_scope
				}
				modifier = {
					add = scope:ai_unfriendly_selected_spell_scope_value
				}
				set_variable = { name = selected_spell value = scope:ai_unfriendly_selected_spell_scope } 
				start_preparing_spell_effect = { SPELL = scope:ai_unfriendly_selected_spell_scope }
				set_ai_spell_overcharge_effect = yes
			}
			0 = {
				trigger = {
					exists = scope:ai_friendly_selected_spell_scope
				}
				modifier = {
					add = scope:ai_friendly_selected_spell_scope_value
				}
				set_variable = { name = selected_spell value = scope:ai_friendly_selected_spell_scope } 
				start_preparing_spell_effect = { SPELL = scope:ai_friendly_selected_spell_scope }
				set_ai_spell_overcharge_effect = yes
			}
		}
		
		if = {
			limit = { 
				exists = var:selected_spell
				var:selected_spell.spell_success_chance >= 10
				scope:caster = { 
					has_stored_magic_trigger = { VALUE = root.var:selected_spell.total_spell_cost }
				}
				root.gold >= root.var:selected_spell.total_spell_gold_cost
				trigger_if = {
					limit = { exists = var:casting_through_court_magi_var }
					scope:caster.court_magi_spellcasting_acceptance_value > 0
				}
			}
			random = {
				chance = var:selected_spell.spell_success_chance
				var:selected_spell = { 
					save_scope_as = selected_spell_scope
				}
				determine_spellcasting_outcome_effect = yes
				pick_spell_location_effect = yes
				debug_log = "AI CASTED MAGIC"
				debug_log_scopes = yes
				if = {
					limit = { exists = scope:always_succeed_spellcasting }
					trigger_event = magic.0001
				}
				else = {
					trigger_event = {
						on_action = magic_backfire_events
					}
				}
			}
		}
	}

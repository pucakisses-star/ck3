scripted_effect set_blackspawn_mission_difficulty_effect = {
	if = {
		limit = { story_owner = { has_trait = gh_blackspawn_5 } }
		set_variable = { name = mission_difficulty value = 5 }
		set_local_variable = { name = mission_difficulty_remaining value = 5 }	# For the while loop
	}
	else_if = {
		limit = { story_owner = { has_trait = gh_blackspawn_4 } }
		set_variable = { name = mission_difficulty value = 4 }
		set_local_variable = { name = mission_difficulty_remaining value = 4 }
	}
	else_if = {
		limit = { story_owner = { has_trait = gh_blackspawn_3 } }
		set_variable = { name = mission_difficulty value = 3 }
		set_local_variable = { name = mission_difficulty_remaining value = 3 }
	}
	else_if = {
		limit = { story_owner = { has_trait = gh_blackspawn_2 } }
		set_variable = { name = mission_difficulty value = 2 }
		set_local_variable = { name = mission_difficulty_remaining value = 2 }
	}
	else = {
		set_variable = { name = mission_difficulty value = 1 }
		set_local_variable = { name = mission_difficulty_remaining value = 1 }
	}
}

scripted_effect generate_blackspawn_kill_mission_attributes_effect = {
	random_list = {
		1 = { # A specific number?
			trigger = { NOT = { exists = var:mission_multiplicity } }
			set_variable = {
				name = mission_multiplicity
				value = local_var:mission_difficulty_remaining
			}
		}
		# Or someone specific? (TODO)
		1 = { # Or a specific number of someones with specific characteristic?
			random_list = {
				1 = { # A specific tier?
					trigger = {
						NOT = { exists = var:mission_target_primary_title_tier }
						any_ruler = {
							in_diplomatic_range = root
							highest_held_title_tier >= local_var:mission_difficulty_remaining
						}
					}
					set_variable = { name = mission_target_primary_title_tier value = local_var:mission_difficulty_remaining }
					set_local_variable = {	# For loop should stop once this condition is hit, since it's going to cause the mission to target the highest possible title with the remaining mission difficulty points (and therefore no other attributes can be added anymore)
						name = mission_difficulty_remaining
						value = 0
					}
				}
				1 = { # A specific number of virtues?
					trigger = {
						NOT = { exists = var:mission_target_virtue_number }
					}
					set_variable = { 
						name = mission_target_virtue_number 
						value = {
							value = local_var:mission_difficulty_remaining
							max = 2
						}
					}
				}
			}
		}
		1 = { # In a specific fashion?
			trigger = { NOT = { exists = var:mission_kill_method } }
			random_list = {
				1 = {	# Murder
					set_variable = { name = mission_kill_method value = flag:murder }
				}
				2 = {	# Sacrifice
					set_variable = { name = mission_kill_method value = flag:sacrifice }
				}
				#1 = {	# Magic GH TODO: Find a way to get the murderer when someone dies of a hostile targeted spell
				#	set_variable = { name = mission_kill_method value = flag:magic_targeted_spell }
				#}
			}
		}
	}
}

scripted_effect generate_blackspawn_raid_mission_attributes_effect = {	# Generate Raid Mission Attributes
	random_list = {
		1 = { # A specific number?
			trigger = { NOT = { exists = var:mission_multiplicity } }
			set_variable = {
				name = mission_multiplicity
				value = local_var:mission_difficulty_remaining
			}
		}
		1 = { # Specific development level?
			trigger = { NOT = { exists = var:mission_target_development } }
			set_variable = {
				name = mission_target_development
				value = {
					value = local_var:mission_difficulty_remaining
					subtract = 1
					multiply = 20
					max = 95
				}
			}
			root.primary_title = {
				ordered_title_to_title_neighboring_county = {
					order_by = development_level
					save_scope_as = county_development_check
				}
			}
			if = {	# Checking highest nearest development then adjusting target development based on that
				limit = { var:mission_target_development > scope:county_development_check.development_level }
				set_variable = {
					name = mission_target_development
					value = scope:county_development_check.development_level
				}
			}
			random_list = { # Randomize target development, higher mission difficulty means higher target developments
				1 = { # Nothing happens
				}
				1 = {
					change_variable = {
						name = mission_target_development
						add = 1
					}
				}
				0 = {
					modifier = {
						add = local_var:mission_difficulty_remaining
						local_var:mission_difficulty_remaining > 1
					}
					change_variable = {
						name = mission_target_development
						add = local_var:mission_difficulty_remaining
					}
				}
				0 = {
					modifier = {
						add = local_var:mission_difficulty_remaining
						local_var:mission_difficulty_remaining > 2
					}
					change_variable = {
						name = mission_target_development
						add = local_var:mission_difficulty_remaining
					}
				}
				0 = {
					modifier = {
						add = local_var:mission_difficulty_remaining
						local_var:mission_difficulty_remaining > 3
					}
					change_variable = {
						name = mission_target_development
						add = local_var:mission_difficulty_remaining
					}
				}
				0 = {
					modifier = {
						add = local_var:mission_difficulty_remaining
						local_var:mission_difficulty_remaining > 4
					}
					change_variable = {
						name = mission_target_development
						add = local_var:mission_difficulty_remaining
					}
				}
			}
		}
	}
}

scripted_effect generate_blackspawn_mission_effect = {
	create_divine_mission_effect = yes
	if = {
		limit = { exists = scope:new_divine_mission_scope }
		scope:new_divine_mission_scope = {
			set_blackspawn_mission_difficulty_effect = yes
			
			if = {
				limit = { exists = local_var:mission_difficulty_remaining }
				random_list = {
					1 = {	# Murder someone
						set_variable = { name = mission_type value = flag:mission_type_kill }
						while = {
							limit = { local_var:mission_difficulty_remaining > 0 }
							generate_blackspawn_kill_mission_attributes_effect = yes
							change_local_variable = {
								name = mission_difficulty_remaining
								subtract = 2
							}
						}
					}
					1 = {	#Raid province
						set_variable = { name = mission_type value = flag:mission_type_raid }
						while = {
							limit = { local_var:mission_difficulty_remaining > 0 }
							generate_blackspawn_raid_mission_attributes_effect = yes
							change_local_variable = {
								name = mission_difficulty_remaining
								subtract = 2
							}
						}
					}
					#1 = {	#Send children to pit
					#	trigger = { root.capital_province = { geographical_region = GH_geographic_redlands_mainland }}
					#	set_variable = { name = mission_type value = flag:mission_type_child_pit }
					#	while = {
					#		limit = { local_var:mission_difficulty_remaining > 0 }
					#		generate_blackspawn_child_pit_mission_attributes_effect = yes
					#		change_local_variable = {
					#			name = mission_difficulty_remaining
					#			subtract = 2
					#		}
					#	}
					#}
				}
			}
		}
		add_to_variable_list = {
			name = ongoing_missions
			target = scope:new_divine_mission_scope
		}
	}
}

blackspawn.0002 = { # Murder Divine Mission
	type = character_event
	hidden = yes
	content_source = dlc_gh
	immediate = {
		every_in_list = {
			variable = ongoing_missions
			limit = {
				var:mission_type ?= flag:mission_type_kill
				trigger_if = {
					limit = { exists = var:mission_target_primary_title_tier }
					scope:mission_target.highest_held_title_tier >= var:mission_target_primary_title_tier
				}
				trigger_if = {
					limit = { exists = var:mission_target_virtue_number }
					scope:mission_target.num_virtuous_traits >= var:mission_target_virtue_number
				}
				trigger_if = {
					limit = { exists = var:mission_kill_method }
					OR = {
						AND = {
							scope:mission_target = { death_reason = death_murder }
							var:mission_kill_method = flag:murder
						}
						AND = {
							scope:mission_target = { death_reason = death_ritual }
							var:mission_kill_method = flag:sacrifice
						}
						AND = {
							scope:mission_target = { death_reason = death_magic_targeted_spell }
							var:mission_kill_method = flag:magic_targeted_spell
						}
					}
				}
			}
			if = {
				limit = { NOT = { exists = var:mission_multiplicity } }
				save_scope_as = mission_scope
				root = { trigger_event = blackspawn.0011 }
			}
			else_if = { # Would the code logic be improved if we set this variable at the beginning?
				limit = { NOT = { exists = var:completed_subtasks } }
				set_variable = {
					name = completed_subtasks
					value = 1
				}
				if = {  # Kill missions with 1 target were not completing properly, copied the subtasks check code here
					limit = { var:completed_subtasks >= var:mission_multiplicity }
					save_scope_as = mission_scope
					root = { trigger_event = blackspawn.0011 }
				}
				else = {
					root = {
						send_interface_toast = {
							title = blackspawn_mission_update_title
							left_icon = this
							decrease_divine_mission_penalties_effect = yes
						}
					}
				}
			}
			else = {
				change_variable = {
					name = completed_subtasks
					add = 1
				}
				if = {
					limit = { var:completed_subtasks >= var:mission_multiplicity }
					save_scope_as = mission_scope
					root = { trigger_event = blackspawn.0011 }
				}
				else = {
					root = {
						send_interface_toast = {
							title = blackspawn_mission_update_title
							left_icon = this
							decrease_divine_mission_penalties_effect = yes
						}
					}
				}
			}
		}
	}
}

	random_list = {
		5 = {
			save_scope_value_as = {
				name = mission_reward_scope
				value = flag:dark_aspect_increase
			}
		}
		3 = {
			save_scope_value_as = {
				name = mission_reward_scope
				value = flag:arcana_increase
			}
		}
		3 = {
			save_scope_value_as = {
				name = mission_reward_scope
				value = flag:intrigue_increase
			}
		}
		0 = {
			modifier = {
				add = var:accrued_blackspawn_trait_progression_chance
				exists = var:accrued_blackspawn_trait_progression_chance
			}
			save_scope_value_as = {
				name = mission_reward_scope
				value = flag:blackspawn_trait_progression
			}
		}
	}
}

blackspawn.0011 = {
	type = character_event
	content_source = dlc_gh
	title = blackspawn.0011.t
	desc = blackspawn.0011.desc
	theme = witchcraft
	override_background = { reference = terrain_scope }
	
	left_portrait = {
		character = root
		animation = happiness
	}
	
	immediate = {
		title:c_ofnammm.title_province = { save_scope_as = background_terrain_scope }
		pick_blackspawn_mission_reward_effect = yes
	}
	
	option = {	# Dark aspect increase
		name = blackspawn.0011.a
		
		trigger = {
			OR = {
				NOT = { exists = scope:mission_reward_scope }	# Fallback if for some mysterious reason no reward is picked
				scope:mission_reward_scope = flag:dark_aspect_increase
			}
		}
		
		add_aspect_skill_effect = { ASPECT = dark VALUE = 1 }
	}
	
	option = {	# Arcana increase
		name = blackspawn.0011.b
		
		trigger = {
			scope:mission_reward_scope ?= flag:arcana_increase
		}
		
		add_arcana_skill = { VALUE = 1 }
	}
	
	option = {	# Intrigue increase
		name = blackspawn.0011.c
		
		trigger = {
			scope:mission_reward_scope ?= flag:intrigue_increase
		}
		
		add_intrigue_skill = 1
	}
	
	option = {	# Blackspawn rank increase
		name = blackspawn.0011.d
		
		trigger = {
			scope:mission_reward_scope ?= flag:blackspawn_trait_progression
		}
		
		change_trait_rank = {
			trait = blackspawn
			rank = 1
		}
	}
	
	after = {
		if = {
			limit = { scope:mission_reward_scope ?= flag:blackspawn_trait_progression }
			remove_variable = accrued_blackspawn_trait_progression_chance
		}
		else = {
			if = {
				limit = { NOT = { exists = var:accrued_blackspawn_trait_progression_chance } }
				set_variable = {
					name = accrued_blackspawn_trait_progression_chance
					value = 1
				}
			}
			else = {
				change_variable = {
					name = accrued_blackspawn_trait_progression_chance
					add = 1
				}
			}
		}
		
		if = {
			limit = { exists = scope:mission_scope }
			scope:mission_scope = { end_story = yes }
		}
	}
}

# Initialization, picks a random epicenter
gedeora.0001 = {
	scope = none
	hidden = yes
	
	trigger = {
		NOT = { has_global_variable = recent_gedeora }
	}
	
	immediate = {
		random_county = {
			limit = {
				NOT = { religion = religion:wasteland_religion }
				is_nomad_title = no # PDX, why?
			}
			save_scope_as = gedeora_epicenter
			if = {
				limit = { NOT = { has_game_rule = gedeora_frequency_armageddon } }
				set_global_variable = {
					name = recent_gedeora
					value = scope:gedeora_epicenter
					days = 90
				}
			}
			else = {
				set_global_variable = {
					name = recent_gedeora
					value = scope:gedeora_epicenter
					days = 10
				}
			}
			pick_gedeora_primary_effect = yes
			pick_gedeora_target_populace = yes
			pick_gedeora_secondary_effect = yes
			random_list = {
				75 = { 
					set_variable = { # Needs to be local since several Gedeora variables might overlap in the 90 day cooldown
						name = small_gedeora
						days = 90
					}
					modifier = {
						has_game_rule = gedeora_severity_hiccups
						factor = 2
					}
					modifier = {
						has_game_rule = gedeora_severity_dangerous
						factor = 0.5
					}
					modifier = {
						OR = {
							has_game_rule = gedeora_severity_severe
							has_game_rule = gedeora_severity_ridiculous
						}
						add = -74
					}
				}
				15 = { 
					set_variable = {
						name = medium_gedeora
						days = 90
					}
					modifier = {
						has_game_rule = gedeora_severity_hiccups
						factor = 2
					}
					modifier = {
						has_game_rule = gedeora_severity_dangerous
						factor = 0.5
					}
					modifier = {
						OR = {
							has_game_rule = gedeora_severity_severe
							has_game_rule = gedeora_severity_ridiculous
						}
						add = -11
					}
				}
				5 = { 
					set_variable = {
						name = big_gedeora
						days = 90
					}
					modifier = {
						has_game_rule = gedeora_severity_ridiculous
						factor = 10
					}
				}
				4 = { 
					set_variable = {
						name = huge_gedeora
						days = 90
					}
					modifier = {
						has_game_rule = gedeora_severity_severe
						add = 11
					}
					modifier = {
						has_game_rule = gedeora_severity_ridiculous
						factor = 100
					}
				}
				1 = {
					set_variable = {
						name = global_gedeora
						days = 90
					}
					modifier = {
						has_game_rule = gedeora_severity_severe
						add = 74
					}
					modifier = {
						has_game_rule = gedeora_severity_ridiculous
						factor = 1000
					}
				} 
			}
			# Since gedeora_distance_value is randomized, we want to save one instance so it doesn't fluctuate for the same Gedeora
			# For some reason trying to used fixed_value inside a script_value doesn't work, so I'm making a scripted_effect instead
			set_gedeora_distance = yes
			if = {
				limit = { exists = var:gedeora_distance } # to make sure it exists, and prevent a literal 1 million errors
				every_living_character = {
					limit = {
						exists = location
						is_ruin = no
						OR = {
							has_variable = global_gedeora
							location = {
								squared_distance = {
									target = scope:gedeora_epicenter.title_province
									value <= scope:gedeora_epicenter.var:gedeora_distance
								}
							}
						}
					}
					trigger_event = gedeora.0010
				}
			}
		}
	}	
}

		name = {

		}
		name = {
			trigger = { scope:gedeora_epicenter = { has_variable = generic_evil_gedeora } }

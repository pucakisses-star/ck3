scripted_trigger gh_health_event_is_worried_trigger = {
	OR = {
		ai_compassion >= medium_positive_ai_value
		dread_modified_ai_boldness = {
			dreaded_character = $CHARACTER$
			value <= medium_negative_ai_value
		}
		has_relation_friend = $CHARACTER$
		has_relation_lover = $CHARACTER$
		opinion = {
			target = $CHARACTER$
			value >= high_positive_opinion
		}
	}
	$CHARACTER$.health < fine_health
}

# THE MIASMA
# ALL WHO DO NOT KILL THE SPREADING SHADOW SHALL SUFFER IN ITS DARKNESS
# LET HE WHO SEEKS THE LIGHT GUIDE US
# AND DELIVER US FROM THE SUFFERING
# PRAISE THE MAKER
# - a proverb of unknown origin, found written in blood after the Shaded City of Virayl's destruction

gh_health.5998 = { #The Miasma story cycle pacing control
	hidden = yes

	immediate = {
		if = {
			limit = {
				any_neighboring_top_liege_realm_owner = {
					has_character_flag = shade_stricken_splash
				}
			}
			if = {
				limit = {
					has_character_flag = shade_stricken_closer
				}
				remove_character_flag = shade_stricken_closer
			}
			add_character_flag = shade_stricken_nearby
		}
		else_if = {
			limit = {
				any_neighboring_top_liege_realm_owner = {
					any_neighboring_top_liege_realm_owner = {
						has_character_flag = shade_stricken_splash
					}
				}
			}	
			add_character_flag = shade_stricken_closer
		}
		if = {
			limit = {
				OR = {
					has_character_flag = shade_stricken_closer
					has_character_flag = shade_stricken_nearby
				}
				NOT = {
					has_character_flag = shade_stricken_info_event
				}
			}
			trigger_event = {
				id = gh_health.5999
				days = { 2 4 }
			}
		}
	}
}

# The Miasma is closing in
# Used to trigger The Miasma Story cycle
# by Nick Meredith
gh_health.5999 = {

	title = gh_health.5999.t

	theme = plague
	content_source = dlc_gh
	override_background = { reference = study }
	left_portrait = {
		character = root
		animation = worry
	}
	cooldown = { years = 100 }

	trigger = {
		is_ai = no
		#has_dlc_feature = legends_of_the_dead
		is_available_adult = yes

			any_sub_realm_county = { #Black Plague is already here, no need to talk about it closing in

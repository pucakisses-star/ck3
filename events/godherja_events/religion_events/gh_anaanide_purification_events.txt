# Your High Guide Disapproves
# by ScionEol
anaanides_purification.1001 = {
	type = character_event
	content_source = dlc_gh
	title = anaanides_purification.1001.t
	desc = anaanides_purification.1001.desc
	theme = learning_theology_focus
	left_portrait = {
		character = scope:priest
		animation = disapproval
	}

	immediate = {
		cp:councillor_court_chaplain = {
			save_scope_as = priest
		}
	}

	option = { # Disregard the High Guide
		name = anaanides_purification.1001.a
		scope:priest = {
			add_opinion = {
				target = root
				modifier = impious_opinion
				opinion = -20
			}
		}
		add_prestige = 100
		add_piety = -150
		ai_chance = {
			base = 100
		}
	}

	option = { # Heed the High Guide
		name = anaanides_purification.1001.b
		scope:priest = {
			add_opinion = {
				target = root
				modifier = pious_opinion
				opinion = 10
			}
		}
		add_prestige = -100
		add_piety = 150
		ai_chance = {
			base = 30
		}
	}
}

# A Vassal Disapproves
# by ScionEol
scripted_trigger anaanides_purification_1003_vassal_trigger = {
	is_available = yes
	faith = root.faith
	has_trait = diligent
	NOT = {
		has_trait = lazy
	}
}

anaanides_purification.1003 = {
	type = letter_event
	opening = anaanides_purification.1003.opening
	desc = anaanides_purification.1003.desc
	sender = {
		character = scope:vassal
		animation = disapproval
	}
	
	trigger = {
		any_vassal = {
			anaanides_purification_1003_vassal_trigger = yes
		}
	}

	immediate = {
		random_vassal = {
			limit = {
				NOR = {
					has_relation_rival = root
					has_relation_nemesis = root
				}
				anaanides_purification_1003_vassal_trigger = yes
			}
			alternative_limit = {
				anaanides_purification_1003_vassal_trigger = yes
			}
			weight = {
				base = 1
				modifier = {
					add = 10
					is_of_major_interest_to_root_trigger = yes
				}
				modifier = {
					add = 5
					is_of_minor_interest_to_root_trigger = yes
				}
			}
			save_scope_as = vassal
		}
	}

	option = { # Disregard Vassal
		name = anaanides_purification.1003.a
		progress_towards_rival_effect = {
			CHARACTER = scope:vassal
			OPINION = 0
			REASON = rival_called_out_sin
		}
		scope:vassal = {
			add_opinion = {
				target = root
				modifier = impious_opinion
				opinion = -20
			}
		}
		ai_chance = {
			base = 100
		}
	}

	option = { # Back out
		name = anaanides_purification.1001.b
		scope:vassal = {
			add_opinion = {
				target = root
				modifier = pious_opinion
				opinion = 10
			}
		}
		ai_chance = {
			base = 30
		}
	}
}

# A Vassal Approves
# by ScionEol
scripted_trigger anaanides_purification_1005_vassal_trigger = {
	is_available_ai = yes
	faith = root.faith
	has_trait = zealous
}

anaanides_purification.1005 = {
	type = letter_event
	opening = anaanides_purification.1005.opening
	desc = anaanides_purification.1005.desc
	sender = {
		character = scope:vassal
		animation = admiration
	}
	
	trigger = {
		any_vassal = {
			anaanides_purification_1005_vassal_trigger = yes
		}
	}

	immediate = {
		random_vassal = {
			limit = {
				anaanides_purification_1005_vassal_trigger = yes
			}
			weight = {
				base = 1
				modifier = {
					add = 10
					is_of_major_interest_to_root_trigger = yes
				}
				modifier = {
					add = 5
					is_of_minor_interest_to_root_trigger = yes
				}
			}
			save_scope_as = vassal
		}
	}

	option = { # Yes, I am very devout
		name = anaanides_purification.1005.a
		scope:vassal = {
			add_opinion = {
				target = root
				modifier = pious_opinion
				opinion = 30
			}
		}
		ai_chance = {
			base = 100
		}
	}

	option = { # Let me show you what I'm doing!
		name = anaanides_purification.1005.b
		progress_towards_friend_effect = {
			CHARACTER = scope:vassal
			OPINION = 0
			REASON = friend_guided_right_path
		}
		scope:vassal = {
			add_piety = 150
		}
		ai_chance = {
			base = 50
		}
	}
}

# An Unexpected Miracle
# by ScionEol
anaanides_purification.1008 = {
	type = character_event
	content_source = dlc_gh
	title = anaanides_purification.1008.t
	desc = anaanides_purification.1008.desc
	theme = learning_theology_focus
	left_portrait = {
		character = root
		animation = happiness
	}
	
	trigger = {
		OR = {
			has_trait = ill
			has_trait = pneumonic
		}
	}

	option = { # Phew!
		name = anaanides_purification.1008.a
		if = {
			limit = { has_trait = ill }
			recover_from_disease_effect = {
				DISEASE = ill
			}
		}
		if = {
			limit = { has_trait = pneumonic }
			recover_from_disease_effect = {
				DISEASE = pneumonic
			}
		}
	}
}

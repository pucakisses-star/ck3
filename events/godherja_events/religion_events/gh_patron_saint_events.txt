# 0001 - Battle Patron Saint Affinity Increase   #
# 0101-0200 - Medical Experiment Scheme          #
# 0201-0300 - Heal The Sick Decision			 #
##################################################

patron_saint_event.0001 = {
	hidden = yes
	scope = combat_side

	trigger = {
		exists = side_commander
		side_commander = { 
			is_alive = yes 
			character_has_any_patron_saint = yes
		}
	}

	immediate = {
		save_scope_as = relevant_combat_side
		side_commander = { 
			if = {
				limit = { any_character_patron_saint_has_aspect_trigger = { ASPECT = values_battle_victory } }
				patron_saint_affinity_from_battle_victory_check_effect = yes
			}
			else_if = {
				limit = { any_character_patron_saint_has_aspect_trigger = { ASPECT = values_battle_slaughter } }
				patron_saint_affinity_from_battle_slaughter_check_effect = yes
			}
		}
	}
}

patron_saint_event.0002 = {
	hidden = yes
	scope = army

	trigger = {
		scope:raider = {
			character_has_any_patron_saint = yes
			any_character_patron_saint_has_aspect_trigger = { ASPECT = values_raiding }
		}
	}

	immediate = {
		scope:raider = {
			if = { 
				limit = { 
					NOT = { has_variable = accumulated_raid_loot }
				}
				set_variable = {
					name = accumulated_raid_loot
					value = root.raid_loot
				}
			}
			else = {
				change_variable = {
					name = accumulated_raid_loot
					add = root.raid_loot
				}
			}
		}
	}
}

patron_saint_event.0003 = {
	hidden = yes
	scope = army

	trigger = {
		scope:raider = {
			character_has_any_patron_saint = yes
			any_character_patron_saint_has_aspect_trigger = { ASPECT = values_raiding_antislavery }
		}
		scope:county.faith = {
			OR = {
				has_doctrine = doctrine_slavery_encouraged
				has_doctrine = doctrine_slavery_demanded
			}
		}
	}

	immediate = {
		save_scope_value_as = {
			name = antislavery_points
			value = {
				value = scope:county.development_level
				divide = 10
				add = 5
			}
		}
		scope:raider = {
			if = { 
				limit = { 
					NOT = { has_variable = accumulated_antislave_raids }
				}
				set_variable = {
					name = accumulated_antislave_raids
					value = scope:antislavery_points
				}
			}
			else = {
				change_variable = {
					name = accumulated_antislave_raids
					add = scope:antislavery_points
				}
			}
		}
	}
}

# Medical Experiments Success
patron_saint_event.0101 = {
	type = character_event
	content_source = dlc_gh
	title = patron_saint_event.0101.t
	desc = patron_saint_event.0101.desc
	theme = prison
	
	immediate = {
		random_court_position_holder = {
			type = court_physician_court_position
			save_scope_as = court_physician
		}
		scope:target = { save_scope_as = victim }
		scope:victim = {
			medical_experiment_damage_effects = yes
		}
		medical_experiments_success_experience_effect = yes
		medical_experiment_stress_effect = yes
	}
	
	left_portrait = {
		character = scope:victim
		animation = fear

	}
	right_portrait = {
		character = root
		animation = schadenfreude
	}
	widget = {
		gui = "event_window_widget_scheme"
		container = "custom_widgets_container"
	}

	option = {
		name = patron_saint_event.0101.a
		
		scope:scheme = {
			end_scheme = yes
		}
		
		medical_experiments_kill_victim_if_necessary = yes
		
		ai_chance = {
			base = 100
		}
	}
	
}

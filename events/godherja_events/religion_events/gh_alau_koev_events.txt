			craven = major_stress_impact_gain
			cynical = medium_stress_impact_gain
		}
		add_trait = disfigured
		add_piety = 500
		add_prestige = 750

		trigger_event = {
			id = alau_koev.0004
			days = 2
		}

		name = alau_koev.0002.d

					title = alau_koev.0002.d.success.t

					add_piety = minor_piety_gain
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}

				desc = alau_koev.0002.d.failure
				send_interface_toast = {
					title = alau_koev.0002.d.failure.t

			}

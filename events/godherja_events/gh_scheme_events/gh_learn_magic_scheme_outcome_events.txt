# Learn Magic School Success
learn_magic_outcome.0001 = {
	type = character_event
	content_source = dlc_gh
	title = learn_magic_outcome.0001.t
	desc = learn_magic_outcome.0001.desc
	theme = education
	left_portrait = {
		character = root
		animation = happiness
	}
	right_portrait = {
		character = scope:target
		animation = personality_rational
	}
	widget = {
		gui = "event_window_widget_scheme"
		container = "custom_widgets_container"
	}

	option = {
		name = learn_magic_outcome.0001.a
		
		
		learn_magic_school_effect = { MAGIC_SCHOOL = var:magic_school_taught_by_scheme }
		learn_magic_knowledge_from_character_and_magic_school_effect = { MIN_SPELLS = 2 MAX_SPELLS = 2 MAX_SPELL_LEVEL = 4 MENTOR = scope:target MAGIC_SCHOOL = scope:owner.var:magic_school_taught_by_scheme }
		remove_variable = magic_school_taught_by_scheme
		scope:scheme = {
			end_scheme = yes
		}
		
		ai_chance = {
			base = 100
		}
	}
	
	after = {
		add_to_magi_list = yes
	}
}

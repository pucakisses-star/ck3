# Consider changing all of the below (and vanilla equivalents) to non-inline scripted effects, since they create non-reproducible errors that may cause desyncs

# Syncs stress lost from successfully completing a non-criminal contract scheme.
scripted_effect gh_laamp_base_contract_schemes_acknowledge_success_stress_effect = { # Multiplayer desync candidate, equivalent to the non-gh version
	stress_impact = { arrogant = minor_stress_impact_loss }
}

# Syncs stress gained from failing a non-criminal contract scheme.
scripted_effect gh_laamp_base_contract_schemes_acknowledge_failure_stress_effect = { # Same as above
	if = {
		limit = { scope:task_contract_tier >= 2 }
		stress_impact = {
			base = miniscule_stress_impact_gain
			humble = minor_stress_impact_loss
			arrogant = minor_stress_impact_gain
		}
	}
}

# Syncs stress gained from successfully completing a non-criminal contract scheme.
scripted_effect gh_laamp_base_contract_schemes_acknowledge_success_criminal_stress_effect = { # Same as above
	stress_impact = {
		arrogant = minor_stress_impact_loss
		arbitrary = minor_stress_impact_loss
		just = medium_stress_impact_gain
	}
}

# Syncs stress gained from failing a non-criminal contract scheme.
scripted_effect gh_laamp_base_contract_schemes_acknowledge_failure_criminal_stress_effect = { # Same as above
	stress_impact = {
		base = minor_stress_impact_gain
		humble = minor_stress_impact_loss
		arrogant = minor_stress_impact_gain
		just = medium_stress_impact_gain
	}
}

# Syncs arbitrary duel difficulty + rewards to the tier of a contract, so that the same effect can dish out scaling rewards/have scaling odds if you're impressing a king rather than a baron.
scripted_effect gh_laamp_base_contract_schemes_check_contract_difficulty_and_rewards_effect = { # Same as above
	task_contract = {
		if = {
			limit = { task_contract_tier >= task_contract_t3_value }
			save_scope_value_as = {
				name = duel_diff
				value = @laamp_base_contract_schemes_on_going_duel_diff_t3_value
			}
			save_scope_value_as = {
				name = prestige_gain
				value = major_prestige_gain
			}
			save_scope_value_as = {
				name = piety_gain
				value = major_piety_gain
			}
			save_scope_value_as = {
				name = provisions_gain
				value = massive_provisions_gain
			}
		}
		else_if = {
			limit = { task_contract_tier >= task_contract_t2_value }
			save_scope_value_as = {
				name = duel_diff
				value = @laamp_base_contract_schemes_on_going_duel_diff_t2_value
			}
			save_scope_value_as = {
				name = prestige_gain
				value = medium_prestige_gain
			}
			save_scope_value_as = {
				name = piety_gain
				value = medium_piety_gain
			}
			save_scope_value_as = {
				name = provisions_gain
				value = major_provisions_gain
			}
		}
		else = {
			save_scope_value_as = {
				name = duel_diff
				value = @laamp_base_contract_schemes_on_going_duel_diff_t1_value
			}
			save_scope_value_as = {
				name = prestige_gain
				value = minor_prestige_gain
			}
			save_scope_value_as = {
				name = piety_gain
				value = minor_piety_gain
			}
			save_scope_value_as = {
				name = provisions_gain
				value = medium_provisions_gain 
			}
		}
	}
}

	}
	right_portrait = {
		character = scope:employer
		animation = happiness
	}
	override_background = { reference = alley_day }

	immediate = {
		scope:employer = {
			random_dummy_gender_non_combatant_effect = { SCOPE_NAME = dummy_gender }
		}
		if = {
			limit = { exists = scope:task_contract.var:animal_type }
			set_variable = {
				name = animal_type
				value = scope:task_contract.var:animal_type
			}
		}
		else = {
			select_local_animal_effect = { TYPE = fodder }
		}
		laamp_base_contract_schemes_set_up_outcome_immediate_effect = yes
		random_list = {
			67 = {
				scope:scheme.task_contract = { complete_task_contract = success_standard }
			}
			33 = {
				scope:scheme.task_contract = { complete_task_contract = success_critical }
			}
		}
		# Employer outcome.
		scope:employer = {
			add_stress = major_stress_loss
		}
		# And wrap things up.
		scope:scheme ?= { end_scheme = yes }
	}

	# Acknowledge.
	option = {
		name = gh_laamp_base_contract_schemes.1002.a
		flavor = laamp_base_contract_schemes.0000.f.contract_concluded

		gh_laamp_base_contract_schemes_acknowledge_success_stress_effect = yes
	}
}

# Escort Refugees
gh_laamp_base_contract_schemes.9001 = {
	type = character_event
	title = gh_laamp_base_contract_schemes.1001.t
	desc = gh_laamp_base_contract_schemes.1001.desc
	theme = martial
	left_portrait = {
		character = root
		animation = marshal
	}
	right_portrait = {
		character = scope:employer
		animation = worry
	}
	override_background = { reference = terrain }
	widget = {
		gui = "event_window_widget_scheme"
		container = "custom_widgets_container"
	}

	immediate = {
		scope:task_contract.task_contract_employer = {
			save_scope_as = employer
			save_scope_as = bg_override_char
		}
		random_dummy_gender_soldier_effect = { SCOPE_NAME = dummy_gender_soldier }
	}

	# focus on successful agents
	option = {
		name = gh_laamp_base_contract_schemes.9001.a
		reason = give_me_success_agents
		flavor = EVENT_WINDOW_OPTION_AGENTS_SUCCESS

		scope:scheme = {
			# Success
			add_agent_slot = agent_outrider
			add_agent_slot = agent_outrider
			add_agent_slot = agent_scout
			add_agent_slot = agent_comrade_in_arms
			# Speed
			add_agent_slot = agent_supplier
		}
	}
	# Focus on speedy agents
	option = {
		name = gh_laamp_base_contract_schemes.9001.b
		reason = give_me_speed_agents
		flavor = EVENT_WINDOW_OPTION_AGENTS_SUCCESS

		scope:scheme = {
			# Success
			
			add_agent_slot = agent_outrider

			# Speed
			add_agent_slot = agent_lookout_speed
			add_agent_slot = agent_lookout_speed
			add_agent_slot = agent_supplier
			add_agent_slot = agent_supplier
		}
	}
	# Go for a balanced agent mix
	option = {
		name = gh_laamp_base_contract_schemes.9001.c
		reason = give_me_balanced_agents
		flavor = EVENT_WINDOW_OPTION_AGENTS_BALANCED

		scope:scheme = {
			# Success
			add_agent_slot = agent_outrider
			add_agent_slot = agent_scout
			add_agent_slot = agent_comrade_in_arms
			# Speed
			add_agent_slot = agent_supplier
			add_agent_slot = agent_lookout_speed
		}
		# AI always selects balanced agents
		ai_chance = { base = 1000 }
	}

	# Now show us the panel.
	after = {
		open_view = {
			view = intrigue_window
			player = root
		}
	}
}

#	Success event.
gh_laamp_base_contract_schemes.9002 = {
	type = character_event
	window = scheme_successful_event
	title = gh_laamp_base_contract_schemes.9002.t
	desc = gh_laamp_base_contract_schemes.9002.desc
	theme = martial
	center_portrait = {
		character = root
		animation = marshal
	}
	right_portrait = {
		character = scope:employer
		animation = happiness
	}
	override_background = { reference = courtyard }

	immediate = {
		laamp_base_contract_schemes_set_up_outcome_immediate_effect = yes
		random_list = {
			67 = {
				scope:scheme.task_contract = { complete_task_contract = success_standard }
			}
			33 = {
				scope:scheme.task_contract = { complete_task_contract = success_critical }
			}
		}
		# Employer outcome.
		scope:employer = {
			if = {
				limit = { contract_employer_can_apply_county_modifier_trigger = yes }
				capital_county = {
					add_county_modifier = {
						modifier = laamp_migration_9001_success_employer_modifier
						years = 10
					}
				}
			}
		}
		# And wrap things up.
		scope:scheme ?= { end_scheme = yes }
	}

	# Acknowledge.
	option = {
		name = gh_laamp_base_contract_schemes.9002.a
		flavor = laamp_base_contract_schemes.0000.f.contract_concluded

		gh_laamp_base_contract_schemes_acknowledge_success_stress_effect = yes
	}
}

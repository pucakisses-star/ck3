namespace = sophocos_quest

scripted_trigger is_able_to_climb = { # Checks if the character can handle the climbing
	NOR = {
		has_trait = wounded_3
		has_trait = incapable
		has_trait = infirm
		is_imprisoned = yes
	}
}

scripted_effect sophocos_quest_climb_reward = { # Handle progression level
	if = { # Continue ascending
		limit = { root.var:sophocos_tier < 5 }
		custom_tooltip = sophocos_quest_climb.success.tt
		change_variable = {
			name = sophocos_tier
			add = 1
		}
		trigger_event = {
			on_action = sophocos_quest_ongoing
			days = 1
		}
	}
	else = { # Reach the top
		custom_tooltip = sophocos_quest_climb.final.tt
		trigger_event = {
			on_action = sophocos_quest_final
			days = 1
		}
	}
}

scripted_effect sophocos_quest_climb = { # Try to ascend further
	duel = {
		skill = prowess
		value = medium_skill_rating
		50 = { # Success
			desc = sophocos_quest_climb.success.desc
			compare_modifier = { # Climbing gets progressively harder
				value = {
					value = scope:duel_value
					subtract = {
						value = 25
						multiply = root.var:sophocos_tier
					}
				}
			}
			sophocos_quest_climb_reward = yes
		}
		50 = { # Failure
			desc = sophocos_quest_climb.failure.desc
			sophocos_quest_abandon_climb = yes
		}
	}
}



scripted_effect sophocos_quest_abandon_climb = { # You stop climbing and leave
	custom_tooltip = sophocos_quest_abandon.tt
	send_interface_toast = { # Inform the player that they have failed
		title = sophocos_quest_abandon_climb.name
		left_icon = root
		add_piety = {
			value = major_piety_value
			multiply = root.var:sophocos_tier
		}
		give_pilgrim_trait_or_xp_effect = yes
		pilgrimage_arrival_stress_loss_effect = yes
	}
	remove_variable = sophocos_tier
}

scripted_effect physical_health_change = {
	if = {
		limit = { has_trait = physique_good }
		change_trait_rank = {
			trait = physique_good
			rank = $VAL$
			max = 3
		}
	}
	else_if = {
		limit = {
			has_trait_rank = {
				trait = physique_bad
				rank = 3
			}
		}
		if = {
			limit = { $VAL$ > 0 }
			change_trait_rank = {
				trait = physique_bad
				rank = -1
			}
		}
		else = { death = { death_reason = death_physique_bad_3 } }
	}
	else_if = {
		limit = { has_trait = physique_bad }
		change_trait_rank = {
			trait = physique_bad
			rank = {
				value = $VAL$
				multiply = -1
			}
		}
	}
	else = { add_trait = physique_good_1 }
}

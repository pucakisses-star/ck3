	type = character_event
	content_source = dlc_gh

	title = gh_acw_yearly.0002.t

	left_portrait = {
		character = root

	}
	
	right_portrait = {
		character = scope:amsari_pretender
		animation = anger
	}

	trigger = {
		gh_basic_gh_acw_yearly_flavor_trigger = yes
		days_of_continuous_peace >= 1095
		any_neighboring_top_liege_realm_owner = {
			gh_acw_basic_declare_war_trigger = yes
			this.max_military_strength <= prev.max_military_strength
			save_temporary_scope_as = potential_defender
			prev = {
				can_declare_war = {
					casus_belli = gh_acw_enforce_pretender_cb
					defender = scope:potential_defender
				}
			}
		}		
	}

	cooldown = { years = 20 }
	
	immediate = {
		root.var:gh_amsari_civil_war_info_story.var:amsari_civil_war_supporting ?= { save_scope_as = amsari_pretender }
		scope:amsari_pretender ?= {
			add_opinion = {
				target = root
				modifier = annoyed_opinion
				opinion = -10
			}
		}
		random_neighboring_top_liege_realm_owner = {
			limit = {
				gh_acw_basic_declare_war_trigger = yes
				this.max_military_strength <= prev.max_military_strength
				save_temporary_scope_as = potential_defender
				prev = {
					can_declare_war = {
						casus_belli = gh_acw_enforce_pretender_cb
						defender = scope:potential_defender
					}
				}
			}
			save_scope_as = defending_ruler
		}
	}

	option = {
		name = gh_acw_yearly.0002.a
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_compassion = 2
				ai_vengefulness = -0.5
			}
		}
	}

	option = {
		name = gh_acw_yearly.0002.b
		duel = {
			target = scope:amsari_pretender
			skill = diplomacy
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 1.5
					min = -49
				}
				modifier = {
					tier_difference = { target = root value >= 1 }
					add = 10
				}
				desc = gh_acw_yearly.0002.b.success
				send_interface_toast = {
					title = gh_acw_yearly.0002.b.success
					left_icon = scope:other
					add_dread = minor_dread_gain
					scope:amsari_pretender ?= {
						add_opinion = {
							target = root
							modifier = gh_acw_backed_down
							opinion = 20
						}
					}
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				desc = gh_acw_yearly.0002.b.failure
				send_interface_toast = {
					title = gh_acw_yearly.0002.b.failure
					left_icon = scope:other
					progress_towards_rival_effect = {
						REASON = rival_concerns_dismissed
						CHARACTER = scope:amsari_pretender
						OPINION = default_rival_opinion
					}
				}
			}
		}
		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_compassion = -1
				ai_vengefulness = -1
			}
		}
	}
	
	option = {
		name = gh_acw_yearly.0002.c
		scope:amsari_pretender ?= {
			add_opinion = {
				target = root
				modifier = grateful_opinion
				opinion = 20
			}
		}

			cb = gh_acw_enforce_pretender_cb

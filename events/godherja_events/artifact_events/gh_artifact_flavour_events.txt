	type = character_event
	content_source = dlc_gh
	title = artifacts_flavor.1000.t
	desc = artifacts_flavor.1000.desc
	theme = crown
	left_portrait = {
		character = root
		animation = beg
	}
	artifact = {
		target = scope:dancing_automata
		position = lower_center_portrait
	}
	
	trigger = {
		any_character_artifact = { has_variable = dancing_automata }
	}

	immediate = {
		every_character_artifact = {
			limit = { has_variable = dancing_automata }
			save_scope_as = dancing_automata
		}
	}

	option = {
		name = artifacts_flavor.1000.a
		root = {
			add_stress = 35
		}
		stress_impact = {
			paranoid = minor_stress_impact_gain
		}
	}
}

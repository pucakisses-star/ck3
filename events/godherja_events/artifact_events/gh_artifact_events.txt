#	A book seller visits your court
gh_artifact.0002 = {
	type = character_event
	content_source = dlc_gh
	title = gh_artifact.0002.t
	desc = gh_artifact.0002.desc
	theme = learning_scholarship_focus
	orphan = yes
	left_portrait = {
		character = root
		animation = personality_rational
	}
	right_portrait = {
		character = scope:book_seller
		animation = personality_rational
	}
	artifact = {
		target = scope:first_displayed_artifact
		position = lower_left_portrait
		trigger = { exists = scope:first_displayed_artifact }
	}
	artifact = {
		target = scope:second_displayed_artifact
		position = lower_center_portrait
		trigger = { exists = scope:second_displayed_artifact }
	}
	artifact = {
		target = scope:third_displayed_artifact
		position = lower_right_portrait
		trigger = { exists = scope:third_displayed_artifact }
	}

	trigger = {
		#Standard checks.
		is_available_even_at_war_adult = yes
		NOT = { has_variable = recently_visited_by_book_seller }
		is_landed = yes
		is_human = yes
		exists = scope:book_seller
	}

	immediate = {
		set_variable = {
			name = recently_visited_by_book_seller
			days = 3650
		}
		hidden_effect = {
			random_county = {
				limit = {
					squared_distance = {
						target = root.capital_county
						value <= squared_distance_huge
					}
					NOR = {
						culture = culture:wasteland
						religion = religion:wasteland_religion
					}
				}
				save_scope_as = book_seller_origin_county
			}
			#Create the book seller.
			create_character = {
				location = root.capital_province
				template = book_seller_template
				save_scope_as = book_seller
			}
			scope:book_seller = { 
				generate_random_historical_book_effect = yes
				scope:newly_created_artifact = { save_scope_as = first_displayed_artifact }
				generate_random_historical_book_effect = yes
				scope:newly_created_artifact = { save_scope_as = second_displayed_artifact }
				generate_random_historical_book_effect = yes
				scope:newly_created_artifact = { save_scope_as = third_displayed_artifact }
			}
		}
	}
	
	option = {
		name = gh_artifact.0002.steal_all_books
		trigger = {  
			OR = {
				has_trait = sadistic
				has_trait = callous
				has_trait = arbitrary
			}
			NOT = { has_trait = just }
		}
		add_prestige = major_prestige_loss
		if = {
			limit = {
				faith = {
					NOR = {
						trait_is_virtue = sadistic
						trait_is_virtue = callous
						trait_is_virtue = arbitrary
					}
				}
			}
			add_piety = major_piety_loss
		}
		scope:book_seller = {
			every_character_artifact = {
				set_owner = root
			}
		}
	}
	
	option = {
		name = gh_artifact.0002.a
		trigger = { exists = scope:first_displayed_artifact }
		pay_short_term_gold = {
			target = scope:book_seller
			gold = scope:first_displayed_artifact.artifact_book_gold_value
		}
		scope:first_displayed_artifact = {
			set_owner = root
		}
	}
	
	option = {
		name = gh_artifact.0002.b
		trigger = { exists = scope:second_displayed_artifact }
		pay_short_term_gold = {
			target = scope:book_seller
			gold = scope:second_displayed_artifact.artifact_book_gold_value
		}
		scope:second_displayed_artifact = {
			set_owner = root
		}
	}
	
	option = {
		name = gh_artifact.0002.c
		trigger = { exists = scope:third_displayed_artifact }
		pay_short_term_gold = {
			target = scope:book_seller
			gold = scope:third_displayed_artifact.artifact_book_gold_value
		}
		scope:third_displayed_artifact = {
			set_owner = root
		}
	}
	
	option = {
		name = gh_artifact.0002.d
	}
}

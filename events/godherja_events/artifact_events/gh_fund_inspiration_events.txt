# Check used to make sure you're not currently on cooldown for inspiration events (so that we don't spam the player)
# Has to be used in the inspiration_owner's scope!
scripted_trigger gh_has_no_inspiration_event_cooldown = {
	NOR = {
		has_character_flag = fund_weapon_inspiration_event_cooldown
		has_character_flag = fund_armor_inspiration_event_cooldown
		has_character_flag = fund_book_inspiration_event_cooldown
		has_character_flag = fund_weaver_inspiration_event_cooldown
		has_character_flag = fund_adventure_inspiration_event_cooldown
		has_character_flag = fund_artisan_inspiration_event_cooldown
		has_character_flag = fund_smith_inspiration_event_cooldown
		has_character_flag = fund_alchemy_inspiration_event_cooldown
		has_character_flag = fund_research_inspiration_event_cooldown
	}
}

# Effect to add appropriate cooldown flag based on the inspiration type
# Has to be used in the inspiration_owner's scope!
scripted_effect add_inspiration_cooldown_flag = {
	if = {
		limit = {
			inspiration = { has_inspiration_type = weapon_inspiration }
		}
		add_character_flag = {
			flag = fund_weapon_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = armor_inspiration }
		}
		add_character_flag = {
			flag = fund_armor_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = book_inspiration }
		}
		add_character_flag = {
			flag = fund_book_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = weaver_inspiration }
		}
		add_character_flag = {
			flag = fund_weaver_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = adventure_inspiration }
		}
		add_character_flag = {
			flag = fund_adventure_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = artisan_inspiration }
		}
		add_character_flag = {
			flag = fund_artisan_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = smith_inspiration }
		}
		add_character_flag = {
			flag = fund_smith_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = alchemy_inspiration }
		}
		add_character_flag = {
			flag = fund_alchemy_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
	else_if = {
		limit = {
			inspiration = { has_inspiration_type = research_inspiration }
		}
		add_character_flag = {
			flag = fund_research_inspiration_event_cooldown
			days = { 30 60 }
		}
	}
}

gh_fund_inspiration.1002 = { # Inspiration: Crystal Blood
	type = character_event
	content_source = dlc_gh
	title = gh_fund_inspiration.1002.t
	desc = gh_fund_inspiration.1002.desc
	theme = witchcraft
	override_background = { reference = corridor_night }
	left_portrait = {
		character = scope:inspiration_owner
		animation = personality_irrational
	}

	trigger = {
		scope:inspiration_owner = {
			var:artifact_smith_type ?= flag:smith_type_oltenic_mask
			NOT = { has_variable = chosen_mask_material }
			gh_has_no_inspiration_event_cooldown = yes
			OR = {
				has_trait = lunatic_1
				has_trait = lunatic_genetic
				has_trait = possessed_1
				has_trait = possessed_genetic
				has_trait = eccentric
			}
		}
	}

	weight_multiplier = {
		base = 1
	}

	immediate = {
		scope:inspiration_owner = {
			add_inspiration_cooldown_flag = yes
			add_trait = ill
		}
	}

	option = { # Refuse
		name = gh_fund_inspiration.1002.a
		custom_tooltip = gh_fund_inspiration.1002.a.tt
		scope:inspiration_owner = {
			add_opinion = {
				target = root
				modifier = refusal_opinion
				opinion = -20
			}
		}
		stress_impact = {
			temperate = minor_stress_impact_loss
			content = minor_stress_impact_loss
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_greed = 0.1
				ai_rationality = 2
			}
		}
	}

	option = { # Agree
		name = gh_fund_inspiration.1002.b
		custom_tooltip = gh_fund_inspiration.1002.b.tt
		scope:inspiration = { invest_gold = 50 }
		scope:inspiration_owner = {
			set_variable = {
				name = chosen_mask_material
				value = flag:bloodcrystal
			}
			change_artifact_quality_effect = { AMOUNT = 4 }
		}
		stress_impact = {
			eccentric = major_stress_impact_loss
			arbitrary = medium_stress_impact_loss
		}
		ai_chance = {
			base = 0
			ai_value_modifier = {
				ai_greed = -0.1
				ai_rationality = -1
			}
			modifier = {
				add = 25
				OR = {
					has_trait = lunatic_1
					has_trait = lunatic_genetic
					has_trait = possessed_1
					has_trait = possessed_genetic
				}
			}
			modifier = {
				factor = 0
				short_term_gold < 50
			}
		}
	}

	option = { # Dismiss them
		name = gh_fund_inspiration.1002.c
		if = {
			limit = { exists = scope:inspiration }
			end_inspiration_sponsorship = scope:inspiration
		}
		scope:inspiration_owner = {
			add_opinion = {
				target = root
				modifier = hate_opinion
				opinion = -30
			}
			banish = yes
		}
		stress_impact = {
			paranoid = medium_stress_impact_loss

		}
		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_greed = 0.3
				ai_rationality = 0.3
			}
		}
	}
}

	immediate = {
		scope:inspiration_owner = {
			set_variable = { # Only one offer of enchantment per weapon
				name = chosen_weapon_enchantment
				value = flag:none
			}
			add_inspiration_cooldown_flag = yes
		}
		if = { # If the spell is known by the Inspiration Owner, no other action is needed
			limit = { NOT = { magic_spell_is_known_by_trigger = { MAGI = scope:inspiration_owner SPELL = global_var:weapons_of_perdition } } }
			if = { # If the spell is known by the Court Magi, they get a prominent role
				limit = { cp:councillor_court_magi ?= { magic_spell_is_known_by_trigger = { MAGI = this SPELL = global_var:weapons_of_perdition } } }
				cp:councillor_court_magi = { save_scope_as = magi }
			}
			else = { # If neither the Inspiration Owner nor the Court Magi know the spell, an outsider is brought in
				random_pool_character = { # Fetch an existing character if possible
					province = prev.location
					limit = { magic_spell_is_known_by_trigger = { MAGI = this SPELL = global_var:weapons_of_perdition } }
					save_scope_as = magi
				}
				if = { # Create a new Magi character if not
					limit = { NOT = { exists = scope:magi } }
					if = { # A local, if possible
						limit = { root.capital_county.culture = { culture_knows_magic_school = { MAGIC_SCHOOL = global_var:wane_tender_harmoniarchy } } }
						root.capital_county = {
							culture = { save_scope_as = magi_culture }
							faith = { save_scope_as = magi_faith }
						}
					}
					else = { # An outsider, if not
						random_culture_global = {
							limit = {
								culture_knows_magic_school = { MAGIC_SCHOOL = global_var:wane_tender_harmoniarchy }
								any_culture_county = { exists = yes } # No dead cultures please
							}
							save_scope_as = magi_culture
							random_culture_county = { faith = { save_scope_as = magi_faith } }
						}
					}
					create_character = {
						location = prev.location
						culture = scope:magi_culture
						faith = scope:magi_faith
						template = perdition_enchanter_template
						save_scope_as = magi
					}
				}
				if = { # Setup for an event option
					limit = { scope:magi ?= { can_be_court_magi_trigger = { COURT_OWNER = root } } }
					cp:councillor_court_magi ?= { save_scope_as = court_magi }
				}
			}
		}
	}

	option = { # Accept the enchantment
		name = gh_fund_inspiration.1003.a
		flavor = gh_fund_inspiration.1003.a.flavor
		scope:inspiration = { invest_gold = 150 }
		add_prestige = medium_prestige_gain
		scope:inspiration_owner = {
			change_artifact_quality_effect = { AMOUNT = 4 }
			custom_tooltip = {
				text = gh_fund_inspiration.1003.enchantment.tt
				set_variable = {
					name = chosen_weapon_enchantment
					value = flag:perdition
				}
			}
		}

		stress_impact = {
			sadistic = medium_stress_impact_loss
			callous = medium_stress_impact_loss
			ambitious = medium_stress_impact_loss
			greedy = minor_stress_impact_gain
			compassionate = minor_stress_impact_gain
		}

		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_greed = -0.25
				ai_compassion = -0.25
			}
			modifier = {
				add = 50
				OR = {
					has_trait = lifestyle_blademaster
					has_trait = sadistic
					has_trait = callous
				}
			}
			modifier = {
				factor = 0
				short_term_gold < 100
			}
		}
	}

	option = { # Accept the enchantment, employ the Magi
		name = gh_fund_inspiration.1003.b
		flavor = gh_fund_inspiration.1003.b.flavor
		trigger = {
			scope:magi ?= {
				NOT = { this = root.cp:councillor_court_magi }
				can_be_court_magi_trigger = { COURT_OWNER = root }
			}
			trigger_if = {
				limit = { exists = scope:court_magi }
				scope:court_magi = { can_be_fired_from_council_trigger = { COURT_OWNER = root } }
			}
		}
		scope:inspiration = { invest_gold = 150 }
		add_prestige = medium_prestige_gain
		pay_short_term_gold = {
			target = scope:magi
			gold = 50
		}
		cp:councillor_court_magi ?= {
			prev = { fire_councillor = prev }
			add_opinion = {
				target = root
				modifier = angry_opinion
				opinion = -20
			}
		}
		add_courtier = scope:magi
		assign_councillor_type = { type = councillor_court_magi target = scope:magi }
		scope:magi = {
			add_opinion = {
				target = root
				modifier = hired_me_opinion
				opinion = 20
			}
		}
		scope:inspiration_owner = {
			change_artifact_quality_effect = { AMOUNT = 4 }
			custom_tooltip = {
				text = gh_fund_inspiration.1003.enchantment.tt
				set_variable = {
					name = chosen_weapon_enchantment
					value = flag:perdition
				}
			}
		}

		stress_impact = {
			sadistic = medium_stress_impact_loss
			callous = medium_stress_impact_loss
			ambitious = medium_stress_impact_loss
			greedy = minor_stress_impact_gain
			compassionate = minor_stress_impact_gain
			paranoid = medium_stress_impact_gain
		}

		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_greed = -0.5
				ai_compassion = -0.25
				ai_sociability = 0.5
			}
			modifier = {
				add = 50
				OR = {
					has_trait = lifestyle_blademaster
					has_trait = sadistic
					has_trait = callous
				}
			}
			modifier = {
				factor = 0
				short_term_gold < 200
			}
		}
	}

	option = { # I can enchant it myself, actually
		name = gh_fund_inspiration.1003.c
		flavor = gh_fund_inspiration.1003.c.flavor
		trigger = { magic_spell_is_known_by_trigger = { MAGI = this SPELL = global_var:weapons_of_perdition } }
		scope:inspiration = { invest_gold = 50 }
		add_prestige = {
			value = medium_prestige_gain
			multiply = 2
		}
		scope:inspiration_owner = {
			change_artifact_quality_effect = { AMOUNT = 4 }
			custom_tooltip = {
				text = gh_fund_inspiration.1003.enchantment.tt
				set_variable = {
					name = chosen_weapon_enchantment
					value = flag:perdition
				}
			}
		}

		stress_impact = {
			sadistic = medium_stress_impact_loss
			callous = medium_stress_impact_loss
			ambitious = medium_stress_impact_loss
			diligent = minor_stress_impact_loss
			greedy = miniscule_stress_impact_gain
			lazy = minor_stress_impact_gain
			compassionate = minor_stress_impact_gain
		}

		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_greed = -0.25
				ai_energy = -0.1
				ai_compassion = -0.25
			}
			modifier = {
				add = 50
				OR = {
					has_trait = lifestyle_blademaster
					has_trait = sadistic
					has_trait = callous
				}
			}
			modifier = {
				factor = 0
				short_term_gold < 50
			}
		}
	}

	option = { # No
		name = gh_fund_inspiration.1003.d
		flavor = gh_fund_inspiration.1003.d.flavor
		scope:inspiration_owner = {
			add_opinion = {
				target = root
				modifier = refusal_opinion
				opinion = -15
			}
		}
		scope:magi ?= {
			if = {
				limit = { this = root.cp:councillor_court_magi }
				add_opinion = {
					target = root
					modifier = refusal_opinion
					opinion = -15
				}
			}
		}

		stress_impact = {
			compassionate = minor_stress_impact_loss
			ambitious = minor_stress_impact_gain
		}

		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_greed = 0.25
				ai_energy = -0.25
			}
		}
	}

	after = {
		scope:magi ?= {
			if = {
				limit = { NOT = { employer = root } }
				silent_disappearance_effect = yes
			}
		}
	}
}

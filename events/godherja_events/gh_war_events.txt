#Event for the execution of Aeschraes if he loses a war
war_event.3104 = {
	type = character_event
	content_source = dlc_gh
	
	title = war_event.3104.t
	desc = war_event.3104.desc

	theme = prison
	left_portrait = {
		character = root
		animation = war_over_win
	}
	right_portrait = {
		character = character:westernimperial_aeschraes_aagiokrata_1
		animation = rage
	}

	trigger = {
		NOT = { has_character_flag = war_event_3104_open }
	}

	immediate = {
		add_character_flag = war_event_3104_open
		character:westernimperial_aeschraes_aagiokrata_1 = {
			save_scope_as = prisoner_a
		}
	}

	 option = {
		name = war_event.3104.a	#Execute Aeschraes

		scope:prisoner_a = {
			death = {
				death_reason = death_execution
				killer = root
			}
		}
		add_trait = gh_aeschraes_killer
		GH_axeinarchon_check = yes
		add_piety = massive_piety_gain
		add_prestige = massive_prestige_gain
		remove_character_flag = war_event_3104_open
		if = {
			limit = { has_any_nickname = no }
			give_nickname = nick_archon_killer
		}
		faith:aeschraes_aagiokrata = {
			change_fervor = {
				value = -100
				desc = fervor_loss_aeschraes_executed
			}
		}
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = war_event.3104.b
		remove_character_flag = war_event_3104_open
		ai_chance = {
			base = 0
		}
	}
}

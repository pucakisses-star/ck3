# Sets up invasion
arch_lich_invasion.0001 = {
	scope = none
	hidden = yes

	trigger = { NOT = { has_game_rule = no_arch_lich } }

	immediate = { # GH_TODO: Change back to chance-based for release
		if = {
			limit = {
					has_game_rule = default_arch_lich 
				}
			trigger_event = {
				id = arch_lich_invasion.0002
				years = { 60 120 }
			}
		}
		else = {
			random_list = {
				33 = {}
				66 = {
					if = {
						limit = {
								has_game_rule = chance_of_arch_lich 
							}
						trigger_event = {
							id = arch_lich_invasion.0002
							years = { 60 120 }
						}
					}
					else_if = {
						limit = { has_game_rule = random_arch_lich } # If it's random, then Arch Lich & Fogeatercan be active at same time
						trigger_event = {
							id = arch_lich_invasion.0002
							years = { 10 200 }
						}
					}
				}
			}
		}
	}
}

arch_lich_invasion.0002 = { # Lich buildup prior to invading Chevalie
	scope = none
	hidden = yes

	immediate = {
		debug_log = "Arch Lich Phase 1"
		debug_log_date = yes
		set_global_variable = {
			name = arch_lich_active
			value = yes
		}
		if = { # Find a suitable Lich to lead the invasion
			limit = {
				any_independent_ruler = {
					is_undead = yes
					has_trait = necromancy_undead_lich
				}
			}
			random_independent_ruler = {
				limit = {
					is_undead = yes
					has_trait = necromancy_undead_lich
				}
				save_scope_as = arch_lich
			}
			scope:arch_lich = {
				if = {
					limit = {
						NOT = { has_trait = education_lich_5 }
					}
					change_trait_rank = {
						trait = education_lich
						rank = 1
					}
				}
				add_character_flag = future_arch_lich
			}
		}
		title:c_province4700 = { # Let's add some awakened undead as vassals so its not liches galore
			set_variable = {
				name = pending_undead_population
				value = { 14 23 }
			}
		}
		while = {
			count = title:c_province4700.var:pending_undead_population
			random_list = {
				1 = {
					trigger_event = {
						id = arch_lich_invasion.0003
						years = { 5 20 }
					}
				}
				1 = {
					trigger_event = {
						id = arch_lich_invasion.0003
						years = { 15 20 }
					}
				}
				1 = {
					trigger_event = {
						id = arch_lich_invasion.0003
						years = { 15 25 }
					}
				}
			}
		}
		trigger_event = { # Invasion begins
			id = arch_lich_invasion.0004
			years = 45 # GH_TODO: Change back to 45 for release
		}
	}
}

scripted_effect populate_undead_effect = { # spawns awakened undead so that its not only Liches in the domain
	random_county_in_region = {
		region = GH_geographic_oejeynica
		limit = {
			holder = { is_ruin = yes }
			duchy = { NOT = { exists = holder } }
		}
		save_scope_as = undead_county
		create_character = {
			location = scope:undead_county.title_province
			template = skeleton_adult_character_template
			save_scope_as = undead_county_holder
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		scope:undead_county = {
			add_to_list = undead_lieutenant_titles
			random_list = {
				4 = { }
				1 = { # Make it multi-county
					trigger = {
						duchy = {
							NOT = {
								exists = holder
								any_de_jure_county_holder = { is_ruin = no }
							}
						}
					}
					while = {
						count = { 1 2 }
						random_neighboring_county = {
							limit = {
								holder = { is_ruin = yes }
								NOT = { is_in_list = undead_lieutenant_titles }
							}
							add_to_list = undead_lieutenant_titles
						}
					}
				}
				1 = { # Make it a Grand Tribe
					trigger = {
						duchy = {
							NOT = {
								exists = holder
								any_de_jure_county_holder = { is_ruin = no }
							}
						}
					}
					duchy = { add_to_list = undead_lieutenant_titles }
					while = {
						count = { 1 2 }
						random_neighboring_county = {
							limit = {
								holder = { is_ruin = yes }
								NOT = { is_in_list = undead_lieutenant_titles }
							}
							add_to_list = undead_lieutenant_titles
						}
					}
				}
			}
		}
		every_in_list = {
			list = undead_lieutenant_titles
			change_title_holder = {
				holder = scope:undead_county_holder
				change = scope:change
			}
		}
		scope:undead_county_holder = { becomes_independent = { change = scope:change } }
		resolve_title_and_vassal_change = scope:change
		scope:undead_county_holder = {
			every_sub_realm_county = { remove_ruin = { COUNTY = this } }
			if = {
				limit = { NOT = { has_government = lichdom_government } }
				change_government = lichdom_government
			}
			#every_courtier = { convert_wasteland_to_undead_effect = yes }
		}
		while = {
			count = { 7 14 }
			create_character = {
				gender_female_chance = 50
				employer = scope:undead_county_holder
				template = skeleton_adult_character_template
			}
		}
	}
}

arch_lich_invasion.0003 = { # Spawn undead in Lichdoms
	scope = none
	hidden = yes

	immediate = {
		title:c_province4700 = {
			if = {
				limit = { var:pending_undead_population > 0 }
				populate_undead_effect = yes
				change_variable = {
					name = pending_undead_population
					add = -1
				}
			}
			else = {
				remove_variable = pending_undead_population
			}
		}
	}
}

arch_lich_invasion.2003 = { # Subjugation refused
	type = letter_event
	opening = arch_lich_invasion.2003.opening
	desc = arch_lich_invasion.2003.desc
	sender = {
		character = scope:demand_recipient
		animation = personality_bold
	}

	immediate = {
		scope:demand_recipient.primary_title = { save_scope_as = target_title }
	}

	option = { # War it is
		name = arch_lich_invasion.2003.a
		if = {
			limit = { exists = scope:target_title.empire }
			start_war = {
				cb = lich_invasion_cb
				target = scope:target_title.holder
				target_title = scope:target_title.empire
			}
		}
		else = {
			start_war = {
				cb = lich_invasion_cb
				target = scope:target_title.holder
				target_title = scope:target_title.holder.capital_province.empire
			}
		}
	}
}

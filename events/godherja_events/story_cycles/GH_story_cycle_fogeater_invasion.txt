fogeater_invasion.0001 = { # Setting up the invastion
	scope = none
	hidden = yes

	trigger = { NOT = { has_game_rule = never_fogeater_invasion } }

	immediate = {
		if = {
			limit = { has_game_rule = default_fogeater_invasion }
			if = { # 1254 bookmark
				limit = { game_start_date < 1260.1.1 }
				trigger_event = {
					id = fogeater_invasion.0002
					years = { 76 96 }
				}
			}
			else_if = { # 1298 bookmark
				limit = { game_start_date < 1300.1.1 }
				trigger_event = {
					id = fogeater_invasion.0002
					years = { 32 52 }
				}
			}
		}
		else_if = {
			limit = { has_game_rule = random_fogeater_invasion }
			if = { # 1254 bookmark
				limit = { game_start_date < 1260.1.1 }
				trigger_event = {
					id = fogeater_invasion.0002
					years = { 5 150 }
				}
			}
			else_if = { # 1298 bookmark
				limit = { game_start_date < 1300.1.1 }
				trigger_event = {
					id = fogeater_invasion.0002
					years = { 5 100 }
				}
			}
		}
	}
}

fogeater_invasion.0002 = { # Fogeater buildup prior to the invasion
	scope = none
	hidden = yes

	immediate = {
		if = { # Decide spawn location
			limit = { has_game_rule = default_fogeater_spawn_location }
			random_list = {
				70 = {
					set_global_variable = {
						name = fogeater_spawn_location_kalath
						value = yes
					}
				}
				30 = {
					set_global_variable = {
						name = fogeater_spawn_location_amaghea
						value = yes
					}
				}
			}
		}
		else_if = {
			limit = { has_game_rule = kalath_fogeater_spawn_location }
			set_global_variable = {
				name = fogeater_spawn_location_kalath
				value = yes
			}
		}
		else_if = {
			limit = { has_game_rule = aironoi_fogeater_spawn_location }
			set_global_variable = {
				name = fogeater_spawn_location_amaghea
				value = yes
			}
		}
		title:c_the_eternal_grove = { # Determine fogeater populations
			if = {
				limit = { has_global_variable = fogeater_spawn_location_kalath }
				set_variable = {
					name = pending_fogeater_population_kalath
					value = { 11 14 }
				}
				set_variable = {
					name = pending_fogeater_population_amaghea
					value = { 6 9 }
				}
			}
			else_if = {
				limit = { has_global_variable = fogeater_spawn_location_amaghea }
				set_variable = {
					name = pending_fogeater_population_kalath
					value = { 6 9 }
				}
				set_variable = {
					name = pending_fogeater_population_amaghea
					value = { 11 14 }
				}
			}
			set_variable = {
				name = pending_fogeater_population_eramasic_river_delta
				value = { 9 12 }
			}
			set_variable = {
				name = pending_fogeater_population_interior
				value = { 5 8 }
			}
			set_variable = {
				name = pending_fogeater_population_total
				value = {
					value = 0
					add = var:pending_fogeater_population_kalath
					add = var:pending_fogeater_population_amaghea
					add = var:pending_fogeater_population_eramasic_river_delta
					add = var:pending_fogeater_population_interior
				}
			}
		}
		while = {
			count = title:c_the_eternal_grove.var:pending_fogeater_population_total
			random_list = {
				1 = {
					trigger_event = {
						id = fogeater_invasion.0003
						years = { 5 45 }
					}
				}
				1 = {
					trigger_event = {
						id = fogeater_invasion.0003
						years = { 15 45 }
					}
				}
				1 = {
					trigger_event = {
						id = fogeater_invasion.0003
						years = { 25 45 }
					}
				}
				1 = {
					trigger_event = {
						id = fogeater_invasion.0003
						years = { 35 45 }
					}
				}
			}
		}
		trigger_event = { # Imminent invasion
			id = fogeater_invasion.0004
			years = 45
		}
	}
}

scripted_effect populate_fogeater_tribes_effect = { # Spawn a new Fogeater tribe of random size
	random_county_in_region = {
		region = $REGION$
		limit = {
			holder = { is_ruin = yes }
			duchy = { NOT = { exists = holder } }
		}
		save_scope_as = fogeater_county
		create_character = {
			location = scope:fogeater_county.title_province
			template = fogeater_chieftain_template
			save_scope_as = fogeater_county_holder
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		scope:fogeater_county = {
			add_to_list = fogeater_tribe_titles
			random_list = {
				4 = { }
				1 = { # Make it a multi-county Tribe
					trigger = {
						duchy = {
							NOT = {
								exists = holder
								any_de_jure_county_holder = { is_ruin = no }
							}
						}
					}
					while = {
						count = { 1 2 }
						random_neighboring_county = {
							limit = {
								holder = { is_ruin = yes }
								NOT = { is_in_list = fogeater_tribe_titles }
							}
							add_to_list = fogeater_tribe_titles
						}
					}
				}
				1 = { # Make it a Grand Tribe
					trigger = {
						duchy = {
							NOT = {
								exists = holder
								any_de_jure_county_holder = { is_ruin = no }
							}
						}
					}
					duchy = { add_to_list = fogeater_tribe_titles }
					while = {
						count = { 1 2 }
						random_neighboring_county = {
							limit = {
								holder = { is_ruin = yes }
								NOT = { is_in_list = fogeater_tribe_titles }
							}
							add_to_list = fogeater_tribe_titles
						}
					}
				}
			}
		}
		every_in_list = {
			list = fogeater_tribe_titles
			change_title_holder = {
				holder = scope:fogeater_county_holder
				change = scope:change
			}
		}
		scope:fogeater_county_holder = { becomes_independent = { change = scope:change } }
		resolve_title_and_vassal_change = scope:change
		scope:fogeater_county_holder = {
			every_sub_realm_county = { remove_ruin = { COUNTY = this } }
			if = {
				limit = { NOT = { has_government = migratory_government } }
				change_government = migratory_government
			}
			every_courtier = { convert_wasteland_to_fogeater_effect = yes }
		}
		while = {
			count = { 7 14 }
			create_character = {
				gender_female_chance = 50
				employer = scope:fogeater_county_holder
				template = new_warrior_character
				faith = scope:fogeater_county_holder.faith
				culture = scope:fogeater_county_holder.culture
			}
		}
	}
}

fogeater_invasion.0003 = { # Spawn a fogeater tribe
	scope = none
	hidden = yes

	immediate = {
		title:c_the_eternal_grove = {
			random_list = {
				1 = {
					trigger = { var:pending_fogeater_population_kalath > 0 }
					populate_fogeater_tribes_effect = { REGION = fogeater_buildup_region_kalathipsomi }
					change_variable = {
						name = pending_fogeater_population_kalath
						add = -1
					}
					modifier = { add = var:pending_fogeater_population_kalath }
				}
				1 = {
					trigger = { var:pending_fogeater_population_amaghea > 0 }
					populate_fogeater_tribes_effect = { REGION = fogeater_buildup_region_amaghea }
					change_variable = {
						name = pending_fogeater_population_amaghea
						add = -1
					}
					modifier = { add = var:pending_fogeater_population_amaghea }
				}
				1 = {
					trigger = { var:pending_fogeater_population_eramasic_river_delta > 0 }
					populate_fogeater_tribes_effect = { REGION = fogeater_buildup_region_eramasic_river_delta }
					change_variable = {
						name = pending_fogeater_population_eramasic_river_delta
						add = -1
					}
					modifier = { add = var:pending_fogeater_population_eramasic_river_delta }
				}
				1 = {
					trigger = { var:pending_fogeater_population_interior > 0 }
					populate_fogeater_tribes_effect = { REGION = fogeater_buildup_region_interior }
					change_variable = {
						name = pending_fogeater_population_interior
						add = -1
					}
					modifier = {
						add = {
							value = var:pending_fogeater_population_interior
							multiply = 3
						}
					}
				}
			}
			change_variable = {
				name = pending_fogeater_population_total
				add = -1
			}
			if = {
				limit = { var:pending_fogeater_population_total = 0 }
				remove_variable = pending_fogeater_population_kalath
				remove_variable = pending_fogeater_population_amaghea
				remove_variable = pending_fogeater_population_eramasic_river_delta
				remove_variable = pending_fogeater_population_interior
				remove_variable = pending_fogeater_population_total
			}
		}
	}
}

dead_magic.0006 = {
	type = character_event	
	content_source = dlc_gh
    title = dead_magic.0006.t
    desc = dead_magic.0006.desc
    theme = learning_scholarship_focus
    left_portrait = root
	
	trigger = {
        NOT = {
            has_character_flag = had_event_dead_magic_0006
        }
    }

    #weight_multiplier = {
	#	base = 1
	#	upweight_for_focus_modifier = { FOCUS =  }
	#}

    immediate = {
        add_character_flag = {
            flag = had_event_dead_magic_0001
            days = 365
        }
    }
	option = {
		name = dead_magic.0006.a
		trigger = {
			any_realm_province = {
				has_building_or_higher = 01_special_aervalr_shard
			}
		}
		remove_short_term_gold = tiny_gold_value
		trigger_event = { id = dead_magic.0007 }
		
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = dead_magic.0006.b
		remove_short_term_gold = medium_gold_value
		trigger_event = { id = dead_magic.0007 }
		
		ai_chance = {
			base = 25
			ai_value_modifier = {
				ai_greed = -0.75
			}
		}
	}
	option = {
		name = dead_magic.0006.c
		ai_chance = {
			base = 0
			ai_value_modifier = {
				ai_greed = 0.75
			}
		}
	}
}

dead_magic.0007 = {
	type = character_event
	content_source = dlc_gh
	title = dead_magic.0007.t
	desc = dead_magic.0007.desc
	theme = learning_scholarship_focus
	left_portrait = root
	
	immediate = {
		if = {
			limit = {
				exists = cp:councillor_court_magi
				cp:councillor_court_magi = {
					stone_aspect_skill_trigger = { VALUE = 5 }
				}
			}
			cp:councillor_court_magi = {
				save_scope_as = court_magi
			}
		}
	}
	
	option = {
		name = dead_magic.0007.a
		custom_tooltip = magic_prowess_duel
		random_list = {
			60 = { # failure
				modifier = {
					add = {
						value = root.var:magic_prowess
						multiply = -3
					}
				}
				modifier = {
					add = {
						value = magi_potency
						multiply = -8
					}
				}
				send_interface_toast = {
					title = dead_magic.0007.a.failure.t
					left_icon = root
				}
				add_prestige = miniscule_prestige_loss
			}
			40 = { # success
				if = {
					limit = { NOT = { exists = var:uncapped_magic_gain_activated } }
					custom_tooltip = capped_magic.tt
				}
				modifier = {
					add = {
						value = root.var:magic_prowess
						multiply = 3
					}
				}
				modifier = {
					add = {
						value = magi_potency
						multiply = 8
					}
				}
				send_interface_toast = {
					title = dead_magic.0007.a.success.t
					left_icon = root
					if = {
						limit = {
							exists = var:uncapped_magic_gain_activated
							NOT = { has_stored_magic_trigger = { VALUE = character_extended_magic_baseline_value } }
						}
						gain_magical_power_capped_effect = { VALUE = medium_magic_value CAP = character_extended_magic_baseline_value }
					}
					else = {
						limit = { NOT = { has_stored_magic_trigger = { VALUE = character_magic_baseline_value } } }
						gain_magical_power_capped_effect = { VALUE = medium_magic_value CAP = character_magic_baseline_value }
					}
					add_arcana_lifestyle_xp = medium_lifestyle_xp
				}
			}
		}
		ai_chance = {
			base = 50
		}
	}
	option = {
		name = dead_magic.0007.b
		trigger = {
			exists = scope:court_magi
		}
		custom_tooltip = magic_prowess_duel
		random_list = {
			60 = { # failure
				modifier = {
					add = {
						value = root.var:magic_prowess
						multiply = -3
					}
				}
				modifier = {
					add = {
						value = magi_potency
						multiply = -8
					}
				}
				modifier = {
					add = {
						value = scope:court_magi.var:magic_prowess
						multiply = -2
					}
				}
				modifier = {
					add = {
						value = scope:court_magi.magi_potency
						multiply = -5
					}
				}
				
				max = 97
				send_interface_toast = {
					title = dead_magic.0007.a.failure.t
					left_icon = root
				}
				add_prestige = miniscule_prestige_loss
			}
			40 = { # success
				if = {
					limit = { NOT = { exists = var:uncapped_magic_gain_activated } }
					custom_tooltip = capped_magic.tt
				}
				modifier = {
					add = {
						value = root.var:magic_prowess
						multiply = 6
					}
				}
				modifier = {
					add = {
						value = magi_potency
						multiply = 8
					}
				}
				modifier = {
					add = {
						value = scope:court_magi.var:magic_prowess
						multiply = 2
					}
				}
				modifier = {
					add = {
						value = scope:court_magi.magi_potency
						multiply = 5
					}
				}
				send_interface_toast = {
					title = dead_magic.0007.a.success.t
					left_icon = root
					if = {
						limit = {
							exists = var:uncapped_magic_gain_activated
							NOT = { has_stored_magic_trigger = { VALUE = character_extended_magic_baseline_value } }
						}
						gain_magical_power_capped_effect = { VALUE = medium_magic_value CAP = character_extended_magic_baseline_value }
					}
					else = {
						limit = { NOT = { has_stored_magic_trigger = { VALUE = character_magic_baseline_value } } }
						gain_magical_power_capped_effect = { VALUE = medium_magic_value CAP = character_magic_baseline_value }
					}
					add_arcana_lifestyle_xp = medium_lifestyle_xp
				}
			}
		}
		ai_chance = {
			base = 300
		}
	}
	option = {
		name = dead_magic.0007.c
		ai_chance = {
			base = 0
		}
	}
}

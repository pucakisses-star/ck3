mixed_magic_mind.0011 = { #Sleepwalking Courtier
    type = character_event
    title = mixed_magic_mind.0011.t
    desc = mixed_magic_mind.0011.desc
    theme = witchcraft
    override_background = {
		reference = bedchamber
	}

    left_portrait = root

    right_portrait = {
        character = scope:dreamer
        animation = paranoia
    }

    trigger = {
        any_courtier = {
            is_available_ai_adult = yes
        }
        NOT = {
            has_character_flag = had_event_mixed_magic_mind_0011
        }
    }

    weight_multiplier = {
        base = 1
    }

    immediate = {
		add_character_flag = {
			flag = had_event_mixed_magic_mind_0011
			years = 5
		}
        random_courtier = {
            limit = {
                is_available_ai_adult = yes
            }
            save_scope_as = dreamer
			# TODO: Readd modifier
            #add_character_modifier = {
            #    modifier = sleepwalking
            #    years = 5
            #}
        }
	}
    option = { #attempt to cure the dreams
        name = mixed_magic_mind.0011.a
        trigger = {
            custom_description = {
                text = not_enough_magic_tt
                has_variable = magic_counter
                var:magic_counter >= minor_magic_value
            }
        }
        show_as_unavailable = {
            var:magic_counter < minor_magic_value
        }
        custom_tooltip = minor_magic_cost_tooltip
        if = {
            limit = {
                exists = var:uncapped_magic_gain_activated
                NOT = { has_stored_magic_trigger = { VALUE = character_extended_magic_baseline_value } }
            }
            gain_magical_power_capped_effect = { VALUE = small_magic_cost CAP = character_extended_magic_baseline_value }
        }
        else = {
            limit = { NOT = { has_stored_magic_trigger = { VALUE = character_magic_baseline_value } } }
            gain_magical_power_capped_effect = { VALUE = small_magic_cost CAP = character_magic_baseline_value }
        }
        custom_tooltip = magic_prowess_duel
		random_list = {
			40 = { #success
				modifier = {
					add = {
						value = var:magic_prowess
						multiply = 4
					}
				}
				send_interface_toast = {
                    title = mixed_magic_mind.0011.success.t
                    left_icon = scope:dreamer
                    if = {
                        limit = {
                            can_add_hook = {
                                target = scope:dreamer
                                type = favor_hook
                            }
                        }
                        add_hook = {
                            type = favor_hook
                            target = scope:dreamer
                        }
                    }
                    reverse_add_opinion = {
                        target = scope:dreamer
                        modifier = grateful_opinion
                        opinion = 30
                    }
                    add_arcana_lifestyle_xp = medium_lifestyle_xp
					# TODO: Readd modifier
                    #scope:dreamer = {
                    #    remove_character_modifier = sleepwalking
                    #}
                }
			}
			60 = { #failure
				modifier = {
					add = {
						value = var:magic_prowess
						multiply = -4
					}
				}
				send_interface_toast = {
                    title = mixed_magic_mind.0011.failure.t
                    left_icon = scope:dreamer
                    reverse_add_opinion = {
                        target = scope:dreamer
                        modifier = disappointed_opinion
                        opinion = -15
                    }
                    add_arcana_lifestyle_xp = miniscule_lifestyle_xp
                }
			}
		}
        stress_impact = {
            compassionate = minor_stress_impact_loss
        }
        ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_compassion = 2
			}
		}
    }
    option = { #study the dreams
        name = mixed_magic_mind.0011.b
        add_arcana_lifestyle_xp = major_lifestyle_xp
        reverse_add_opinion = {
            target = scope:dreamer
            modifier = cruelty_opinion
            opinion = -25
        }
        stress_impact = {
            compassionate = medium_stress_impact_gain
            just = minor_stress_impact_gain
            trusting = minor_stress_impact_gain
        }
        ai_chance = {
			base = 50
		}
    }
    option = { #Draw magic from the dreams
        name = mixed_magic_mind.0011.c
        add_arcana_lifestyle_xp = medium_lifestyle_xp
        scope:dreamer = {
			# TODO: Readd modifier
            #remove_character_modifier = sleepwalking
            #add_character_modifier = {
            #    modifier = severe_sleepwalking
            #    years = 5
            #}
        }
        reverse_add_opinion = {
            target = scope:dreamer
            modifier = cruelty_opinion
            opinion = -50
        }
        if = {
            limit = {
                exists = var:uncapped_magic_gain_activated
                NOT = { has_stored_magic_trigger = { VALUE = character_extended_magic_baseline_value } }
            }
            gain_magical_power_capped_effect = { VALUE = small_magic_value CAP = character_extended_magic_baseline_value }
        }
        else = {
            limit = { NOT = { has_stored_magic_trigger = { VALUE = character_magic_baseline_value } } }
            gain_magical_power_capped_effect = { VALUE = small_magic_value CAP = character_magic_baseline_value }
        }
        custom_tooltip = small_magic_gain_tooltip
		# TODO: Readd modifier
        #add_character_modifier = {
        #    modifier = experimented_on_courtiers
        #    years = 5
        #}
        stress_impact = {
            compassionate = major_stress_impact_gain
            just = medium_stress_impact_gain
            trusting = medium_stress_impact_gain
            sadistic = minor_stress_impact_loss
        }
        ai_chance = {
			base = 50
            ai_value_modifier = {
				ai_compassion = -2
			}
		}
    }
}

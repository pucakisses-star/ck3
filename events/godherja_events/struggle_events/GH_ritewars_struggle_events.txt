		 	}
	 	}

					target = root
					modifier = strengthened_friendship_opinion
			 	}
			}	
		 	else = {
			 	progress_towards_friend_effect = {
					REASON = friend_returned_refugee
					CHARACTER = root
					OPINION = 0
			 	}

			 	limit = {

					}

			}

				scope:great_physician = {
					silent_disappearance_effect = yes
				}
			}
		}

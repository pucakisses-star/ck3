kashirya_decision.0001 = {
	title = taryaad_reformed.0001.t
	desc = taryaad_reformed.0001.desc
	type = character_event
	content_source = dlc_gh
	theme = realm
	left_portrait = {
		character = scope:taryaad_reformer
		animation = personality_bold
	}

	immediate = {
		play_music_cue = "mx_cue_epic_sacral_moment"	
		form_taryaad_saramsaasan_effect = yes

		if = {
			limit = { NOT = { government_allows = administrative } }
			convert_to_administrative_from_feudalism_effect = yes
		}
	}

	option = {
		name = taryaad_reformed.0001.a
		remove_nickname = yes
		give_nickname = nick_true_heir_of_asojiva # 'The True Heir Of Asojiva'
		if = {
			limit = {
				prestige_level < 5
			}
			add_prestige_level = 1
		}
		
		add_trait = gh_heir_of_asojiva 
		
		dynasty = {
			add_dynasty_prestige = 250
		}
	}
}

scripted_effect shift_duchy_to_migration_title_effect = {
	if = {
		limit = { NOT = { de_jure_liege = $NEW_DE_JURE_LIEGE$ } }
		de_jure_liege = { save_temporary_scope_as = former_de_jure_liege }
		set_de_jure_liege_title = $NEW_DE_JURE_LIEGE$
		if = {
			limit = {
				exists = scope:former_de_jure_liege
				scope:former_de_jure_liege.direct_de_jure_vassals_value < 3 
			}
			scope:former_de_jure_liege = {
				every_in_de_jure_hierarchy = {
					limit = { tier = prev.direct_de_jure_vassal_tier_value }
					set_de_jure_liege_title = $NEW_DE_JURE_LIEGE$
				}
			}
		}
	}
}

sjalvolki_decisions.3006 = {
	hidden = yes

	immediate = {

		if = {
			limit = { has_title = title:e_migrant_coalition }
			if = {
				limit = { title:d_cosmopol = { NOT = { holder = root } } }
				get_title = title:d_cosmopol
			}
			title:d_cosmopol = {
				remove_from_list = migration_duchies
			}
			if = {
				limit = { title:c_oraispol = { NOT = { holder = root } } }
				get_title = title:c_oraispol
			}
			if = {
				limit = { title:c_magnonion = { NOT = { holder = root } } }
				get_title = title:c_magnonion
			}
			if = {
				limit = { title:c_valthronaix = { NOT = { holder = root } } }
				get_title = title:c_valthronaix 
			}
			title:c_oraispol = {
				set_county_culture = root.culture
				set_county_faith = root.faith
				remove_from_list = migration_counties
				add_county_modifier = { modifier = gh_sjalvolki_resettlement_modifier years = 30 }
			}
			title:c_magnonion = {
				set_county_culture = root.culture
				set_county_faith = root.faith
				remove_from_list = migration_counties
				add_county_modifier = { modifier = gh_sjalvolki_resettlement_modifier years = 30 }
			}
			title:c_valthronaix = {
				add_county_modifier = { modifier = gh_sjalvolki_resettlement_modifier years = 30 }
			}
			set_realm_capital = title:c_oraispol
			
			create_title_and_vassal_change = {
				type = independency
				save_scope_as = change
				add_claim_on_loss = no
			}
			
			every_held_title = {
				limit = {
					OR = {
						tier = tier_county
						tier = tier_duchy
						tier = tier_kingdom
					}
					NOR = {
						kingdom = title:k_palitake
						empire = title:e_etepezea
					}
				}
			
				if = {
					limit = {
						tier >= tier_duchy
					}
					prev = {
						destroy_title = prev
					}
				}
				else_if = {
					limit = { NOT = { has_variable = protected_from_migration_destruction } }
					create_character = {
						location = root.top_liege.capital_province
						gender_female_chance = 50
						template_character = root
						random_traits = yes
						age = { 20 50 }
						
						save_scope_as = fill
					}
					
					change_title_holder = {
						holder = scope:fill
						change = scope:change
					}
					
					if = {
						limit = {
							tier = tier_county
						}
						
						scope:fill = {
							change_liege = {
								CHANGE = scope:change
								LIEGE = scope:destroyer
							}
						}
					}
				}
			}
			resolve_title_and_vassal_change = scope:change

			if = {
				limit = { has_title = title:k_daukeni }
				destroy_title = title:k_daukeni
			}
		}
	}
}

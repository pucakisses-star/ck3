gh_consult_dream_interpreter_decision.0001 = {
	type = character_event	
	theme = faith
	title = {
		random_valid = {
			desc = gh_consult_dream_interpreter_decision.0001.t.first
			desc = gh_consult_dream_interpreter_decision.0001.t.second
		}
	}
	desc = {
		random_valid = {
			desc = gh_consult_dream_interpreter_decision.0001.desc.first
			desc = gh_consult_dream_interpreter_decision.0001.desc.second
		}
	}
	
	left_portrait = {
		character = root
		animation = thinking
	}

	option = {
		name = gh_consult_dream_interpreter_decision.0001.a
		trigger = { 
			any_held_title = {
				tier = tier_county
				any_county_province = {
					has_building_or_higher = gh_house_of_dreams_01
				}
			}
		}
		add_character_flag = {
			flag = visit_house_of_dream
			days = 5
		}
	}
	option = {
		name = gh_consult_dream_interpreter_decision.0001.b
		trigger = { 
			employs_court_position = dream_interpreter_court_position
			any_court_position_holder = {
				type = dream_interpreter_court_position
				is_physically_able_adult = yes
				has_court_event_flag = no
				is_ai = yes
			}
		}
		add_character_flag = {
			flag = visit_dream_interpreter
			days = 5
		}
	}
	option = {
		name = gh_consult_dream_interpreter_decision.0001.c
	}

	after = {
		if = { 
			limit = { has_character_flag = visit_dream_interpreter }
			random_court_position_holder = {
				type = dream_interpreter_court_position
				save_scope_as = dream_interpreter
			}
			trigger_event = { on_action = consult_dream_interpreter_events }
		}
		else_if = {
			limit = { has_character_flag = visit_house_of_dream }
			create_character = {
				location = root.capital_province
				template = priest_character_template
				save_scope_as = dream_interpreter
			}
			trigger_event = { on_action = consult_dream_interpreter_events }
		}
	}
}

gh_consult_dream_interpreter_decision.0005 = {
	type = character_event	
	theme = faith
	title = gh_consult_dream_interpreter_decision.0005.t
	desc = gh_consult_dream_interpreter_decision.0005.desc
	
	trigger = {
		any_targeting_scheme = {
			is_scheme_exposed = no
			scheme_defender = root
			scheme_type = murder
		}
	}
	
	left_portrait = {
		character = root
		animation = worry
	}
	right_portrait = {
		character = scope:dream_interpreter
		animation = thinking
	}

	immediate = {
		random_targeting_scheme = {
			limit = {
				is_scheme_exposed = no
				scheme_defender = root
				scheme_type = murder
			}
			expose_scheme = yes
		}
	}
	option = {
		name = gh_consult_dream_interpreter_decision.0005.a
		trigger = {
			exists = cp:councillor_spymaster
			cp:councillor_spymaster ?= {
				NOT = { is_performing_council_task = task_disrupt_schemes }
			}
		}
		
		cp:councillor_spymaster = {
			set_council_task = {
				task_type = task_disrupt_schemes
				target = root
			}
		}
		ai_chance = {
			base = 100
			modifier = {
				OR = {
					has_trait = sadistic
					has_trait = callous
					has_trait = torturer
					has_trait = wrathful
					has_trait = vengeful
				}
				add = 50
			}
			modifier = {
				has_trait = cynical
				add = 50
			}
		}
	}
	option = {
		name = gh_consult_dream_interpreter_decision.0005.b
		ai_chance = {
			base = 50
		}
	}
}

			scheme_type = murder
		}
	}
	
	left_portrait = {
		character = root
		animation = fear
	}
	right_portrait = {
		character = scope:dream_interpreter
		animation = thinking
	}

	immediate = {
		stress_impact = {
			paranoid = minor_stress_impact_gain
			ambitious = miniscule_stress_impact_gain
			diligent = miniscule_stress_impact_gain
		}
		random_targeting_scheme = {
			limit = {
				is_scheme_exposed = no
				scheme_defender = root
				scheme_type = murder
			}
			scheme_owner = { save_scope_as = murder_schemer }
			expose_scheme = yes
			add_scheme_progress = scheme_progress_loss
		}	
		attempted_murder_opinion_effect = {
			VICTIM = root
			MURDERER = scope:murder_schemer
		}
	}
	option = {
		name = gh_consult_dream_interpreter_decision.0006.a
		stress_impact = {
			paranoid = minor_stress_impact_gain
			ambitious = miniscule_stress_impact_loss
		}
		start_scheme = {
			type = murder
			target_character = scope:murder_schemer
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_vengefulness = 1
			}
			modifier = {
				OR = {
					has_trait = sadistic
					has_trait = callous
					has_trait = torturer
					has_trait = wrathful
					has_trait = vengeful
				}
				add = 50
			}
			modifier = {
				factor = 0
				any_scheme = {
					scheme_type = murder
					scheme_owner = root
				}
			}
		}
	}
	option = {
		name = gh_consult_dream_interpreter_decision.0006.b
		stress_impact = {
			paranoid = minor_stress_impact_gain
		}
		ai_chance = {
			base = 50
		}
	}
}

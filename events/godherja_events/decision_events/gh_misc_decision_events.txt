#FOUDNING METRO EVENT - priming
gh_misc_decision_events.002 = {
	type = character_event
	content_source = dlc_gh
	title = gh_misc_decision_events.002.t
	desc = gh_misc_decision_events.002.desc
	theme = learning
	left_portrait = {
		character = ROOT 
		animation = personality_rational
	}
	
	immediate = { #saving scope for localization
		capital_province = {
			save_scope_as = capital
		}
	}

	option = { # Administrative District
		name = gh_misc_decision_events.002.a
		trigger = {
			OR = {
				diplomacy >= 20
				martial >= 20
			}
		}
		trigger_event = gh_misc_decision_events.003

		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_honor = 1
				ai_greed = -0.2
			}
		}
	}

	option = { # Economic District
		name = gh_misc_decision_events.002.b
		trigger = {
			OR = {
				stewardship >= 20
				intrigue >= 20
			}
		}
		trigger_event = gh_misc_decision_events.004

		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_rationality = 1
			}
		}
	}

	option = { # Scholarly District
		name = gh_misc_decision_events.002.c
		trigger = {
			OR = {
				learning >= 20
				root.character_arcana_baseline_value >= 20
			}
		}
		trigger_event = gh_misc_decision_events.005

		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_zeal = 1
				ai_greed = -0.2
			}
		}
	}
}

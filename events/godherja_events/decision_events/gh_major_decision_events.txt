namespace = gh_major_decisions

# GH Bloodlines

gh_major_decisions.0001 = {
	type = character_event
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				exists = mother
				mother = {
					faith = root.faith
					has_trait = gh_blood_of_prophet
				}
			}
			AND = {
				exists = father
				father = {
					faith = root.faith
					has_trait = gh_blood_of_prophet
				}
			}
			AND = {
				exists = scope:bloodline_founder
				OR = {
					is_grandchild_of = scope:bloodline_founder 
					is_great_grandchild_of = scope:bloodline_founder
				}
			}
		}
		NOT = {
			has_trait = gh_blood_of_prophet
		}
	}
	
	immediate = {
		if = {
			limit = {
				OR = {
					AND = {
						exists = mother
						mother = {							
							has_trait = gh_blood_of_prophet
						}
					}
					AND = {
						exists = father
						father = {							
							has_trait = gh_blood_of_prophet
						}
					}
				}
			}
			add_trait = gh_blood_of_prophet
		}
		every_child = {
			trigger_event = gh_major_decisions.0001
		}
	}
}

world_story.0001 = { #The Fall of Oraispol
	title = world_story.0001.t
	type = character_event
	content_source = dlc_gh
	theme = battle
	
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					culture = { has_cultural_pillar = heritage_aversarian }
				}
				desc = world_story.0001.desc # Aversarian
			}
			triggered_desc = {
				trigger = {
					culture = { has_cultural_pillar = heritage_sjalvolki }
				}
				desc = world_story.0001.desc.sjalvolki # Sjavolki
			}
			triggered_desc = {
				trigger = {
					culture = { has_cultural_pillar = heritage_iyrossi }
				}
				desc = world_story.0001.desc.iyrossi # Iyrossi
			}
			triggered_desc = {
				trigger = {
					NOT = {
						OR = {
							culture = { has_cultural_pillar = heritage_aversarian }
							culture = { has_cultural_pillar = heritage_iyrossi }
							culture = { has_cultural_pillar = heritage_sjalvolki }
						}
					}
				}
				desc = world_story.0001.desc.generic # Generic
			}
		}
	}
	
	right_portrait = { 
		character = scope:oraispol_conqueror
		animation = marshal
	}

	option = {
		trigger = {
			culture = { has_cultural_pillar = heritage_aversarian }
		}
		name = world_story.0001.a
	}
	option = {
		trigger = {
			OR = {
				culture = { has_cultural_pillar = heritage_iyrossi }
				culture = { has_cultural_pillar = heritage_sjalvolki }
			}
		}
		name = world_story.0001.b
	}
	option = {
		trigger = {
			NOT = {
				OR = {
					culture = { has_cultural_pillar = heritage_aversarian }
					culture = { has_cultural_pillar = heritage_iyrossi }
					culture = { has_cultural_pillar = heritage_sjalvolki }
				}
			}
		}
		name = world_story.0001.c
	}
}

world_story.0100 = { # Niconaneus takes Oraispol
	title = world_story.0100.t
	#desc = world_story.0100.desc
	type = character_event
	content_source = dlc_gh
	theme = battle
	
	desc = world_story.0100.desc
	
	right_portrait = { 
		character = root
		animation = marshal
	}

	immediate = {
		title:c_oraispol = { save_scope_as = new_capital }
		title:e_aversaria_voreias = { save_scope_as = aversarian_polity }
	}

	option = { # There is still much to be done
		name = world_story.0100.a
		add_prestige = 1500
		add_prestige_level = 1
		if = { # Relocate the capital
			limit = { primary_title = title:e_aversaria_voreias }
			title:e_aversaria_voreias = { set_capital_county = title:c_oraispol }
			custom_tooltip = world_story.0100.tt
		}
	}
}

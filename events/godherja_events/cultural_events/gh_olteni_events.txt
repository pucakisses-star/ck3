gh_olteni_events.0001 = { # Character loots an Olteni mask in battle
	type = character_event
	content_source = dlc_gh
	title = gh_olteni_events.0001.t
	desc = gh_olteni_events.0001.desc
	theme = battle

	left_portrait = {
		character = root
		scripted_animation = duel_celebrate
	}
	artifact = {
		target = scope:looted_mask
		position = lower_center_portrait
	}
	lower_right_portrait = scope:dead_combatant

	weight_multiplier = {
		base = 1
	}

	immediate = {
		show_as_tooltip = {
			scope:dead_combatant = {
				death = {
					death_reason = death_battle
					killer = root
				}
			}
		}
		scope:looted_mask = {
			if = {
				limit = { exists = artifact_owner.house }
				artifact_owner.house = { save_scope_as = looted_mask_house }
			}
			set_owner = {
				target = root
				history = {
					type = taken_in_battle
					actor = scope:dead_combatant
					recipient = root
					location = root.location
				}
			}
			if = {
				limit = {
					exists = scope:looted_mask_house
					NOT = { any_artifact_house_claimant = { this = scope:looted_mask_house } }
				}
				scope:looted_mask_house = { add_house_artifact_claim = prev }
			}
		}
		hidden_effect = {
			scope:dead_combatant = {
				death = {
					death_reason = death_battle
					killer = root
				}
			}
		}
	}

	option = {
		name = gh_olteni_events.0001.a
		add_prestige = medium_prestige_gain
		stress_impact = {
			ambitious = medium_stress_impact_loss
			brave = minor_stress_impact_loss
			arrogant = minor_stress_impact_loss
		}
	}
}

##################################################
#
# A Wandering Elder Event Chain
#
# A wandering elder is spotted in the realm, prompting
# the player to either seek them out or do nothing.
# Should the player decide to seek the elder out,
# with the following outcomes:
# A mystic, granting relevant trait xp
# A scholar, granting learning
# A useless vagrant
# A Magi to teach the Defiled Nadir spell + outcomes
# The elder disappears without trace
# The elder refuses to meet a sinful character
#
# gh_olteni_events.0101 - gh_olteni_events.0110
#
# by Deadr0x

gh_olteni_events.0101 = { # A Wanderer
	type = character_event
	content_source = dlc_gh
	title = gh_olteni_events.0101.t
	desc = gh_olteni_events.0101.desc
	theme = realm

	right_portrait = {
		character = root
		animation = personality_rational
	}

	trigger = {
		NOT = { has_character_flag = had_event_gh_olteni_events_0101 }
		oltenic_mask_user_cultural_trigger = yes
		is_available_at_peace_adult = yes
	}

	weight_multiplier = {
		base = 1
	}

	immediate = {
		add_character_flag = {
			flag = had_event_gh_olteni_events_0101
			years = 20
		}
	}

	option = { # Extend an offer of hospitality
		name = gh_olteni_events.0101.a
		custom_tooltip = gh_olteni_events.0101.a.tt
		remove_short_term_gold = tiny_gold_value
		add_piety = minor_piety_gain
		hidden_effect = {
			random_list = {
				10 = { # Mystic
					trigger_event = {
						id = gh_olteni_events.0102
						days = { 3 14 }
					}
				}
				10 = { # Scholar
					trigger_event = {
						id = gh_olteni_events.0103
						days = { 3 14 }
					}
				}
				10 = { # Vagrant
					trigger_event = {
						id = gh_olteni_events.0104
						days = { 3 14 }
					}
				}
				5 = { # Magi
					trigger = {
						can_access_magic = yes
						NOT = { magic_spell_is_known_by_trigger = { MAGI = root SPELL = global_var:defiled_nadir } }
					}
					modifier = {
						add = {
							add = 3
							multiply = num_virtuous_traits
						}
					}
					trigger_event = {
						id = gh_olteni_events.0105
						days = { 3 14 }
					}
				}
				10 = { # Disappeared
					trigger_event = {
						id = gh_olteni_events.0106
						days = { 14 28 }
					}
				}
				0 = { # Refused to come
					modifier = {
						add = {
							add = 20
							multiply = num_sinful_traits
						}
					}
					modifier = {
						add = 50
						piety_level < 1
					}
					modifier = {
						factor = 0
						piety_level > 2
					}
					trigger_event = {
						id = gh_olteni_events.0107
						days = { 3 14 }
					}
				}
			}
		}
		stress_impact = {
			zealous = minor_stress_impact_loss
			compassionate = minor_stress_impact_loss
			generous = minor_stress_impact_loss
			trusting = miniscule_stress_impact_loss
			cynical = minor_stress_impact_gain
			paranoid = minor_stress_impact_gain
			greedy = miniscule_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_greed = -0.2
				ai_zeal = 0.5
				ai_compassion = 0.2
			}
			modifier = {
				add = {
					add = 20
					multiply = num_virtuous_traits
				}
				add = {
					add = -20
					multiply = num_sinful_traits
				}
			}
		}
	}

	option = { # Ignore them
		name = gh_olteni_events.0101.b
		add_piety = miniscule_piety_loss
		stress_impact = {
			paranoid = medium_stress_impact_loss
			shy = minor_stress_impact_loss
			cynical = minor_stress_impact_loss
			lazy = miniscule_stress_impact_loss
			zealous = medium_stress_impact_gain
			compassionate = minor_stress_impact_gain
			generous = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_energy = -0.3
				ai_zeal = -0.5
				ai_compassion = -0.1
			}
			modifier = {
				add = {
					add = -20
					multiply = num_virtuous_traits
				}
				add = {
					add = 20
					multiply = num_sinful_traits
				}
			}
		}
	}
}

	theme = realm
	override_background = { reference = study }

		animation = personality_zealous
	}

	immediate = {
		create_character = {
			template = olteni_elder_mystic_template
			location = root.capital_province
			save_scope_as = elder
		}
	}

	option = { # We should learn from each other!
		name = gh_olteni_events.0102.a
		trigger = {
			has_trait_xp = {
				trait = lifestyle_mystic
				value = 100
			}
		}
		trait = lifestyle_mystic
		add_learning_skill = 3
		add_piety = major_piety_gain
		ai_chance = {
			base = 200
			ai_value_modifier = {
				ai_energy = 0.5
				ai_sociability = 0.5
			}
		}
	}

	option = { # Learn mystical knowledge
		name = gh_olteni_events.0102.b
		trigger = {
			NOT = {
				has_trait_xp = {
					trait = lifestyle_mystic
					value = 100
				}
			}
		}
		custom_tooltip = gh_olteni_events.0102.b.tt
		if = {
			limit = { NOT = { has_trait = lifestyle_mystic } }
			add_trait = lifestyle_mystic
			add_trait_xp = {
				trait = lifestyle_mystic
				value = { 10 30 }
			}
		}
		else = { 
			add_trait_xp = {
				trait = lifestyle_mystic
				value = { 20 40 }
			}
		}
		stress_impact = {
			zealous = medium_stress_impact_loss

		}
		ai_chance = {
			base = 100

				ai_energy = 0.2
				ai_zeal = 0.5
			}
			modifier = {
				has_trait = lifestyle_mystic
				add = 100
			}
		}
	}

	option = { # Just let them stay
		name = gh_olteni_events.0103.c
		add_piety = minor_piety_gain
		stress_impact = {
			paranoid = medium_stress_impact_loss
			zealous = minor_stress_impact_gain
			ambitious = minor_stress_impact_gain
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_energy = -0.3
				ai_zeal = -0.5
				ai_sociability = -0.2
			}
			modifier = {
				add = {
					add = -20
					multiply = num_virtuous_traits
				}
				add = {
					add = 20
					multiply = num_sinful_traits
				}
			}
		}

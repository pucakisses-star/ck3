scripted_trigger weak_redlander_heir_trigger = {
	OR = {
		piety_level < 0
		NAND = {
			has_trait = gh_blackspawn_4
			has_trait = gh_blackspawn_5
		}
	}
}

scripted_trigger medium_redlander_heir_trigger = {
	OR = {
		piety_level > 3
		has_trait = gh_blackspawn_1
		has_trait = gh_blackspawn_2
		has_trait = gh_blackspawn_3
		has_trait = gh_blackspawn_4
		has_trait = gh_blackspawn_5
	}
}

scripted_trigger strong_redlander_heir_trigger = {
	OR = {
		piety_level = 5
		has_trait = gh_blackspawn_4
		has_trait = gh_blackspawn_5
	}
}

scripted_effect gh_redlander_events_0001_option_effect = { # Determine heir quality
	random_list = {
		10 = { # Weak heir
			trigger = { weak_redlander_heir_trigger = yes }
			modifier = {
				add = 5
				piety_level < 0
			}
			modifier = {
				add = 5
				piety_level < 1
			}
			modifier = {
				add = -5
				has_trait = gh_blackspawn_3
			}
			gh_redlander_events_0001_weak_pool_effect = yes
		}
		10 = { # Medium heir
			trigger = { medium_redlander_heir_trigger = yes }
			gh_redlander_events_0001_medium_pool_effect = yes
			modifier = {
				add = -5
				has_trait = gh_blackspawn_5
			}
		}
		10 = { # Strong heir
			trigger = { strong_redlander_heir_trigger = yes }
			gh_redlander_events_0001_strong_pool_effect = yes
			modifier = {
				add = -5
				piety_level = 5
			}
		}
	}
}

scripted_effect gh_redlander_events_0001_rubberband_effect = { # Limits the maximum level of the blackspawn trait
	if = {
		limit = { root = { has_trait = gh_blackspawn_4 } }
		if = {
			limit = { has_trait = gh_blackspawn_5 }
			remove_trait = gh_blackspawn_5
			add_trait = gh_blackspawn_4
		}
	}
	else_if = {
		limit = { root = { has_trait = gh_blackspawn_3 } }
		if = {
			limit = { has_trait = gh_blackspawn_5 }
			remove_trait = gh_blackspawn_5
			add_trait = gh_blackspawn_3
		}
		else_if = {
			limit = { has_trait = gh_blackspawn_4 }
			remove_trait = gh_blackspawn_4
			add_trait = gh_blackspawn_3
		}
	}
	else_if = {
		limit = { root = { has_trait = gh_blackspawn_2 } }
		if = {
			limit = { has_trait = gh_blackspawn_5 }
			remove_trait = gh_blackspawn_5
			add_trait = gh_blackspawn_2
		}
		else_if = {
			limit = { has_trait = gh_blackspawn_4 }
			remove_trait = gh_blackspawn_4
			add_trait = gh_blackspawn_2
		}
		else_if = {
			limit = { has_trait = gh_blackspawn_3 }
			remove_trait = gh_blackspawn_3
			add_trait = gh_blackspawn_2
		}
	}
	else_if = {
		limit = {
			root = {
				OR = {
					has_trait = gh_blackspawn_1
					NAND = {
						has_trait = gh_blackspawn_1
						has_trait = gh_blackspawn_2
						has_trait = gh_blackspawn_3
						has_trait = gh_blackspawn_4
						has_trait = gh_blackspawn_5
					}
				}
			}
		}
		if = {
			limit = { has_trait = gh_blackspawn_5 }
			remove_trait = gh_blackspawn_5
			add_trait = gh_blackspawn_1
		}
		else_if = {
			limit = { has_trait = gh_blackspawn_4 }
			remove_trait = gh_blackspawn_4
			add_trait = gh_blackspawn_1
		}
		else_if = {
			limit = { has_trait = gh_blackspawn_3 }
			remove_trait = gh_blackspawn_3
			add_trait = gh_blackspawn_1
		}
		else_if = {
			limit = { has_trait = gh_blackspawn_2 }
			remove_trait = gh_blackspawn_2
			add_trait = gh_blackspawn_1
		}
	}
}

gh_redlander_events.0001 = { # Recruit from the pit
	type = character_event
	content_source = dlc_gh
	title = gh_redlander_events.0001.t
	desc = {
		desc = gh_redlander_events.0001.desc.intro
		first_valid = {
			triggered_desc = {
				trigger = { primary_heir ?= { house = root.house } }
				desc = gh_redlander_events.0001.desc.heir
			}
			desc = gh_redlander_events.0001.desc.no_heir
		}
	}
	theme = witchcraft
	override_background = { reference = terrain_scope }

	left_portrait = {
		character = scope:first
		triggered_animation = {
			trigger = { scope:first = { has_education_diplomacy_trigger = yes } }
			animation = personality_irrational
		}
		triggered_animation = {
			trigger = { scope:first = { has_education_martial_trigger = yes } }
			animation = marshal
		}
		triggered_animation = {
			trigger = { scope:first = { has_education_stewardship_trigger = yes } }
			animation = personality_content
		}
		triggered_animation = {
			trigger = { scope:first = { has_education_intrigue_trigger = yes } }
			animation = spymaster
		}
		triggered_animation = {
			trigger = { scope:first = { has_education_learning_trigger = yes } }
			animation = personality_zealous
		}
		triggered_animation = {
			trigger = { scope:first = { has_education_arcana_trigger = yes } }
			animation = personality_rational
		}
	}
	right_portrait = {
		character = scope:second
		triggered_animation = {
			trigger = { scope:second = { has_education_diplomacy_trigger = yes } }
			animation = personality_compassionate
		}
		triggered_animation = {
			trigger = { scope:second = { has_education_martial_trigger = yes } }
			animation = personality_bold
		}
		triggered_animation = {
			trigger = { scope:second = { has_education_stewardship_trigger = yes } }
			animation = personality_greedy
		}
		triggered_animation = {
			trigger = { scope:second = { has_education_intrigue_trigger = yes } }
			animation = scheme
		}
		triggered_animation = {
			trigger = { scope:second = { has_education_learning_trigger = yes } }
			animation = personality_vengeful
		}
		triggered_animation = {
			trigger = { scope:second = { has_education_arcana_trigger = yes } }
			animation = personality_callous
		}
	}
	lower_center_portrait = scope:old_heir

	immediate = {
		title:c_ofnammm.title_province = { save_scope_as = background_terrain_scope }
		gh_redlander_events_0001_option_effect = yes
		scope:redlander_option = { save_scope_as = first }
		gh_redlander_events_0001_option_effect = yes
		scope:redlander_option = { save_scope_as = second }
		scope:first ?= { gh_redlander_events_0001_rubberband_effect = yes }
		scope:second ?= { gh_redlander_events_0001_rubberband_effect = yes }
		if = {
			limit = {
				exists = primary_heir
				primary_heir.house = root.house
			}
			primary_heir = { save_scope_as = old_heir }
		}
	}

	option = { # Pick first option
		name = gh_redlander_events.0001.a
		add_courtier = scope:first
		set_designated_heir = scope:first
		scope:first = { set_house = root.house }
		scope:old_heir ?= {
			add_opinion = {
				target = root
				opinion = -40
				modifier = insulted_opinion
			}
			add_opinion = {
				target = scope:first
				opinion = -60
				modifier = hate_opinion
			}
			if = {
				limit = { NOT = { has_relation_rival = scope:first } }
				set_relation_rival = {
					target = scope:first
					reason = rival_over_position
				}
			}
			custom_tooltip = {
				text = gh_redlander_events.0001.claim_tt
				root = {
					every_held_title = {
						limit = { tier > tier_barony }
						scope:old_heir = { add_pressed_claim = prev }
					}
				}
			}
		}
		hidden_effect = { scope:second = { silent_disappearance_effect = yes } }
	}

	option = { # Pick second option
		name = gh_redlander_events.0001.b
		add_courtier = scope:second
		set_designated_heir = scope:second
		scope:second = { set_house = root.house }
		scope:old_heir ?= {
			add_opinion = {
				target = root
				opinion = -40
				modifier = insulted_opinion
			}
			add_opinion = {
				target = scope:second
				opinion = -60
				modifier = hate_opinion
			}
			if = {
				limit = { NOT = { has_relation_rival = scope:second } }
				set_relation_rival = {
					target = scope:second
					reason = rival_over_position
				}
			}
			custom_tooltip = {
				text = gh_redlander_events.0001.claim_tt
				root = {
					every_held_title = {
						limit = { tier > tier_barony }
						scope:old_heir = { add_pressed_claim = prev }
					}
				}
			}
		}
		hidden_effect = { scope:first = { silent_disappearance_effect = yes } }
	}

	option = { # Dismiss both
		name = gh_redlander_events.0001.c
		trigger = { is_ai = no }
		custom_tooltip = {
			text = gh_redlander_events.0001.c.tt
			scope:first = { silent_disappearance_effect = yes }
			scope:second = { silent_disappearance_effect = yes }
		}
		add_piety = 500
	}
}


gh_redlander_events.0002 = { #Great commander arrives bearing two potential slaves or sacrifices...you decide what to do with them...
	type = character_event
	title = gh_redlander_events.0002.t
	desc = gh_redlander_events.0002.desc
	theme = witchcraft
	override_background = { reference = terrain_scope }

	left_portrait = root
	right_portrait = { 
		character = scope:great_commander
	}
	lower_right_portrait = {
		character = scope:incompetent_thrall_1
		animation = severelywounded
	}
	
	lower_center_portrait = {
		character = scope:incompetent_thrall_2
		animation = severelywounded
	}

	trigger = {
		culture = { has_cultural_pillar = heritage_redlander }
	}

	immediate = {
		title:c_digniadartd.title_province = { save_scope_as = background_terrain_scope }
		create_character = {
			template = great_commander
			location = root.capital_province
			save_scope_as = great_commander
		}
		create_character = {
			template = incompetent_thrall
			location = root.capital_province
			save_scope_as = incompetent_thrall_1
		}
		create_character = {
			template = incompetent_thrall
			location = root.capital_province
			save_scope_as = incompetent_thrall_2
		}
	}

	option = { #OPTION A: Sacrifice all 3 to your Dark Gods
		name = gh_redlander_events.0002.a
		scope:incompetent_thrall_1 = { 
			death = { 
				death_reason = death_sacrificed_to_gods
				killer = root
			} 
		}
		scope:incompetent_thrall_2 = { 
			death = { 
				death_reason = death_sacrificed_to_gods
				killer = root
			} 
		}
		scope:great_commander = { 
			death = { 
				death_reason = death_sacrificed_to_gods
				killer = root
			} 
		}
		add_piety = 300
	}
	option = { #OPTION B: recruit the commander, sacrifce the 2 thralls to your Dark Gods
		name = gh_redlander_events.0002.b
		scope:great_commander = {
			add_trait = loyal
		}
		add_courtier = scope:great_commander
		scope:incompetent_thrall_1 = { 
			death = { 
				death_reason = death_sacrificed_to_gods
				killer = root
			} 
		}
		scope:incompetent_thrall_2 = { 
			death = { 
				death_reason = death_sacrificed_to_gods
				killer = root
			} 
		}
		add_hook = {
			target = scope:great_commander
			type = loyalty_hook
		}
		add_piety = 150
	}
	
}

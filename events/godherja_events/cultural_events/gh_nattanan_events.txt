scripted_effect set_nattanan_court_magi_effect = {
	$COURT_MAGI$ = {
		if = {
			limit = { 
				NOT = { liege = $EMPLOYER$ }
			}
			$EMPLOYER$ = { add_courtier = $COURT_MAGI$ }
		}
	}
	$EMPLOYER$ = {
		#Replace existing court magi if one exists
		if = {
			limit = {
				exists = cp:councillor_court_magi
			}
			cp:councillor_court_magi = {
				save_temporary_scope_as = old_court_magi

				$EMPLOYER$ = {
					fire_councillor = scope:old_court_magi
					assign_councillor_type = {
						type = councillor_court_magi
						target = $COURT_MAGI$
					}
					reverse_add_opinion = {
						target = scope:old_court_magi
						modifier = disappointed_opinion
						opinion = -15
					}
					reverse_add_opinion = {
						target = $COURT_MAGI$
						modifier = hired_me_opinion
					}
				}
			}
		}
		#Else just straight up appoint
		else = {
			assign_councillor_type = {
				type = councillor_court_magi
				target = $COURT_MAGI$
			}
			reverse_add_opinion = {
				target = $COURT_MAGI$
				modifier = hired_me_opinion
			}
		}
	}
}

# Child of Thura Myinthein's line gains proper trait (fired on creation and on_birth).
gh_nattanan_events.0008 = { #aka middle_east_decisions.0012
	type = character_event
	hidden = yes

	trigger = {
		OR = {
			AND = {
				exists = mother
				mother = {
					OR = {
						has_trait = thura_myinthein
						has_trait = thura_myinthein_descendant
					}
				}
			}
			AND = {
				exists = father
				father = {
					OR = {
						has_trait = thura_myinthein
						has_trait = thura_myinthein_descendant
					}
				}
			}
			AND = {
				exists = scope:thura_myinthein_progenitor
				OR = {
					is_grandchild_of = scope:thura_myinthein_progenitor
					is_great_grandchild_of = scope:thura_myinthein_progenitor
				}
			}
		}
		NOR = {
			has_trait = thura_myinthein
			has_trait = thura_myinthein_descendant
		}
	}

	immediate = {
		add_trait = thura_myinthein_descendant
		every_child = {
			trigger_event = gh_nattanan_events.0008
		}
	}
}

		}

			text = gh_sarminian_events.1004.b.decision_tooltip

				switch = {
					trigger = has_cultural_tradition

					tradition_braided_oaths = { remove_culture_tradition = tradition_braided_oaths }

					limit = { NOT = { has_cultural_tradition = tradition_steppe_queens } }

			}
		}

gh_sarminian_events.0005 = { # You sear someone
	type = character_event
	content_source = dlc_gh

	desc = {
		desc = gh_sarminian_events.0005.start.desc
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:actor = {
						OR = {
							ai_compassion < 0
							has_relation_rival = scope:recipient
						}
					}
				}
				desc = gh_sarminian_events.0005.cruel.desc
			}
			desc = gh_sarminian_events.0005.compassionate.desc
		}
	}
	theme = prison
	override_background = { reference = dungeon }
	left_portrait = {
		character = root
		animation = schadenfreude
	}
	right_portrait = {
		character = scope:recipient
		animation = prisondungeon
		outfit_tags = { nightgown }
	}

	trigger = { scope:recipient = { is_imprisoned_by = scope:actor } }

	on_trigger_fail = {
		scope:recipient = {
			if = {
				limit = { has_character_flag = is_being_tortured }
				remove_character_flag = is_being_tortured
			}
		}
	}

	immediate = {
		play_music_cue = "mx_cue_prison"
		hidden_effect = {
			save_court_physician_as_effect = { SCOPE_NAME = physician }
			if = {
				limit = { NOT = { exists = scope:physician } }
				random_dummy_gender_physician_effect = yes
				scope:dummy_physician_gender = { save_scope_as = physician }
			}
		}
	}

	option = {
		name = {
			trigger = {
				OR = {
					ai_compassion < 0
					has_relation_rival = scope:recipient
				}
			}
			text = gh_sarminian_events.0005.a
		}
		name = {
			trigger = {
				NOR = {
					ai_compassion < 0
					has_relation_rival = scope:recipient
				}
			}
			text = gh_sarminian_events.0005.b
		}
		show_as_tooltip = { sear_recipient_effect = yes }
		scope:recipient = { trigger_event = gh_sarminian_events.0006 }
	}
}

##################################################
#
# A Friendly Archery Competition Event Chain
#
# Two able-bodied eligible knights approach character
# with an offer to participate in an archery competition.
# If agreed, it is possible to succeed, lose, or bluder.
# In case of a blunder, there exists a change to wound,
# kill, or take out an eye of a courtier.
# Alternatively, player heir can be sent to participate
# instead with similar potential outcomes that the player
# character is informed of post-factum.
#
# gh_sarminian_events.0101 - gh_sarminian_events.0114
#
# by Deadr0x

scripted_trigger gh_sarminian_events_0101_first_trigger = { # Find first valid contestant
	eligible_for_gh_steppe_queenhood = yes
    is_available_healthy_adult = yes
    is_available_at_peace_adult = yes
    prowess >= 7
	is_ai = yes
	save_temporary_scope_as = first
}

scripted_trigger gh_sarminian_events_0101_second_trigger = { # Find second valid contestant
	eligible_for_gh_steppe_queenhood = yes
    is_available_healthy_adult = yes
    is_available_at_peace_adult = yes
    prowess >= 7
	NOT = { this = scope:first }
	is_ai = yes
}

scripted_trigger within_arrowflight = { # Potential woundees
	location = root.capital_province
	is_imprisoned = no
	NOR = {
		this = scope:first
		this = scope:second
	}
	is_ai = yes
}

marcher_lich.0001 = {
	title = marcher_lich.0001.t
	desc = marcher_lich.0001.desc
	type = character_event
	content_source = dlc_gh
	theme = battle
	override_background = {
		reference = lich_raid
	}
	left_portrait = { 
		character = root
		animation = shock
	}
	right_portrait = {
		character = scope:raiding_lich
		animation = celebrate_sword
	}

	trigger = {
		NOT = { has_character_flag = raided_by_lich }
		any_sub_realm_county = {
			any_title_to_title_neighboring_and_across_water_county = {
				title_province = { geographical_region = lichdoms_biozone }
			}
		}
	}
	immediate = {
		random_ruler = {
			limit = {
				is_ai = yes
				has_culture = culture:lich_culture
				has_faith = faith:rejection
			}
			save_scope_as = raiding_lich
		}
		random_sub_realm_county = {
			limit = {
				any_title_to_title_neighboring_and_across_water_county = {
					title_province = { geographical_region = lichdoms_biozone }
				}
			}
			save_scope_as = county_raided_by_lich
		}
		add_character_flag = raided_by_lich
	}

	option = {
		name = marcher_lich.0001.a
		scope:county_raided_by_lich = {
			change_development_level = -3
		}
	}
	option = {
		name = marcher_lich.0001.b
		trigger = {
			stewardship >= 12
			is_available_adult_or_is_commanding = yes
		}
		remove_short_term_gold = medium_gold_value
	}
	option = {
		name = marcher_lich.0001.c
		trigger = {
			martial >= 12
			is_available_adult_or_is_commanding = yes
		}
		configure_start_single_combat_effect = {
			SC_INITIATOR = root
			SC_ATTACKER = root
			SC_DEFENDER = scope:raiding_lich
			FATALITY = possible
			FIXED = no
			LOCALE = terrain_scope
			OUTPUT_EVENT = marcher_lich.0002
			INVALIDATION_EVENT = marcher_lich.0002
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = 1
				ai_greed = 0.25
				ai_honor = -0.5
			}
		}
	}
}

marcher_lich.0002 = {
	type = character_event
	hidden = yes

	immediate = {
		send_interface_toast = {
			title = marcher_lich.0002.t
			left_icon = scope:sc_victor
			right_icon = scope:sc_loser
			scope:sc_victor = { add_prestige = medium_prestige_gain }
			scope:sc_loser = { 
				add_stress = major_stress_impact_gain 
			}
			scope:sc_victor = {
				if = {
					limit = { 
						is_undead = no
						has_dlc_feature = legends
					}
					create_legend_seed = {
						type = heroic
						quality = famed
						chronicle = legendary_battle
						properties = {
							location = scope:county_raided_by_lich.title_province
							winner = scope:sc_victor
							loser = scope:sc_loser
						}
					}
				}
			}
		}
	}
}

marcher_lich.1001 = {
	type = character_event
	title = marcher_lich.1001.t
	desc = marcher_lich.1001.desc
	content_source = dlc_gh
	theme = battle
	override_background = {
		reference = lich_raid
	}
	left_portrait = {
		character = root
		animation = aggressive_sword
	}
	right_portrait = {
		character = scope:defending_lich
		animation = anger
	}
	lower_left_portrait = {
		trigger = {
			exists = scope:great_champion
		}
		character = scope:great_champion
		animation = anger
	}
	lower_center_portrait = {
		trigger = {
			exists = scope:great_magi
		}
		character = scope:great_magi
		animation = anger
	}

	immediate = {
		if = {
			limit = { exists = scope:previous_controller }
			scope:previous_controller = { save_scope_as = defending_lich }
		}
		if = {
			limit = { exists = scope:county }
			scope:county = { save_scope_as = lich_county }
		}
		random_knight = {
			limit = {
				can_start_single_combat_trigger = yes
				prowess > average_skill_rating
			}
			save_scope_as = great_champion
		}
		if = {
			limit = {
				any_knight = {
					can_start_single_combat_trigger = yes
					can_access_magic = yes
					prowess > average_skill_rating
					NOT = { this = scope:great_champion }
				}
			}
			random_knight = {
				limit = {
					NOT = { this = scope:great_champion }
					can_start_single_combat_trigger = yes
					can_access_magic = yes
					prowess > average_skill_rating
				}
				save_scope_as = great_magi
			}
		}
	}

	option = { # It's time to D-D-D-DUEL, do it yourself
		name = marcher_lich.1001.a
		trigger = {
			is_available_adult_or_is_commanding = yes
		}
		add_internal_flag = dangerous
		custom_tooltip = marcher_lich.1001.potential_artifacts.tt

			SC_INITIATOR = root
			SC_ATTACKER = root

			FIXED = no
			LOCALE = terrain_scope
			OUTPUT_EVENT = marcher_lich.1002
			INVALIDATION_EVENT = marcher_lich.1002
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = 1
				ai_greed = 0.25
				ai_honor = -0.5
			}
		}
	}

	option = { # Make your knight fight them instead!
		name = marcher_lich.1001.b
		trigger = {
			exists = scope:great_champion
		}
		add_internal_flag = dangerous
		custom_tooltip = marcher_lich.1001.potential_artifacts.tt
		configure_start_single_combat_effect = {
			SC_INITIATOR = root
			SC_ATTACKER = scope:great_champion
			SC_DEFENDER = scope:defending_lich
			FATALITY = always
			FIXED = no
			LOCALE = terrain_scope
			OUTPUT_EVENT = marcher_lich.1002
			INVALIDATION_EVENT = marcher_lich.1002
		}
		ai_chance = {
			base = 100
			ai_value_modifier = {
				ai_boldness = 1
				ai_greed = 0.25
				ai_honor = -0.5
			}
		}
	}

	option = { # Make your Magi fight them instead!!
		name = marcher_lich.1001.c
		trigger = {
			exists = scope:great_magi
		}
		add_internal_flag = special
		custom_tooltip = marcher_lich.1001.potential_artifacts.tt

		name = marcher_lich.1001.d
		duel = {
			skill = prowess
			value = average_skill_rating
			80 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = medium_positive_duel_skill_multiplier
				}
				desc = marcher_lich.1001.successful.desc

				show_as_tooltip = {
					add_prestige = medium_prestige_gain
					custom_tooltip = marcher_lich.1001.potential_artifacts.tt
				}

				hidden_effect = {
					trigger_event = marcher_lich.1003
				}
			}

			20 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = low_negative_duel_skill_multiplier 
				}
				desc = marcher_lich.1001.failure.desc

				show_as_tooltip = {
					add_prestige = medium_prestige_loss
					custom_tooltip = marcher_lich.1001.wounded.tt
				}

				hidden_effect = {
					send_interface_toast = {
						title = marcher_lich.1001.failure.desc
						left_icon = root

						add_prestige = medium_prestige_loss
						increase_wounds_effect = { REASON = fight }
						if = {
							limit = { exists = scope:war }
							scope:war = {
								end_war = white_peace
							}
						}
					}
				}
			}
		}
	}

	option = { # Perhaps this was too big a risk, grab what we can and scram!
		name = marcher_lich.1001.e
		loot_necropolis_effect_gold = yes
		add_prestige = medium_prestige_loss
		if = {
			limit = { exists = scope:war }
			scope:war = {
				end_war = white_peace
			}
		}
	}
}

marcher_lich.1002 = { # Sends out victory/loss for the Necropolis Fight
	type = character_event
	hidden = yes

	immediate = {
		send_interface_toast = {
			title = marcher_lich.1002.t
			left_icon = scope:sc_victor
			right_icon = scope:sc_loser
			scope:sc_victor = {
				add_prestige = medium_prestige_gain
				loot_necropolis_effect_gold = yes
				loot_necropolis_effect_artifact = yes
				GH_achiev_whatisdeadmayyetdie_check = yes
				if = {
					limit = { 
						is_undead = no
						has_any_nickname = no
					}
					set_nickname_effect = { NICKNAME = nick_slayer_of_liches }
					if = {
						limit = { exists = scope:lich_county }
						spawn_ruin = { COUNTY = scope:lich_county }
					}
				}
				if = {
					limit = { 
						is_undead = no
						has_dlc_feature = legends
					}
					create_legend_seed = {
						type = heroic
						quality = famed
						chronicle = legendary_battle
						properties = {
							location = scope:lich_county.title_province
							winner = scope:sc_victor
							loser = scope:sc_loser
						}
					}
				}
			}
			if = {
				limit = { exists = scope:war }
				scope:war = {
					end_war = invalidated
				}
			}
		}
	}
}

			}
		}

			death = { death_reason = death_battle }
		}
	}

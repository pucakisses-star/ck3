namespace = marcher_lifepath

marcher_lifepath.0001 = { #on maturity, choose a path
	title = marcher_lifepath.0001.t
	type = character_event
	content_source = dlc_gh
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					age > 16
				}
				desc = marcher_lifepath.0001.desc.conversion
			}
			desc = marcher_lifepath.0001.desc
		}
	}
	theme = faith
	left_portrait = { 
		character = root
	}
	
	trigger = {
		faith = {
			has_variable = faith_has_lifepaths
		}
		NOR = {
			has_trait = way_of_swords_follower
			has_trait = way_of_family_follower
			has_trait = way_of_coins_follower
			has_trait = way_of_harvests_follower
			has_trait = way_of_justice_follower
			has_trait = way_of_knowledge_follower
			has_trait = way_of_crows_follower
			has_trait = way_of_crowns_follower
			has_trait = way_of_nullification_follower
			has_trait = way_of_hunt_follower
			has_trait = way_of_sword_follower
			has_trait = way_of_stars_follower
			has_trait = way_of_shepherd_follower
			has_trait = way_of_blood_follower
			has_trait = way_of_pain_follower
			has_trait = way_of_fear_follower
			has_trait = way_of_dread_follower
			has_trait = way_of_peace_follower
			has_trait = way_of_war_follower
			has_trait = way_of_death_follower
			root.faith = { has_doctrine_parameter = sjalvolki_lifepaths }
			root.faith = { has_doctrine_parameter = evil_lifepaths }
			root.faith = { has_doctrine_parameter = tripath_lifepaths }
		}
	}	

	option = {
		name = marcher_lifepath.0001.a
		trigger = {
			is_valid_lifepath_trigger = { LIFEPATH = way_of_swords }
		}
		custom_tooltip = marcher_lifepath.0001.a.tt
		add_trait = way_of_swords_follower
		ai_chance = { base = 12 }
	}
	option = {
		name = marcher_lifepath.0001.b
		trigger = {
			is_valid_lifepath_trigger = { LIFEPATH = way_of_family }
		}
		custom_tooltip = marcher_lifepath.0001.b.tt
		add_trait = way_of_family_follower
		ai_chance = { base = 10 }
	}
	option = {
		name = marcher_lifepath.0001.c
		trigger = {
			is_valid_lifepath_trigger = { LIFEPATH = way_of_coins }
		}
		custom_tooltip = marcher_lifepath.0001.c.tt
		add_trait = way_of_coins_follower
		ai_chance = { base = 12 }
	}
	option = {
		name = marcher_lifepath.0001.d
		trigger = {
			is_valid_lifepath_trigger = { LIFEPATH = way_of_harvests }
		}
		custom_tooltip = marcher_lifepath.0001.d.tt
		add_trait = way_of_harvests_follower
		ai_chance = { base = 4 }
	}
	option = {
		name = marcher_lifepath.0001.f
		trigger = {
			is_valid_lifepath_trigger = { LIFEPATH = way_of_crowns }
		}
		custom_tooltip = marcher_lifepath.0001.f.tt
		add_trait = way_of_crowns_follower
		ai_chance = { base = 20 }
	}
	option = {
		name = marcher_lifepath.0001.e
		custom_tooltip = marcher_lifepath.0001.e.tt
		trigger_event = marcher_lifepath.0002
		ai_chance = { base = 18 }
	}
}

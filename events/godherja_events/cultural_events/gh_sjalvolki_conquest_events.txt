# Call to arms to a vassal
sjalvolki_conquest.0001 = {
	type = character_event
	content_source = dlc_gh
	theme = war
	title = sjalvolki_conquest.0001.t
	desc = sjalvolki_conquest.0001.desc
	
	trigger = {
		NOT = {
			has_character_flag = aversarian_migration
		}
		OR = {
			is_landed = yes
			is_imperrech_landless_adventurer = yes	
		}
	}
	
	override_background = {
		reference = throne_room_tribal
	}
	
	left_portrait = {
		character = scope:attacker
		animation = war_attacker
	}
	
	right_portrait = {
		character = root
		animation = personality_bold
	}
	widget = {
		gui = "event_window_widget_migration"
		container = "custom_widgets_container"
	}
	
	immediate = {
		if = {
			limit = {
				NOT = {
					has_variable = sjalvolki_conquest_contribution
				}
			}
			set_variable = {
				name = sjalvolki_conquest_contribution
				value = 0
			}
		}
		if = {
			limit = {
				NOT = {
					has_variable = sjalvolki_conquest_tribe_size
				}
			}
			set_variable = {
				name = sjalvolki_conquest_tribe_size
				value = 14
			}
		}
		if = {
			limit = {
				highest_held_title_tier = tier_county
				NOT = {
					has_character_flag = reduced_tribe_size
				}
			}
			change_variable = {
				name = sjalvolki_conquest_tribe_size
				subtract = 5
			}
			add_character_flag = reduced_tribe_size
		}
		if = {
			limit = {
				highest_held_title_tier = tier_duchy
				NOT = {
					has_character_flag = reduced_tribe_size
				}
			}
			change_variable = {
				name = sjalvolki_conquest_tribe_size
				subtract = 3
			}
			add_character_flag = reduced_tribe_size
		}
		if = {
			limit = {
				is_imperrech_landless_adventurer = yes
				NOT = { has_character_flag = imperrech_landless_adventurer }
			}
			add_character_flag = imperrech_landless_adventurer
		}
	}
	
	option = {	# Gather up troops and fight yourself!
		name = sjalvolki_conquest.0001.a
		trigger = { 
			NOR = { 
				has_character_flag = holdover_in_old_lands 
				has_character_flag = created_for_migration
			}
			trigger_if = {
				limit = { is_landless_adventurer = yes }
				domicile = {
					provisions >= medium_provisions_value
				}
			}
		}
		custom_tooltip = risky_tt
		custom_tooltip = current_contribution_tt
		show_as_unavailable = {
			always = yes
		}
		if = {
			limit = { var:sjalvolki_conquest_tribe_size <= 0 }
			remove_short_term_gold = medium_gold_value
		}
		if = {
			limit = { is_landless_adventurer = yes }
			domicile = {
				change_provisions = medium_provisions_loss
			}
		}
		else = {
			custom_tooltip = saps_development_tt
			change_variable = {
				name = sjalvolki_conquest_tribe_size
				subtract = 1
			}
		}
		add_prestige = sjalvolki_conquest_prestige_from_gathering_troops_value
		if = {
			limit = { exists = dynasty }
			dynasty = {
				add_dynasty_prestige = prev.sjalvolki_conquest_contribution_from_gathering_troops_value
			}
		}
		scope:attacker = {
			if = {
				limit = { NOT = { has_variable = sjalvolki_conquest_total_host_size } }
				set_variable = {
					name = sjalvolki_conquest_total_host_size
					value = root.sjalvolki_conquest_gathered_troops_number_value
				}
			}
			else = {
				change_variable = {
					name = sjalvolki_conquest_total_host_size
					add = root.sjalvolki_conquest_gathered_troops_number_value
				}
			}
		}
		change_variable = {
			name = sjalvolki_conquest_contribution
			add = sjalvolki_conquest_contribution_from_gathering_troops_value
		}
		ai_chance = {
			base = 90
			modifier = {
				factor = 5
				has_trait = ambitious
			}
			modifier = {
				factor = 0.2
				has_trait = craven
			}
			modifier = {
				factor = 0.1
				OR = {
					culture = { has_cultural_pillar = heritage_aversarian } # Aversarians should be less likely to seek fame and glory against Aversarians, generally
					religion = religion:aversarinas_aagiokrata_religion
				}
			}
			modifier = {
				factor = 0
				AND = {
					has_trait = content	# Content characters will only join out of a sense of duty
					NOR = {
						has_trait = just
						has_trait = honest
						has_relation_friend = scope:attacker
					}
				}
			}
			modifier = { # They should be less interested in moving farther, generally
				factor = 0.25
				has_character_flag = sjalvolki_vanguard
			}
			modifier = { # Clanlanders shouldnt be interested 
				factor = 0
				culture = { has_cultural_pillar = heritage_clanlander }
			}
			ai_value_modifier = {
				ai_boldness = 0.5
			}
		}
		scope:war = {
			if = {
				limit = { 
					root.highest_held_title_tier = tier_kingdom
					NOT = { root = { is_participant_in_war = scope:war } }
				}
				add_attacker = ROOT
			}
		}
		trigger_event = {
			on_action = sjalvolki_conquest_events
			days = { 15 30 }
		}
	}
	
	option = {	# Not interested at the moment
		name = sjalvolki_conquest.0001.b
		trigger = { 
			NOR = { 
				has_character_flag = holdover_in_old_lands 
				has_character_flag = created_for_migration
			}
			
		}
		ai_chance = {	# AI will never pick this unless they have legitimate temporary circumstances that make participating difficult
			base = 0
			modifier = {
				add = 150
				is_pregnant = yes
			}
			modifier = {
				add = 300
				has_trait = wounded_3
			}
			modifier = {
				add = 200
				has_trait = wounded_2
			}
			modifier = {
				add = 100
				has_trait = wounded_1
			}
			modifier = {
				add = 50
				has_trait = ill
			}
			modifier = {	# Ambitious characters are extremely likely to stay in the conquest until they get a kingdom
				factor = 0.5
				has_trait = ambitious
				NOR = {
					highest_held_title_tier >= tier_kingdom
					capital_county = { is_sjalvolki_kingdom_capital = yes }
				}
			}
			modifier = {
				add = 100
				is_landless_adventurer = yes
				domicile = {
					provisions < medium_provisions_value
				}
			}
		}
	}
	
	option = {	# Not interested, ever
		name = sjalvolki_conquest.0001.c
		custom_tooltip = no_more_sjalvolki_conquest_tt
		add_character_flag = aversarian_migration
		every_child = { add_character_flag = aversarian_migration }
		if = {
			limit = {
				is_landed = yes
			}
			if = {
				limit = { 
					var:sjalvolki_conquest_tribe_size > 0 
					trigger_if = {
						limit = { exists = capital_county }
						NOT = { capital_county = { has_county_modifier = gh_sjalvolki_resettlement_modifier } }
					}
				}
				capital_county = {
					change_development_level = prev.var:sjalvolki_conquest_tribe_size
					add_county_modifier = {
						modifier = gh_sjalvolki_resettlement_modifier
						years = 40
					}
				}
			}
			else = {
				capital_county = {
					add_county_modifier = {
						modifier = gh_sjalvolki_resettlement_modifier
						years = 30
					}
				}
			}
			if = {
				limit = {
					is_ai = yes
					exists = capital_county
					NOT = { culture = { has_cultural_pillar = heritage_clanlander } }
				}
				add_character_modifier = {
					modifier = gh_sjalvolki_resettlement_focus_modifier
					years = 30
				}
				root.capital_county = {
					set_county_culture = root.culture
					set_county_faith = root.faith
				} 
			}
			if = { # We want Clanlanders to get a different modifier so they're able to access Vassalization wars more
				limit = {
					is_ai = yes
					exists = capital_county
					culture = { has_cultural_pillar = heritage_clanlander }
				}
				add_character_modifier = {
					modifier = gh_sjalvolki_rebuilding_focus_modifier
					years = 30
				}
				root.capital_county = {
					set_county_culture = root.culture
					set_county_faith = root.faith
				} 
			}
		}
		ai_chance = {
			base = 10
			modifier ={
				factor = 0
				years_from_game_start <= 10
				NOR = {
					has_character_flag = already_received_lands
					has_character_flag = created_for_migration
				}
			}
			modifier = {
				factor = 0
				AND = {
					highest_held_title_tier >= tier_kingdom
					years_from_game_start <= 12
				}
				NOR = {
					has_character_flag = already_received_lands
					has_character_flag = created_for_migration
					has_character_flag = sjalvolki_vanguard
				}
			}
			modifier = {
				factor = 0
				AND = {
					has_religion = religion:sjalvolki_religion
					years_from_game_start <= 7
				}
				NOR = {
					has_character_flag = already_received_lands
					has_character_flag = created_for_migration
					has_character_flag = sjalvolki_vanguard
				}
			}
			modifier = {
				factor = 50
				any_parent = {
					has_character_flag = already_received_lands
				}
			}
			modifier = {	# Ambitious characters are extremely likely to stay in the conquest until they get a kingdom
				factor = 0.5
				has_trait = ambitious
				NOR = {
					highest_held_title_tier >= tier_kingdom
					capital_county = { is_sjalvolki_kingdom_capital = yes }
				}
			}
			modifier = {	# Characters become drastically more likely to drop out after receiving land unless motivated by glory or duty
				factor = 2
				has_character_flag = already_received_lands
				NOR = {
					AND = {
						has_trait = arrogant
						OR = {
							has_trait = brave
							has_trait = ambitious
						}
					}
					AND = {
						has_trait = just
						has_trait = honest
					}
					has_relation_friend = scope:attacker
				}
			}
			modifier = {
				factor = 4
				capital_province = {
					OR = {
						GH_any_farmlands_terrain_trigger = yes
						GH_any_floodplains_terrain_trigger = yes
						GH_any_oasis_terrain_trigger = yes
					}
				}
			}
			modifier = {
				factor = 0.2
				capital_province = {
					OR = {
						GH_any_desert_terrain_trigger = yes
						GH_any_steppe_terrain_trigger = yes
						GH_any_mountains_terrain_trigger = yes
						GH_any_wetlands_terrain_trigger = yes
					}
				}
			}
			modifier = {
				factor = 100
				capital_province = {
					has_special_building = yes
				}
			}
			modifier = { # Vanguard are likely to stick where they start
				factor = 100 
				OR = {
					has_character_flag = sjalvolki_vanguard
					liege = { has_title = title:k_fenvir } # just in case
				}
			}
			modifier = {
				factor = 100
				OR = {
					culture = { has_cultural_pillar = heritage_aversarian } # Aversarians should be less likely to seek fame and glory against Aversarians, generally
					religion = religion:aversarinas_aagiokrata_religion
				}
			}
			modifier = {
				factor = 100
				OR = { # Clanlanders should be less likely to seek fame and glory, they already got their land back
					culture = { has_cultural_pillar = heritage_clanlander }
					religion = religion:clanlander_religion
				}
			}
		}
	}
}

# Hidden event for Cenware to get his troops
sjalvolki_conquest.0002 = {
	hidden = yes
	
	immediate = {
		if = {
			limit = { 
				has_variable = sjalvolki_conquest_total_host_size
				is_at_war = yes
			}
			random_character_war = {
				save_scope_as = imperrech_war
			}
			while = {
				limit = {
					root.var:sjalvolki_conquest_total_host_size > 12000
				}
				change_variable = {
					name = sjalvolki_conquest_total_host_size
					subtract = 12000
				}
				spawn_army = {
					name = "IMPERRECH_HOST"
					men_at_arms = {
						type = armored_footmen
						men = 12000
					}
					location = root.capital_province
					war = scope:imperrech_war
				}
			}
			spawn_army = {
				name = "IMPERRECH_HOST"
				men_at_arms = {
					type = armored_footmen
					men = root.var:sjalvolki_conquest_total_host_size
				}
				location = root.capital_province
				war = scope:imperrech_war
			}
			debug_log = "IMPERRECH HOST SIZE: [THIS.Var('sjalvolki_conquest_total_host_size').GetValue|0]"
		}
		set_variable = {
			name = sjalvolki_conquest_total_host_size
			value = 0
		}
	}
}

# Offer to vanguard rulers to swear fealty
sjalvolki_conquest.0003 = {
	type = character_event
	content_source = dlc_gh
	theme = war
	title = sjalvolki_conquest.0003.t
	desc = sjalvolki_conquest.0003.desc
	
	override_background = {
		reference = throne_room_tribal
	}
	
	left_portrait = {
		character = scope:attacker
		animation = war_attacker
	}

	right_portrait = {
		character = root
		animation = war_over_win
	}
	
	option = {	# Swear fealty
		name = sjalvolki_conquest.0003.a
		if = {
			limit = {
				trigger_if = {
					limit = { exists = scope:war }
					scope:war = {
						any_war_defender = {
							this = root
						}
					}
				}
			}
			scope:war = {
				remove_participant = root
			}
		}
		set_variable = sjalvolki_vanguard
		create_title_and_vassal_change = {
			type = swear_fealty
			save_scope_as = title_change
			add_claim_on_loss = no
		}
		change_liege = {
			liege = scope:attacker
			change = scope:title_change
		}
		resolve_title_and_vassal_change = scope:title_change
		custom_tooltip = sjalvolki_conquest_vanguard_tt
		every_held_title = { # Migrate the Vanguard's culture into their personally held counties
			limit = { tier = tier_county }
			set_county_culture = root.culture
			set_county_faith = root.faith
		}
		every_vassal = {
			limit = { highest_held_title_tier >= tier_county }
			trigger_event = sjalvolki_conquest.0014
		}
		ai_chance = {
			base = 100
			modifier = {	# Some characters will never beg for mercy
				factor = 0
				character_behavior_wont_beg_for_mercy_trigger = yes
			}
		}
	}
	
	option = {	# Die on this hill, for some reason
		name = sjalvolki_conquest.0003.b
		
		if = {
			limit = {
				trigger_if = {
					limit = { exists = scope:war }
					scope:war = {
						NOT = {
							any_war_defender = {
								this = root
							}
						}
					}
				}
			}
			scope:war = {
				add_defender = prev
			}
		}
		ai_chance = {
			base = 0
			modifier = {	# Some characters will never beg for mercy
				factor = 100
				character_behavior_wont_beg_for_mercy_trigger = yes
			}
		}
	}
}

				factor = 0
				has_title = title:k_scarlet_chosen

			modifier = { # Sadists should generally not want to accept barbarian rule
				factor = 3
				has_trait = sadistic
			}
		}
	}

	option = { # EP3: Let's get out of dodge!
		name = sjalvolki_conquest.0004.c
		trigger = {
			has_ep3_dlc_trigger = yes # gotta own RtP
			is_valid_for_laampdom = yes
		}
		save_scope_as = adventurer
		if = {
			limit = {
				player_heir ?= {
					NOT = { this = root }
					is_adult = yes
				}
			}
			player_heir = {
				save_scope_as = laamp_heir
				save_scope_as = laamp_inheritor
				custom_tooltip = become_landless_adventurer_decision_title_scope_tt
			}
		}
		else_if = {
			limit = { is_independent_ruler = no }

				save_scope_as = laamp_liege
				save_scope_as = laamp_inheritor
				custom_tooltip = become_landless_adventurer_decision_title_scope_tt
			}
		}
		else = {
			custom_tooltip = become_landless_adventurer_decision_title_created_tt
			if = {
				limit = {
					NOT = { exists = scope:laamp_inheritor }
				}

					create_character = {
						template = laamp_inheritor_template
						culture = root.capital_county.culture
						faith = root.capital_county.faith
						dynasty = generate
						location = root.capital_province
						after_creation = {
							save_scope_as = laamp_inheritor
							add_character_flag = new_laamp_inheritor

				}

		show_as_tooltip = {
			add_prestige = medium_prestige_gain
			add_character_modifier = {
				modifier = ep3_voluntary_laamp_character_modifier
				years = 5
			}
		}
		trigger_event = { id = ep3_laamps.0030 }
		# Debug tracking
		if = {
			limit = {
				is_ai = yes
				debug_only = yes
			}
			if = {
				limit = { exists = global_var:ai_voluntary_laamp_count }
				change_global_variable = {
					name = ai_voluntary_laamp_count
					add = 1
				}
			}
			else = {
				set_global_variable = {
					name = ai_voluntary_laamp_count
					value = 1
				}
			}
			debug_log = "AI became adventurer voluntarily"
			debug_log_scopes = yes
		}
		stress_impact = {
			craven = minor_stress_impact_gain
			content = minor_stress_impact_gain
			lazy = minor_stress_impact_gain
		}

		ai_chance = {
			base = 0
		}
	}
}

sjalvolki_conquest.0005 = { # Event for Cenware when he receives an offer of submission from an Aversarian
	type = character_event
	content_source = dlc_gh
	theme = war
	title = sjalvolki_conquest.0005.t
	desc = sjalvolki_conquest.0005.desc
	
	override_background = {
		reference = throne_room_tribal
	}
	
	left_portrait = {
		character = root
		animation = war_attacker
	}

	right_portrait = {
		character = scope:aversarian_mercy_requester
		animation = beg
	}
	
	option = {	# Give refugee state
		name = sjalvolki_conquest.0005.a
		trigger = {
			OR = {
				scope:aversarian_mercy_requester = { 
					highest_held_title_tier >= tier_duchy 
					OR = {
						AND = {
							OR = {
								has_trait = just
								has_trait = compassionate
							}
							prestige_level >= 3
						}
						is_ai = no
					}
					NOR = {
